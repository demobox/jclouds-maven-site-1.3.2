44  @Test(groups = "live", singleThreaded = true, testName = "OfferingClientLiveTest")

47     public void testListDiskOfferings() throws Exception {
48        Set<DiskOffering> response = client.getOfferingClient().listDiskOfferings();
49        assert null != response;
50        long offeringCount = response.size();
51        assertTrue(offeringCount >= 0);
52        for (DiskOffering offering : response) {
53           try {
54             DiskOffering newDetails = Iterables.getOnlyElement(client.getOfferingClient().listDiskOfferings(
55                 ListDiskOfferingsOptions.Builder.id(offering.getId())));
56             assertEquals(offering, newDetails);
57             assertEquals(offering, client.getOfferingClient().getDiskOffering(offering.getId()));
58             assert offering.getId() > 0 : offering;
59             assert offering.getName() != null : offering;
60             assert offering.getCreated() != null : offering;
61             assert offering.getDisplayText() != null : offering;
62             assert offering.getDiskSize() > 0 || (offering.getDiskSize() == 0 && offering.isCustomized()) : offering;
63             assert offering.getTags() != null : offering;
64  
65           } catch (NoSuchElementException e) {
66              // This bug is present both in 2.2.8 and 2.2.12
67             assertTrue("2.2.8".equals(apiVersion) || "2.2.12".equals(apiVersion));
68           }
69        }
70     }
