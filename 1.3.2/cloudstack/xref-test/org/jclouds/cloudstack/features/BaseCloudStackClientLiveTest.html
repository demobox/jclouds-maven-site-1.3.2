125    public static long getTemplateForZone(CloudStackClient client, long zoneId) {
126       // TODO enum, as this is way too easy to mess up.
127       Set<String> acceptableCategories = ImmutableSet.of("Ubuntu", "CentOS");
128 
129       final Predicate<Template> hypervisorPredicate = new CorrectHypervisorForZone(client).apply(zoneId);
130       final Predicate<Template> osTypePredicate = new OSCategoryIn(client).apply(acceptableCategories);
131 
132       @SuppressWarnings("unchecked")
133       Predicate<Template> templatePredicate = Predicates.<Template> and(TemplatePredicates.isReady(),
134             hypervisorPredicate, osTypePredicate);
135       Iterable<Template> templates = filter(
136             client.getTemplateClient().listTemplates(ListTemplatesOptions.Builder.zoneId(zoneId)), templatePredicate);
137       if (Iterables.any(templates, TemplatePredicates.isPasswordEnabled())) {
138          templates = filter(templates, TemplatePredicates.isPasswordEnabled());
139       }
140       if (Iterables.size(templates) == 0) {
141          throw new NoSuchElementException(templatePredicate.toString());
142       }
143       long templateId = get(templates, 0).getId();
144       return templateId;
145    }

196    public void setupClient() {
197       setupCredentials();

199       computeContext = CloudStackContext.class.cast(new ComputeServiceContextFactory(setupRestProperties()).

223          globalAdminContext = globalAdminComputeContext.getGlobalContext();
224          globalAdminClient = globalAdminContext.getApi();

230       sshFactory = injector.getInstance(SshClient.Factory.class);
231       socketTester = new RetryablePredicate<IPSocket>(new InetSocketAddressConnect(), 180, 1, 1, TimeUnit.SECONDS);

237       virtualMachineRunning = new RetryablePredicate<VirtualMachine>(new VirtualMachineRunning(client), 600, 5, 5,
238             TimeUnit.SECONDS);

240       adminVirtualMachineRunning = new RetryablePredicate<VirtualMachine>(new VirtualMachineRunning(adminClient), 600,
241             5, 5, TimeUnit.SECONDS);
242       injector.injectMembers(adminVirtualMachineRunning);
243       virtualMachineDestroyed = new RetryablePredicate<VirtualMachine>(new VirtualMachineDestroyed(client), 600, 5, 5,
244             TimeUnit.SECONDS);
245       injector.injectMembers(virtualMachineDestroyed);
246       adminVirtualMachineDestroyed = new RetryablePredicate<VirtualMachine>(new VirtualMachineDestroyed(adminClient),
247             600, 5, 5, TimeUnit.SECONDS);

254    protected static User verifyCurrentUserIsOfType(
255          RestContext<? extends CloudStackClient, ? extends CloudStackAsyncClient> context, Account.Type type) {
256       Iterable<User> users = Iterables.concat(context.getApi().getAccountClient().listAccounts());
257       Predicate<User> apiKeyMatches = UserPredicates.apiKeyEquals(context.getIdentity());
258       User currentUser;
259       try {
260          currentUser = Iterables.find(users, apiKeyMatches);
261       } catch (NoSuchElementException e) {
262          throw new NoSuchElementException(String.format("none of the following users match %s: %s", apiKeyMatches,
263                users));
264       }
265 
266       if (currentUser.getAccountType() != type) {
267          Logger.getAnonymousLogger().warning(
268                String.format("Expecting an user with type %s. Got: %s", type.toString(), currentUser.toString()));
269       }
270       return currentUser;
271    }
