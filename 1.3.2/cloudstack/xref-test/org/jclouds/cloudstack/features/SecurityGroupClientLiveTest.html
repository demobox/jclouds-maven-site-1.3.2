187    protected void checkGroup(SecurityGroup group) {
188       // http://bugs.cloud.com/show_bug.cgi?id=8968
189       if (group.getIngressRules().size() <= 1)
190          assertEquals(group, client.getSecurityGroupClient().getSecurityGroup(group.getId()));

193       assert group.getAccount() != null : group;
194       assert group.getDomain() != null : group;
