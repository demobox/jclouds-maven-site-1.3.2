332    private Date makeDate(int year, int month, int date, int hour, int minute, int second, String timeZoneName) {
333       Calendar cal = Calendar.getInstance(TimeZone.getTimeZone(timeZoneName));
334       cal.set(Calendar.YEAR, year);
335       cal.set(Calendar.MONTH, month);
336       cal.set(Calendar.DATE, date);
337       cal.set(Calendar.HOUR_OF_DAY, hour);
338       cal.set(Calendar.MINUTE, minute);
339       cal.set(Calendar.SECOND, second);
340       cal.set(Calendar.MILLISECOND, 0);
341       return cal.getTime();
342    }
