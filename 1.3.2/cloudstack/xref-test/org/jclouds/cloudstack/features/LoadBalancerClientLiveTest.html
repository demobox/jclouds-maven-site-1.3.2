104    @Test(dependsOnMethods = "testCreateVm")
105    public void testCreateLoadBalancerRule() throws Exception {
106       if (networksDisabled)
107          return;
108       int attempts = 0;
109       while (rule == null && attempts < 10) {
110          ip = reuseOrAssociate.apply(network);
111          try {
112             Long jobId = client.getLoadBalancerClient().createLoadBalancerRuleForPublicIP(ip.getId(),
113                   Algorithm.LEASTCONN, prefix, 22, 22);
114             assertTrue(jobComplete.apply(jobId));
115             AsyncJob<LoadBalancerRule> asyncJob = client.getAsyncJobClient().getAsyncJob(jobId);
116             LoadBalancerRule result = asyncJob.getResult();
117             rule = result;
118          } catch (IllegalStateException e) {
119             // very likely an ip conflict, so retry;
120             attempts++;
121          }
122       }
123       assertNotNull(rule, "Failed to get a load balancer rule after " + attempts + " attempts");
124       assert (rule.getPublicIPId() == ip.getId()) : rule;
125       assertEquals(rule.getPublicPort(), 22);
126       assertEquals(rule.getPrivatePort(), 22);
127       assertEquals(rule.getAlgorithm(), Algorithm.LEASTCONN);
128       assertEquals(rule.getName(), prefix);
129       assertEquals(rule.getState(), State.ADD);
130       assertEquals(client.getLoadBalancerClient().listVirtualMachinesAssignedToLoadBalancerRule(rule.getId()).size(), 0);
131       checkRule(rule);
132 
133    }

157       for (int i = 0; i < 5; i++) {// retry loop TODO replace with predicate.

168          }

197    public void testListLoadBalancerRules() throws Exception {
198       Set<LoadBalancerRule> response = client.getLoadBalancerClient().listLoadBalancerRules();
199       assert null != response;
200       assertTrue(response.size() >= 0);
201       for (LoadBalancerRule rule : response) {
202          LoadBalancerRule newDetails = findRuleWithId(rule.getId());
203          assertEquals(rule.getId(), newDetails.getId());
204          checkRule(rule);
205       }
206    }

208    private LoadBalancerRule findRuleWithId(final long id) {
209       return find(client.getLoadBalancerClient().listLoadBalancerRules(), new Predicate<LoadBalancerRule>() {
210 
211          @Override
212          public boolean apply(LoadBalancerRule arg0) {
213             return arg0.getId() == id;
214          }
215 
216       });
217    }

219    protected void checkRule(LoadBalancerRule rule) {
220       assertEquals(rule.getId(), findRuleWithId(rule.getId()).getId());
221       assert rule.getId() > 0 : rule;
222       assert rule.getAccount() != null : rule;
223       assert rule.getAlgorithm() != null : rule;
224       assert rule.getPrivatePort() > 0 : rule;
225       assert rule.getPublicPort() > 0 : rule;
226       assert rule.getDomain() != null : rule;
227       assert rule.getDomainId() > 0 : rule;
228       assert rule.getState() != null : rule;
229       assert rule.getName() != null : rule;
230       assert rule.getPublicIP() != null : rule;
231       assert rule.getPublicIPId() > 0 : rule;
232    }
