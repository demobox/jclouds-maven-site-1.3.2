40  @Test(groups = "unit", testName = "GlobalAccountAsyncClientTest")
41  public class GlobalAccountAsyncClientTest extends BaseCloudStackAsyncClientTest<GlobalAccountAsyncClient> {
42  
43     public void testCreateAccount() throws Exception {
44        Method method = GlobalAccountAsyncClient.class.getMethod("createAccount", String.class, Account.Type.class,
45           String.class, String.class, String.class, String.class, CreateAccountOptions[].class);
46        HttpRequest httpRequest = processor.createRequest(method, "user", Account.Type.USER, "email@example.com",
47           "FirstName", "LastName", "hashed-password");
48  
49        assertRequestLineEquals(httpRequest,
50           "GET http://localhost:8080/client/api?response=json&command=createAccount&password=hashed-password&" +
51              "username=user&email=email%40example.com&accounttype=0&firstname=FirstName&lastname=LastName HTTP/1.1");
52        assertNonPayloadHeadersEqual(httpRequest, "Accept: application/json\n");
53        assertPayloadEquals(httpRequest, null, null, false);
54  
55        assertResponseParserClassEquals(method, httpRequest, ParseFirstJsonValueNamed.class);
56        assertSaxResponseParserClassEquals(method, null);
57        assertExceptionParserClassEquals(method, ReturnNullOnNotFoundOr404.class);
58  
59        checkFilters(httpRequest);
60     }
