52     @Test
53     public void testListTemplates() throws Exception {
54        Set<Template> response = client.getTemplateClient().listTemplates();
55        assert null != response;
56        long templateCount = response.size();
57        assertTrue(templateCount >= 0);
58        for (Template template : response) {
59           Template newDetails = Iterables.getOnlyElement(client.getTemplateClient().listTemplates(
60                 zoneId(template.getZoneId()).id(template.getId())));
61           Logger.CONSOLE.info("Checking template: " + template);
62  
63           assertEquals(template, newDetails);
64           assertEquals(template, client.getTemplateClient().getTemplateInZone(template.getId(), template.getZoneId()));
65           assert template.getId() > 0 : template;
66           assert template.getName() != null : template;
67           assert template.getDisplayText() != null : template;
68           assert template.getCreated() != null : template;
69           assert template.getFormat() != null && template.getFormat() != Template.Format.UNRECOGNIZED : template;
70           assert template.getOSType() != null : template;
71           assert template.getOSTypeId() > 0 : template;
72           assert template.getAccount() != null : template;
73           assert template.getZone() != null : template;
74           assert template.getZoneId() > 0 : template;
75           assert (template.getStatus() == null ||
76              template.getStatus().equals("Download Complete")) : template;
77           assert template.getType() != null && template.getType() != Template.Type.UNRECOGNIZED : template;
78           assert template.getHypervisor() != null : template;
79           assert template.getDomain() != null : template;
80           assert template.getDomainId() > 0 : template;
81        }

137    @Test(enabled = true)
138    public void testRegisterTemplate() throws Exception {
139       Zone zone = Iterables.getFirst(client.getZoneClient().listZones(), null);
140       assertNotNull(zone);
141       Iterable<Network> networks = client.getNetworkClient().listNetworks(ListNetworksOptions.Builder.zoneId(zone.getId()).isDefault(true));
142       networks = Iterables.filter(networks, new Predicate<Network>() {
143          @Override
144          public boolean apply(@Nullable Network network) {
145             return network != null && network.getState().equals("Implemented");
146          }
147       });
148       assertEquals(Iterables.size(networks), 1);
149       Network network = Iterables.getOnlyElement(networks, null);
150       assertNotNull(network);
151       Set<OSType> osTypes = client.getGuestOSClient().listOSTypes();
152       OSType osType = Iterables.getFirst(osTypes, null);
153 
154       // Register a template
155       RegisterTemplateOptions options = RegisterTemplateOptions.Builder.bits(32).isExtractable(true);
156       TemplateMetadata templateMetadata = TemplateMetadata.builder().name(prefix+"-registerTemplate").osTypeId(osType.getId()).displayText("jclouds live testRegisterTemplate").build();
157       Set<Template> templates = client.getTemplateClient().registerTemplate(templateMetadata, "VHD", "XenServer", IMPORT_VHD_URL, zone.getId(), options);
158       registeredTemplate = Iterables.getOnlyElement(templates, null);
159       assertNotNull(registeredTemplate);
160 
161       // Ensure it is available
162       final long zoneId = zone.getId();
163       Predicate<Template> templateReadyPredicate = new Predicate<Template>() {
164          @Override
165          public boolean apply(@Nullable Template template) {
166             if (template == null) return false;
167             Template t2 = client.getTemplateClient().getTemplateInZone(template.getId(), zoneId);
168             Logger.CONSOLE.info("%s", t2.getStatus());
169             return "Download Complete".equals(t2.getStatus());
170          }
171       };
172       assertTrue(new RetryablePredicate<Template>(templateReadyPredicate, 60000).apply(registeredTemplate));
173 
174       // Create a VM that uses this template
175       vmForRegistration = VirtualMachineClientLiveTest.createVirtualMachineInNetwork(network, registeredTemplate.getId(), client, jobComplete, virtualMachineRunning);
176       assertNotNull(vmForRegistration);
177    }
