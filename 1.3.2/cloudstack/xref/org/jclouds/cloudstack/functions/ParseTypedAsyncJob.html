59  @Singleton
60  public class ParseTypedAsyncJob implements Function<AsyncJob<Map<String, JsonBall>>, AsyncJob<?>> {
61     @Resource
62     protected Logger logger = Logger.NULL;
63  
64     @Inject(optional = true)
65     @VisibleForTesting
66     @Named("jclouds.cloudstack.jobresult-type-map")
67     Map<String, Class<?>> typeMap = ImmutableMap.<String, Class<?>>builder()
68        .put("user", User.class)
69        .put("account", Account.class)
70        .put("securitygroup", SecurityGroup.class)
71        .put("portforwardingrule", PortForwardingRule.class)
72        .put("ipforwardingrule", IPForwardingRule.class)
73        .put("network", Network.class)
74        .put("ipaddress", PublicIPAddress.class)
75        .put("virtualmachine", VirtualMachine.class)
76        .put("loadbalancer", LoadBalancerRule.class).build();
77     private final Json json;
78  
79     @Inject
80     public ParseTypedAsyncJob(Json json) {
81        this.json = checkNotNull(json, "json");
82     }
83  
84     public AsyncJob<?> apply(AsyncJob<Map<String, JsonBall>> toParse) {
85        AsyncJob<?> result = toParse;
86        if (toParse.getResult() != null) {
87           if (toParse.getResult().size() == 1) {
88              @SuppressWarnings({"unchecked", "rawtypes"})
89              Builder<Object> builder = AsyncJob.Builder.fromAsyncJobUntyped((AsyncJob) toParse);
90              if (toParse.getResult().containsKey("success")) {
91                 builder.result(null);
92              } else {
93                 Entry<String, JsonBall> entry = Iterables.get(toParse.getResult().entrySet(), 0);
94                 if ("template".equals(entry.getKey())) {
95                    // Sometimes Cloudstack will say 'template' and the payload is a Template object.
96                    // Sometimes Cloudstack will say 'template' and the payload is a TemplateExtraction object.
97                    // The 'state' field only exists on TemplateExtraction, so we can test this to work out what we have actually been given.
98                    Template template = json.fromJson(entry.getValue().toString(), Template.class);
99                    TemplateExtraction templateExtraction = json.fromJson(entry.getValue().toString(), TemplateExtraction.class);
100                   boolean isTemplate = Strings.isNullOrEmpty(templateExtraction.getState());
101                   builder.result(isTemplate ? template : templateExtraction);
102                } else if (typeMap.containsKey(entry.getKey())) {
103                   builder.result(json.fromJson(entry.getValue().toString(), typeMap.get(entry.getKey())));
104                } else {
105                   logger.warn(
106                      "type key %s not configured.  please override default for Map<String, Class<?>> bound to name jclouds.cloudstack.jobresult-type-map",
107                      entry.getKey());
108                   builder.result(entry.getValue().toString());
109                }
110             }
111             result = builder.build();
112          } else if (toParse.getResult().containsKey("errorcode")) {
113             @SuppressWarnings({"unchecked", "rawtypes"})
114             Builder<Object> builder = AsyncJob.Builder.fromAsyncJobUntyped((AsyncJob) toParse);
115             builder.result(null);// avoid classcastexceptions
116             builder.error(new AsyncJobError(ErrorCode.fromValue(toParse.getResult().get("errorcode").toString()), toParse
117                .getResult().containsKey("errortext") ? toParse.getResult().get("errortext").toString()
118                .replace("\"", "") : null));
119             result = builder.build();
120          } else if (toParse.getResult().size() > 1) {
121             logger.warn("unexpected size of async job result; expecting a map with a single element",
122                toParse.getResult());
123          }
124       }
125       return result;
126    }
