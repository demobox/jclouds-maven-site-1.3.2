121    @Override
122    public NodeAndInitialCredentials<VirtualMachine> createNodeWithGroupEncodedIntoName(String group, String name,
123          org.jclouds.compute.domain.Template template) {
124       checkNotNull(template, "template was null");
125       checkNotNull(template.getOptions(), "template options was null");
126       checkArgument(template.getOptions().getClass().isAssignableFrom(CloudStackTemplateOptions.class),
127             "options class %s should have been assignable from CloudStackTemplateOptions", template.getOptions()
128                   .getClass());
129       Map<Long, Network> networks = networkSupplier.get();
130 
131       final long zoneId = Long.parseLong(template.getLocation().getId());
132       Zone zone = null;
133       try {
134          zone = zoneIdToZone.get().get(zoneId);
135       } catch (ExecutionException e) {
136          Throwables.propagate(e);
137       }
138 
139       CloudStackTemplateOptions templateOptions = template.getOptions().as(CloudStackTemplateOptions.class);
140 
141       checkState(optionsConverters.containsKey(zone.getNetworkType()), "no options converter configured for network type %s",zone.getNetworkType());
142       DeployVirtualMachineOptions options = DeployVirtualMachineOptions.Builder.displayName(name).name(name);
143       OptionsConverter optionsConverter = optionsConverters.get(zone.getNetworkType());
144       options = optionsConverter.apply(templateOptions, networks, zoneId, options);
145 
146       if (templateOptions.getIpOnDefaultNetwork() != null) {
147          options.ipOnDefaultNetwork(templateOptions.getIpOnDefaultNetwork());
148       }
149 
150       if (templateOptions.getIpsToNetworks().size() > 0) {
151          options.ipsToNetworks(templateOptions.getIpsToNetworks());
152       }
153 
154       if (templateOptions.getKeyPair() != null) {
155          options.keyPair(templateOptions.getKeyPair());
156          if (templateOptions.getRunScript() != null) {
157             checkArgument(
158                   credentialStore.containsKey("keypair#" + templateOptions.getKeyPair()),
159                   "no private key configured for: %s; please use options.overrideLoginCredentialWith(rsa_private_text)",
160                   templateOptions.getKeyPair());
161          }
162       }
163 
164       long templateId = Long.parseLong(template.getImage().getId());
165       long serviceOfferingId = Long.parseLong(template.getHardware().getId());
166 
167       logger.info("serviceOfferingId %d, templateId %d, zoneId %d, options %s%n", serviceOfferingId, templateId,
168             zoneId, options);
169       AsyncCreateResponse job = client.getVirtualMachineClient().deployVirtualMachineInZone(zoneId, serviceOfferingId,
170             templateId, options);
171       VirtualMachine vm = blockUntilJobCompletesAndReturnResult.<VirtualMachine> apply(job);
172       LoginCredentials credentials = null;
173       if (vm.isPasswordEnabled()) {
174          assert vm.getPassword() != null : vm;
175          credentials = LoginCredentials.builder().password(vm.getPassword()).build();
176       } else {
177          credentials = LoginCredentials.fromCredentials(credentialStore.get("keypair#" + templateOptions.getKeyPair()));
178       }
179       if (templateOptions.shouldSetupStaticNat()) {
180          // TODO: possibly not all network ids, do we want to do this
181          for (long networkId : options.getNetworkIds()) {
182             logger.debug(">> creating static NAT for virtualMachine(%s) in network(%s)", vm.getId(), networkId);
183             PublicIPAddress ip = staticNATVMInNetwork.create(networks.get(networkId)).apply(vm);
184             logger.trace("<< static NATed IPAddress(%s) to virtualMachine(%s)", ip.getId(), vm.getId());
185             List<Integer> ports = Ints.asList(templateOptions.getInboundPorts());
186             logger.debug(">> setting up IP forwarding for IPAddress(%s) rules(%s)", ip.getId(), ports);
187             Set<IPForwardingRule> rules = setupPortForwardingRulesForIP.apply(ip, ports);
188             logger.trace("<< setup %d IP forwarding rules on IPAddress(%s)", rules.size(), ip.getId());
189          }
190       }
191       return new NodeAndInitialCredentials<VirtualMachine>(vm, vm.getId() + "", credentials);
192    }

224    @Override
225    public void destroyNode(String id) {
226       long virtualMachineId = Long.parseLong(id);
227       // There was a bug in 2.2.12 release happening when static nat IP address
228       // was being released, and corresponding firewall rules were left behind.
229       // So next time the same IP is allocated, it might be not be static nat
230       // enabled, but there are still rules associated with it. And when you try
231       // to release this IP, the release will fail.
232       //
233       // The bug was fixed in 2.2.13 release only, and the current system wasn't
234       // updated yet.
235       //
236       // To avoid the issue, every time you release a static nat ip address, do
237       // the following:
238 
239       // 1) Delete IP forwarding rules associated with IP.
240       Set<Long> ipAddresses = deleteIPForwardingRulesForVMAndReturnDistinctIPs(virtualMachineId);
241 
242       // 2) Disable static nat rule for the IP.
243       disableStaticNATOnIPAddresses(ipAddresses);
244 
245       // 3) Only after 1 and 2 release the IP address.
246       disassociateIPAddresses(ipAddresses);
247 
248       destroyVirtualMachine(virtualMachineId);
249 
250       vmToRules.invalidate(virtualMachineId);
251    }

260    public void destroyVirtualMachine(long virtualMachineId) {
261 
262       Long destroyVirtualMachine = client.getVirtualMachineClient().destroyVirtualMachine(virtualMachineId);
263       if (destroyVirtualMachine != null) {
264          logger.debug(">> destroying virtualMachine(%s) job(%s)", virtualMachineId, destroyVirtualMachine);
265          awaitCompletion(destroyVirtualMachine);

272    public void disableStaticNATOnIPAddresses(Set<Long> ipAddresses) {
273       Builder<Long> jobsToTrack = ImmutableSet.<Long> builder();
274       for (Long ipAddress : ipAddresses) {
275          Long disableStaticNAT = client.getNATClient().disableStaticNATOnPublicIP(ipAddress);
276          if (disableStaticNAT != null) {
277             logger.debug(">> disabling static NAT IPAddress(%s) job(%s)", ipAddress, disableStaticNAT);
278             jobsToTrack.add(disableStaticNAT);
279          }
280       }
281       awaitCompletion(jobsToTrack.build());
282    }
283 
284    public Set<Long> deleteIPForwardingRulesForVMAndReturnDistinctIPs(long virtualMachineId) {
285       Builder<Long> jobsToTrack = ImmutableSet.<Long> builder();
286 
287       // immutable doesn't permit duplicates
288       Set<Long> ipAddresses = Sets.newLinkedHashSet();
289 
290       Set<IPForwardingRule> forwardingRules = client.getNATClient().getIPForwardingRulesForVirtualMachine(
291             virtualMachineId);
292       for (IPForwardingRule rule : forwardingRules) {
293          if (!"Deleting".equals(rule.getState())) {
294             ipAddresses.add(rule.getIPAddressId());
295             Long deleteForwardingRule = client.getNATClient().deleteIPForwardingRule(rule.getId());
296             if (deleteForwardingRule != null) {
297                logger.debug(">> deleting IPForwardingRule(%s) job(%s)", rule.getId(), deleteForwardingRule);
298                jobsToTrack.add(deleteForwardingRule);
299             }
300          }
301       }
302       awaitCompletion(jobsToTrack.build());
303       return ipAddresses;
304    }

306    public void awaitCompletion(Iterable<Long> jobs) {
307       logger.debug(">> awaiting completion of jobs(%s)", jobs);
308       for (long job : jobs)
309          awaitCompletion(job);
310       logger.trace("<< completed jobs(%s)", jobs);
311    }

313    public void awaitCompletion(long job) {
314       boolean completed = jobComplete.apply(job);
315       logger.trace("<< job(%s) complete(%s)", job, completed);
316    }
