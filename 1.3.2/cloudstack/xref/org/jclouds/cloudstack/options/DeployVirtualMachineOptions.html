46     /**
47      * the ID of the disk offering for the virtual machine. If the template is of
48      * ISO format, the diskOfferingId is for the root disk volume. Otherwise this
49      * parameter is used to dinidcate the offering for the data disk volume. If
50      * the templateId parameter passed is from a Template object, the
51      * diskOfferingId refers to a DATA Disk Volume created. If the templateId
52      * parameter passed is from an ISO object, the diskOfferingId refers to a
53      * ROOT Disk Volume created.
54      * 
55      * @param diskofferingid
56      *           the ID of the disk offering
57      */
58     public DeployVirtualMachineOptions diskOfferingId(long diskofferingid) {
59        checkArgument(!queryParameters.containsKey("size"), "Mutually exclusive with size");
60        this.queryParameters.replaceValues("diskofferingid", ImmutableSet.of(diskofferingid + ""));
61        return this;
62     }

125    /**
126     * @param ipsToNetworks
127     *           mapping ip addresses to network ids (2.2.12 only option)
128     */
129    public DeployVirtualMachineOptions ipsToNetworks(Map<String, Long> ipsToNetworks) {
130       int count = 0;
131       for (String ip : ipsToNetworks.keySet()) {
132          this.queryParameters.replaceValues(String.format("ipnetworklist[%d].ip", count), ImmutableSet.of(ip));
133          this.queryParameters.replaceValues(String.format("ipnetworklist[%d].networkid", count),
134                ImmutableSet.of("" + ipsToNetworks.get(ip)));
135          count += 1;
136       }
137       return this;
138    }

158    public Iterable<Long> getNetworkIds() {
159       if (queryParameters.get("networkids").size() == 1) {
160          return Iterables.transform(
161                Splitter.on(",").split(Iterables.getOnlyElement(queryParameters.get("networkids"))),
162                new Function<String, Long>() {
163 
164                   @Override
165                   public Long apply(String arg0) {
166                      return Long.parseLong(arg0);
167                   }
168 
169                });
170       } else {
171          return ImmutableSet.<Long> of();
172       }
173    }

206    /**
207     * @param unencodedData
208     *           an optional binary data that can be sent to the virtual machine
209     *           upon a successful deployment. This binary data must be base64
210     *           encoded before adding it to the request. Currently only HTTP GET
211     *           is supported. Using HTTP GET (via querystring), you can send up
212     *           to 2KB of data after base64 encoding.
213     */
214    public DeployVirtualMachineOptions userData(byte[] unencodedData) {
215       int length = checkNotNull(unencodedData, "unencodedData").length;
216       checkArgument(length > 0, "userData cannot be empty");
217       checkArgument(length <= 2 * 1024, "userData cannot be larger than 2kb");
218       this.queryParameters.replaceValues("userdata", ImmutableSet.of(Base64.encodeBytes(unencodedData)));
219       return this;
220    }
