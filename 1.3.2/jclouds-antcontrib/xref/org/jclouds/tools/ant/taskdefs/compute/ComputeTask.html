82     public static enum Action {
83        CREATE, GET, LIST, LIST_DETAILS, DESTROY, REBOOT, LIST_IMAGES, LIST_SIZES, LIST_LOCATIONS
84     }
85  
86     /**
87      * makes a connection to the compute service and invokes
88      */
89     public void execute() throws BuildException {
90        ComputeServiceContext context = computeMap.getUnchecked(HttpUtils.createUri(provider));
91  
92        try {
93           for (String action : Splitter.on(',').split(actions)) {
94              Action act = Action.valueOf(CaseFormat.LOWER_HYPHEN.to(CaseFormat.UPPER_UNDERSCORE, action));
95              try {
96                 invokeActionOnService(act, context.getComputeService());
97              } catch (RunNodesException e) {
98                 throw new BuildException(e);
99              } catch (IOException e) {
100                throw new BuildException(e);
101             }
102          }
103       } finally {
104          context.close();
105       }
106    }

108    private void invokeActionOnService(Action action, ComputeService computeService) throws RunNodesException, IOException {
109       switch (action) {
110          case CREATE:
111          case GET:
112          case DESTROY:
113          case REBOOT:
114             if (nodeElement != null) {
115                switch (action) {
116                   case CREATE:
117                      create(computeService);
118                      break;
119                   case GET:
120                      get(computeService);
121                      break;
122                   case DESTROY:
123                      destroy(computeService);
124                      break;
125                   case REBOOT:
126                      reboot(computeService);
127                      break;
128                }
129             } else {
130                this.log("missing node element for action: " + action, Project.MSG_ERR);
131             }
132             break;
133          case LIST:
134             list(computeService);
135             break;
136          case LIST_DETAILS:
137             listDetails(computeService);
138             break;
139          case LIST_IMAGES:
140             listImages(computeService);
141             break;
142          case LIST_SIZES:
143             listHardwares(computeService);
144             break;
145          case LIST_LOCATIONS:
146             listLocations(computeService);
147             break;
148          default:
149             this.log("bad action: " + action, Project.MSG_ERR);
150       }
151    }

194    private void create(ComputeService computeService) throws RunNodesException, IOException {
195       String group = nodeElement.getGroup();
196 
197       log(String.format("create group: %s, count: %d, hardware: %s, os: %s", group, nodeElement.getCount(), nodeElement
198                .getHardware(), nodeElement.getOs()));
199 
200       Template template = createTemplateFromElement(nodeElement, computeService);
201 
202       for (NodeMetadata createdNode : computeService.createNodesInGroup(group, nodeElement.getCount(), template)) {
203          logDetails(computeService, createdNode);
204          addNodeDetailsAsProjectProperties(createdNode);
205       }
206    }
207 
208    private void addNodeDetailsAsProjectProperties(NodeMetadata createdNode) {
209       if (nodeElement.getIdproperty() != null)
210          getProject().setProperty(nodeElement.getIdproperty(), createdNode.getProviderId());
211       if (nodeElement.getHostproperty() != null)
212          getProject().setProperty(nodeElement.getHostproperty(), ipOrEmptyString(createdNode.getPublicAddresses()));
213       if (nodeElement.getPasswordproperty() != null && !CredentialUtils.isPrivateKeyCredential(createdNode.getCredentials()))
214          getProject().setProperty(nodeElement.getPasswordproperty(), createdNode.getCredentials().credential);
215       if (nodeElement.getUsernameproperty() != null)
216          getProject().setProperty(nodeElement.getUsernameproperty(), createdNode.getCredentials().identity);
217    }

219    private void reboot(ComputeService computeService) {
220       if (nodeElement.getId() != null) {
221          log(String.format("reboot id: %s", nodeElement.getId()));
222          computeService.rebootNode(nodeElement.getId());
223       } else {
224          log(String.format("reboot group: %s", nodeElement.getGroup()));
225          computeService.rebootNodesMatching(NodePredicates.inGroup(nodeElement.getGroup()));
226       }
227    }

229    private void destroy(ComputeService computeService) {
230       if (nodeElement.getId() != null) {
231          log(String.format("destroy id: %s", nodeElement.getId()));
232          computeService.destroyNode(nodeElement.getId());
233       } else {
234          log(String.format("destroy group: %s", nodeElement.getGroup()));
235          computeService.destroyNodesMatching(NodePredicates.inGroup(nodeElement.getGroup()));
236       }
237    }

239    private void get(ComputeService computeService) {
240       if (nodeElement.getId() != null) {
241          log(String.format("get id: %s", nodeElement.getId()));
242          logDetails(computeService, computeService.getNodeMetadata(nodeElement.getId()));
243       } else {
244          log(String.format("get group: %s", nodeElement.getGroup()));
245          for (ComputeMetadata node : Iterables.filter(computeService.listNodesDetailsMatching(NodePredicates.all()),
246                   NodePredicates.inGroup(nodeElement.getGroup()))) {
247             logDetails(computeService, node);
248          }
249       }
250    }
251 
252    private void logDetails(ComputeService computeService, ComputeMetadata node) {
253       NodeMetadata metadata = node instanceof NodeMetadata ? NodeMetadata.class.cast(node) : computeService
254                .getNodeMetadata(node.getId());
255       log(String.format("   node id=%s, name=%s, group=%s, location=%s, state=%s, publicIp=%s, privateIp=%s, hardware=%s",
256                metadata.getProviderId(), metadata.getName(), metadata.getGroup(), metadata.getLocation(), metadata
257                         .getState(), ComputeTaskUtils.ipOrEmptyString(metadata.getPublicAddresses()),
258                ipOrEmptyString(metadata.getPrivateAddresses()), metadata.getHardware()));
259    }
