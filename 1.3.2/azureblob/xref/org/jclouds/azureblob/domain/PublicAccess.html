21  /**
22   * Indicates whether data in the container may be accessed publicly and the level of access.
23   * 
24   * @author Adrian Cole
25   * 
26   * @see <a href="http://msdn.microsoft.com/en-us/library/dd179469.aspx"/>
27   */
28  public enum PublicAccess {
29     /**
30      * Indicates full public read access for container and blob data. Clients can enumerate blobs
31      * within the container via anonymous request, but cannot enumerate containers within the storage
32      * account.
33      */
34     CONTAINER,
35     /**
36      * Indicates public read access for blobs. Blob data within this container can be read via
37      * anonymous request, but container data is not available. Clients cannot enumerate blobs within
38      * the container via anonymous request.
39      */
40     BLOB,
41     /**
42      * the container is private to the account owner.
43      */
44     PRIVATE;

46  }
