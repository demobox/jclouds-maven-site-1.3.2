66     public static HttpUriRequest convertToApacheRequest(HttpRequest request) {
67        HttpUriRequest apacheRequest;
68        if (request.getMethod().equals(HttpMethod.HEAD)) {
69           apacheRequest = new HttpHead(request.getEndpoint());
70        } else if (request.getMethod().equals(HttpMethod.GET)) {
71           apacheRequest = new HttpGet(request.getEndpoint());
72        } else if (request.getMethod().equals(HttpMethod.DELETE)) {
73           apacheRequest = new HttpDelete(request.getEndpoint());
74        } else if (request.getMethod().equals(HttpMethod.PUT)) {
75           apacheRequest = new HttpPut(request.getEndpoint());
76           apacheRequest.getParams().setBooleanParameter(CoreProtocolPNames.USE_EXPECT_CONTINUE, true);
77        } else if (request.getMethod().equals(HttpMethod.POST)) {
78           apacheRequest = new HttpPost(request.getEndpoint());
79        } else {
80           final String method = request.getMethod();
81           if (request.getPayload() != null)
82              apacheRequest = new HttpEntityEnclosingRequestBase() {
83  
84                 @Override
85                 public String getMethod() {
86                    return method;
87                 }
88  
89              };
90           else
91              apacheRequest = new HttpRequestBase() {
92  
93                 @Override
94                 public String getMethod() {
95                    return method;
96                 }
97  
98              };
99           HttpRequestBase.class.cast(apacheRequest).setURI(request.getEndpoint());
100       }
101       Payload payload = request.getPayload();
102 
103       // Since we may remove headers, ensure they are added to the apache
104       // request after this block
105       if (apacheRequest instanceof HttpEntityEnclosingRequest) {
106          if (payload != null) {
107             addEntityForContent(HttpEntityEnclosingRequest.class.cast(apacheRequest), payload);
108          }
109       } else {
110          apacheRequest.addHeader(HttpHeaders.CONTENT_LENGTH, "0");
111       }
112 
113       for (String header : request.getHeaders().keySet()) {
114          for (String value : request.getHeaders().get(header))
115             // apache automatically tries to add content length header
116             if (!header.equals(HttpHeaders.CONTENT_LENGTH))
117                apacheRequest.addHeader(header, value);
118       }
119       apacheRequest.addHeader(HttpHeaders.USER_AGENT, USER_AGENT);
120       return apacheRequest;
121    }

123    public static void addEntityForContent(HttpEntityEnclosingRequest apacheRequest, Payload payload) {
124       payload = payload instanceof DelegatingPayload ? DelegatingPayload.class.cast(payload).getDelegate() : payload;
125       if (payload instanceof StringPayload) {
126          StringEntity nStringEntity = null;
127          try {
128             nStringEntity = new StringEntity((String) payload.getRawContent());
129          } catch (UnsupportedEncodingException e) {
130             throw new UnsupportedOperationException("Encoding not supported", e);
131          }
132          nStringEntity.setContentType(payload.getContentMetadata().getContentType());
133          apacheRequest.setEntity(nStringEntity);
134       } else if (payload instanceof FilePayload) {
135          apacheRequest.setEntity(new FileEntity((File) payload.getRawContent(), payload.getContentMetadata()
136                .getContentType()));
137       } else if (payload instanceof ByteArrayPayload) {
138          ByteArrayEntity Entity = new ByteArrayEntity((byte[]) payload.getRawContent());
139          Entity.setContentType(payload.getContentMetadata().getContentType());
140          apacheRequest.setEntity(Entity);
141       } else {
142          InputStream inputStream = payload.getInput();
143          if (payload.getContentMetadata().getContentLength() == null)
144             throw new IllegalArgumentException("you must specify size when content is an InputStream");
145          InputStreamEntity Entity = new InputStreamEntity(inputStream, payload.getContentMetadata().getContentLength());
146          Entity.setContentType(payload.getContentMetadata().getContentType());
147          apacheRequest.setEntity(Entity);
148       }
149       if (payload.getContentMetadata().getContentDisposition() != null)
150          apacheRequest.addHeader("Content-Disposition", payload.getContentMetadata().getContentDisposition());
151       if (payload.getContentMetadata().getContentEncoding() != null)
152          apacheRequest.addHeader("Content-Encoding", payload.getContentMetadata().getContentEncoding());
153       if (payload.getContentMetadata().getContentLanguage() != null)
154          apacheRequest.addHeader("Content-Language", payload.getContentMetadata().getContentLanguage());
155       assert (apacheRequest.getEntity() != null);
156    }

158    public static class HttpEntityPayload extends BasePayload<HttpEntity> {
159 
160       HttpEntityPayload(HttpEntity content) {
161          super(content);
162          getContentMetadata().setContentType(content.getContentType().getValue());
163          getContentMetadata().setContentLength(content.getContentLength());
164       }
165 
166       @Override
167       public InputStream getInput() {
168          try {
169             return content.getContent();
170          } catch (IllegalStateException e) {
171             Throwables.propagate(e);
172          } catch (IOException e) {
173             Throwables.propagate(e);
174          }
175          return null;
176       }
177 
178       @Override
179       public boolean isRepeatable() {
180          return content.isRepeatable();
181       }
182 
183       @Override
184       public void release() {
185          try {
186             content.consumeContent();
187          } catch (IOException e) {
188          }
189       }
190 
191       @Override
192       public void writeTo(OutputStream outstream) throws IOException {
193          super.writeTo(outstream);
194       }
195 
196    }
