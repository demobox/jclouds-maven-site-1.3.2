38  /**
39   * Tests behavior of {@code KeyPairByNameHandler}
40   * 
41   * @author Adrian Cole

43  // NOTE:without testName, this will not call @Before* and fail w/NPE during surefire
44  @Test(groups = "unit", testName = "KeyPairByNameHandlerTest")
45  public class KeyPairByNameHandlerTest extends BaseHandlerTest {
46  
47     public void testGood() throws UnknownHostException {
48        InputStream is = getClass().getResourceAsStream("/keysList.xml");
49  
50        KeyPair result = factory.create(
51              addOrgAndNameToHandler(injector.getInstance(KeyPairByNameHandler.class), "org", "default")).parse(is);
52        assertEquals(result,
53              new KeyPair(URI.create("https://services.vcloudexpress.terremark.com/api/v0.8a-ext1.6/extensions/key/9"),
54                    "default", true, null, "4e:af:8a:9f:e9:d2:72:d7:4b:a0:da:98:72:98:4d:7d"));
55  
56     }
57  
58     public void testNotFound() throws UnknownHostException {
59        InputStream is = getClass().getResourceAsStream("/keysList.xml");
60  
61        KeyPair result = factory.create(
62              addOrgAndNameToHandler(injector.getInstance(KeyPairByNameHandler.class), "org", "monster")).parse(is);
63        assertEquals(result, null);
64  
65     }
66  
67     private static KeyPairByNameHandler addOrgAndNameToHandler(KeyPairByNameHandler handler, String org, String name) {
68        GeneratedHttpRequest<?> request = createMock(GeneratedHttpRequest.class);
69        expect(request.getArgs()).andReturn(ImmutableList.<Object> of(org, name)).anyTimes();
70        replay(request);
71        handler.setContext(request);
72        return handler;
73     }
