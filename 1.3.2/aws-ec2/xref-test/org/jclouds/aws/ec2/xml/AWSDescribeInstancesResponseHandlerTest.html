233    static ParseSax<Set<Reservation<? extends RunningInstance>>> createParser() {
234       Injector injector = Guice.createInjector(new SaxParserModule(), new AbstractModule() {
235 
236          @Override
237          protected void configure() {
238             bind(String.class).annotatedWith(Region.class).toInstance("us-east-1");
239             bind(RunningInstance.Builder.class).to(AWSRunningInstance.Builder.class);
240          }
241 
242       });
243       ParseSax<Set<Reservation<? extends RunningInstance>>> parser = (ParseSax<Set<Reservation<? extends RunningInstance>>>) injector
244             .getInstance(ParseSax.Factory.class)
245             .create(injector.getInstance(AWSDescribeInstancesResponseHandler.class));
246       return parser;
247    }

249    public static Set<Reservation<? extends RunningInstance>> parseAWSRunningInstances(String resource) {
250       InputStream is = AWSDescribeInstancesResponseHandlerTest.class.getResourceAsStream(resource);
251       return createParser().parse(is);
252    }
