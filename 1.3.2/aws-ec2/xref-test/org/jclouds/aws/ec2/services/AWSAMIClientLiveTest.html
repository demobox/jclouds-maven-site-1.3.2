89     public void testDescribeImages() {
90        for (String region : context.getApi().getAvailabilityZoneAndRegionServices().describeRegions().keySet()) {
91           Set<Image> allResults = Sets.newLinkedHashSet(client.describeImagesInRegion(region));
92           assertNotNull(allResults);
93           assert allResults.size() >= 2 : allResults.size();
94           Iterator<Image> iterator = allResults.iterator();
95           String id1 = iterator.next().getId();
96           String id2 = iterator.next().getId();
97           Set<Image> twoResults = Sets.newLinkedHashSet(client.describeImagesInRegion(region, imageIds(id1, id2)));
98           assertNotNull(twoResults);
99           assertEquals(twoResults.size(), 2);
100          iterator = twoResults.iterator();
101          assertEquals(iterator.next().getId(), id1);
102          assertEquals(iterator.next().getId(), id2);
103       }
104    }
