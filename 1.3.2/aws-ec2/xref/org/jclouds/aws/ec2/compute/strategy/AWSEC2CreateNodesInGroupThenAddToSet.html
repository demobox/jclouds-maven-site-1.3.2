66  public class AWSEC2CreateNodesInGroupThenAddToSet extends EC2CreateNodesInGroupThenAddToSet {

68     @Resource
69     @Named(ComputeServiceConstants.COMPUTE_LOGGER)
70     protected Logger logger = Logger.NULL;

72     @VisibleForTesting
73     final AWSEC2Client client;

75     final AWSEC2AsyncClient aclient;
76     final boolean generateInstanceNames;

78     @Inject
79     protected AWSEC2CreateNodesInGroupThenAddToSet(
80              AWSEC2Client client,
81              @Named("ELASTICIP") LoadingCache<RegionAndName, String> elasticIpCache,
82              @Named("NODE_RUNNING") Predicate<NodeMetadata> nodeRunning,
83              AWSEC2AsyncClient aclient,
84              @Named(PROPERTY_EC2_GENERATE_INSTANCE_NAMES) boolean generateInstanceNames,
85              Provider<TemplateBuilder> templateBuilderProvider,
86              CreateKeyPairPlacementAndSecurityGroupsAsNeededAndReturnRunOptions createKeyPairAndSecurityGroupsAsNeededAndReturncustomize,
87              AWSEC2InstancePresent instancePresent,
88              Function<RunningInstance, NodeMetadata> runningInstanceToNodeMetadata,
89              LoadingCache<RunningInstance, Credentials> instanceToCredentials, Map<String, Credentials> credentialStore,
90              ComputeUtils utils, SpotInstanceRequestToAWSRunningInstance spotConverter) {
91        super(client, elasticIpCache, nodeRunning, templateBuilderProvider, createKeyPairAndSecurityGroupsAsNeededAndReturncustomize,
92                 instancePresent, runningInstanceToNodeMetadata, instanceToCredentials, credentialStore, utils);
93        this.client = checkNotNull(client, "client");
94        this.aclient = checkNotNull(aclient, "aclient");
95        this.spotConverter = checkNotNull(spotConverter, "spotConverter");
96        this.generateInstanceNames = generateInstanceNames;
97     }

99     @Override
100    protected Iterable<? extends RunningInstance> createNodesInRegionAndZone(String region, String zone, String group,
101             int count, Template template, RunInstancesOptions instanceOptions) {
102       Float spotPrice = getSpotPriceOrNull(template.getOptions());
103       if (spotPrice != null) {
104          LaunchSpecification spec = AWSRunInstancesOptions.class.cast(instanceOptions).getLaunchSpecificationBuilder()
105                   .imageId(template.getImage().getProviderId()).availabilityZone(zone).build();
106          RequestSpotInstancesOptions options = AWSEC2TemplateOptions.class.cast(template.getOptions()).getSpotOptions();
107          if (logger.isDebugEnabled())
108             logger.debug(">> requesting %d spot instances region(%s) price(%f) spec(%s) options(%s)", count, region,
109                      spotPrice, spec, options);
110 
111          return addTagsToInstancesInRegion(template.getOptions().getUserMetadata(), transform(client
112                   .getSpotInstanceServices().requestSpotInstancesInRegion(region, spotPrice, count, spec, options),
113                   spotConverter), region, group);
114       } else {
115          return addTagsToInstancesInRegion(template.getOptions().getUserMetadata(), super.createNodesInRegionAndZone(
116                   region, zone, group, count, template, instanceOptions), region, group);
117       }
118 
119    }
120 
121    public Iterable<? extends RunningInstance> addTagsToInstancesInRegion(Map<String, String> metadata,
122             Iterable<? extends RunningInstance> iterable, String region, String group) {
123       if (metadata.size() > 0 || generateInstanceNames) {
124          for (String id : transform(iterable, new Function<RunningInstance, String>() {
125 
126             @Override
127             public String apply(RunningInstance arg0) {
128                return arg0.getId();
129             }
130 
131          }))
132             aclient.getTagServices()
133                      .createTagsInRegion(region, ImmutableSet.of(id), metadataForId(id, group, metadata));
134       }
135       return iterable;
136    }
137 
138    private Map<String, String> metadataForId(String id, String group, Map<String, String> metadata) {
139       return generateInstanceNames && !metadata.containsKey("Name") ? ImmutableMap.<String, String> builder().putAll(
140                metadata).put("Name", id.replaceAll(".*-", group + "-")).build() : metadata;
141    }
142 
143    private Float getSpotPriceOrNull(TemplateOptions options) {
144       return options instanceof AWSEC2TemplateOptions ? AWSEC2TemplateOptions.class.cast(options).getSpotPrice() : null;
145    }
