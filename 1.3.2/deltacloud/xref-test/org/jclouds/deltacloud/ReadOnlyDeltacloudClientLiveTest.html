99     @Test
100    public void testGetInstanceStatesCanGoFromStartToFinish() throws Exception {
101       Multimap<Instance.State, ? extends Transition> states = client.getInstanceStates();
102       assertNotNull(states);
103       Iterable<Transition> toFinishFromStart = findChainTo(Instance.State.FINISH, Instance.State.START, states);
104       assert Iterables.size(toFinishFromStart) > 0 : toFinishFromStart;
105       Iterable<Transition> toRunningFromStart = findChainTo(Instance.State.RUNNING, Instance.State.START, states);
106       assert Iterables.size(toRunningFromStart) > 0 : toRunningFromStart;
107       Iterable<Transition> toFinishFromRunning = findChainTo(Instance.State.FINISH, Instance.State.RUNNING, states);
108       assert Iterables.size(toFinishFromRunning) > 0 : toFinishFromRunning;
109       assertEquals(ImmutableList.copyOf(Iterables.concat(toRunningFromStart, toFinishFromRunning)), ImmutableList
110                .copyOf(toFinishFromStart));
111    }
112 
113    Iterable<Transition> findChainTo(Instance.State desired, Instance.State currentState,
114             Multimap<Instance.State, ? extends Transition> states) {
115       for (Transition transition : states.get(currentState)) {
116          if (currentState.ordinal() >= transition.getTo().ordinal())
117             continue;
118          if (transition.getTo() == desired)
119             return ImmutableSet.<Transition> of(transition);
120          Iterable<Transition> transitions = findChainTo(desired, transition.getTo(), states);
121          if (Iterables.size(transitions) > 0)
122             return Iterables.concat(ImmutableSet.of(transition), transitions);
123       }
124       return ImmutableSet.<Transition> of();
125    }
