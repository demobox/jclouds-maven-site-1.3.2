82     @Test
83     public void testGetProductPackage() {
84        for (ProductPackage productPackage : accountClient.getActivePackages()) {
85           ProductPackage response = client.getProductPackage(productPackage.getId());

87           assert null != response;
88           assert response.getId() > 0 : response;
89           assert response.getName() != null : response;
90           assert response.getDescription() != null : response;

92           assertTrue(response.getItems().size() >= 0);

101             checkDatacenter(datacenter);
102          }
103       }

130    @Test
131    public void testGetOneGBRamPrice() {
132       // Predicate p =
133       // Predicates.and(ProductItemPredicates.categoryCode("ram"),ProductItemPredicates.capacity(1.0f));
134       Iterable<ProductItem> ramItems = Iterables.filter(cloudServerProductPackage.getItems(),
135             Predicates.and(categoryCode("ram"), capacity(1.0f)));
136 
137       // capacity is key in GB (1Gb = 1.0f)
138       Map<Float, ProductItem> ramToProductItem = Maps.uniqueIndex(ramItems, ProductItems.capacity());
139 
140       ProductItemPrice price = ProductItems.price().apply(ramToProductItem.get(1.0f));
141       assert new Integer(1644).equals(price.getId());
142    }

144    @Test
145    public void testGetTwoCPUCoresPrice() {
146       // If use ProductItemPredicates.categoryCode("guest_core") get duplicate
147       // capacities (units =
148       // PRIVATE_CORE and N/A)
149       Iterable<ProductItem> cpuItems = Iterables.filter(cloudServerProductPackage.getItems(),
150             Predicates.and(units("PRIVATE_CORE"), capacity(2.0f)));
151 
152       // number of cores is the key
153       Map<Float, ProductItem> coresToProductItem = Maps.uniqueIndex(cpuItems, ProductItems.capacity());
154 
155       ProductItemPrice price = ProductItems.price().apply(coresToProductItem.get(2.0f));
156       assert new Integer(1963).equals(price.getId());

171    private void checkProductItem(ProductItem item) {
172       assert item.getId() > 0 : item;
173       assert item.getDescription() != null : item;
174       checkCategories(item.getCategories());
175       // units and capacity may be null
176 
177       assertTrue(item.getPrices().size() >= 0);
178 
179       for (ProductItemPrice price : item.getPrices()) {
180          // ProductItemPrice newDetails =
181          // client.getProductItemPrice(price.getId());
182          // assertEquals(item.getId(), newDetails.getId());
183          checkPrice(price);
184       }
185    }
186 
187    private void checkPrice(ProductItemPrice price) {
188       assert price.getId() > 0 : price;
189       assert price.getItemId() > 0 : price;
190       assert price.getRecurringFee() != null || price.getHourlyRecurringFee() != null : price;
191    }

193    private void checkDatacenter(Datacenter datacenter) {
194       assert datacenter.getId() > 0 : datacenter;
195       assert datacenter.getName() != null : datacenter;
196       assert datacenter.getLongName() != null : datacenter;
197       for (Region region : datacenter.getRegions())
198          checkRegion(region);
199    }
200 
201    private void checkRegion(Region region) {
202       assert !region.getDescription().isEmpty() : region;
203       assert !region.getKeyname().isEmpty() : region;
204    }

206    private void checkAddress(Address address) {
207       assert address.getId() > 0 : address;
208       assert address.getCountry() != null : address;
209       if (!ImmutableSet.of("SG", "NL").contains(address.getCountry()))
210          assert address.getState() != null : address;
211    }

213    private void checkCategories(Set<ProductItemCategory> categories) {
214       for (ProductItemCategory category : categories) {

216          assert category.getName() != null : category;
217          assert category.getCategoryCode() != null : category;
218       }
