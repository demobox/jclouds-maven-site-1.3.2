66        cpuItem = ProductItem.builder()
67              .id(1)
68              .description("2 x 2.0 GHz Cores")
69              .capacity(2F)
70              .category(ProductItemCategory.builder().categoryCode("guest_core").build())
71              .price(ProductItemPrice.builder().id(123).build())
72              .build();
73  
74        ramItem = ProductItem.builder().id(2).description("2GB ram").capacity(2F).category(
75                 ProductItemCategory.builder().categoryCode("ram").build()).price(
76                 ProductItemPrice.builder().id(456).build()).build();
77  
78        volumeItem = ProductItem.builder().id(3).description("100 GB (SAN)").capacity(100F).price(
79                 ProductItemPrice.builder().id(789).build()).category(
80                 ProductItemCategory.builder().categoryCode("guest_disk0").build()).build();

85     @Test
86     public void testHardwareId() {
87        ProductItem item1 = ProductItem.builder().price(ProductItemPrice.builder().id(123).build()).build();
88        ProductItem item2 = ProductItem.builder().price(ProductItemPrice.builder().id(456).build()).build();
89        ProductItem item3 = ProductItem.builder().price(ProductItemPrice.builder().id(789).build()).build();
90  
91        String id = hardwareId().apply(ImmutableList.of(item1, item2, item3));
92        assertEquals("123,456,789", id);
93     }
94  
95     @Test
96     public void testHardware() {
97  
98        Hardware hardware = toHardware.apply(ImmutableSet.of(cpuItem, ramItem, volumeItem));
99  
100       assertEquals("123,456,789", hardware.getId());
101 
102       List<? extends Processor> processors = hardware.getProcessors();
103       assertEquals(1, processors.size());
104       assertEquals(2.0, processors.get(0).getCores());
105 
106       assertEquals(2, hardware.getRam());
107 
108       List<? extends Volume> volumes = hardware.getVolumes();
109       assertEquals(1, volumes.size());
110       Volume volume = volumes.get(0);
111       assertEquals(100F, volume.getSize());
112       assertEquals(Volume.Type.SAN, volume.getType());
113       assertEquals(true, volume.isBootDevice());
114 
115    }
