56     @Test
57     public void testApplyWhereVirtualGuestWithNoPassword() {
58  
59        // notice if we've already parsed this properly here, we can rely on it.
60        VirtualGuest guest = new ParseVirtualGuestWithNoPasswordTest().expected();
61  
62        // setup so that we have an expected Location to be parsed from the guest.
63        Location expectedLocation = DatacenterToLocationTest.function.apply(guest.getDatacenter());
64        Supplier<Set<? extends Location>> locationSupplier = Suppliers.<Set<? extends Location>> ofInstance(ImmutableSet
65              .<Location> of(expectedLocation));
66  
67        VirtualGuestToNodeMetadata parser = new VirtualGuestToNodeMetadata(new FindLocationForVirtualGuest(
68              locationSupplier), new GetHardwareForVirtualGuestMock(), new GetImageForVirtualGuestMock());
69  
70        NodeMetadata node = parser.apply(guest);
71  
72        assertEquals(
73              node,
74              new NodeMetadataBuilder().ids("416788").name("node1000360500").hostname("node1000360500")
75                    .location(expectedLocation).state(NodeState.PENDING)
76                    .publicAddresses(ImmutableSet.of("173.192.29.186"))
77                    .privateAddresses(ImmutableSet.of("10.37.102.194"))
78                    .hardware(new GetHardwareForVirtualGuestMock().getHardware(guest))
79                    .imageId(new GetImageForVirtualGuestMock().getImage(guest).getId())
80                    .operatingSystem(new GetImageForVirtualGuestMock().getImage(guest).getOperatingSystem()).build());
81
