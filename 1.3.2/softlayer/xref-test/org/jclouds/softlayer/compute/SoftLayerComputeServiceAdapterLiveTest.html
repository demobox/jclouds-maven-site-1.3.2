66     private static final PrioritizeCredentialsFromTemplate prioritizeCredentialsFromTemplate = new PrioritizeCredentialsFromTemplate(
67           new DefaultCredentialsFromImageOrOverridingCredentials());
68  
69     @Test
70     public void testCreateNodeWithGroupEncodedIntoNameThenStoreCredentials() {
71        String group = "foo";
72        String name = "node" + new Random().nextInt();
73        Template template = computeContext.getComputeService().templateBuilder().build();
74  
75        // test passing custom options
76        template.getOptions().as(SoftLayerTemplateOptions.class).domainName("me.org");
77  
78        guest = adapter.createNodeWithGroupEncodedIntoName(group, name, template);
79        assertEquals(guest.getNode().getHostname(), name);
80        assertEquals(guest.getNodeId(), guest.getNode().getId() + "");
81        assertEquals(guest.getNode().getDomain(), template.getOptions().as(SoftLayerTemplateOptions.class)
82              .getDomainName());
83        assert InetAddresses.isInetAddress(guest.getNode().getPrimaryBackendIpAddress()) : guest;
84        doConnectViaSsh(guest.getNode(), prioritizeCredentialsFromTemplate.apply(template, guest.getCredentials()));
85     }
86  
87     protected void doConnectViaSsh(VirtualGuest guest, LoginCredentials creds) {
88        SshClient ssh = computeContext.utils().sshFactory().create(new IPSocket(guest.getPrimaryIpAddress(), 22), creds);
89        try {
90           ssh.connect();
91           ExecResponse hello = ssh.exec("echo hello");
92           assertEquals(hello.getOutput().trim(), "hello");
93           System.err.println(ssh.exec("df -k").getOutput());
94           System.err.println(ssh.exec("mount").getOutput());
95           System.err.println(ssh.exec("uname -a").getOutput());
96        } finally {
97           if (ssh != null)
98              ssh.disconnect();
99        }
100    }
