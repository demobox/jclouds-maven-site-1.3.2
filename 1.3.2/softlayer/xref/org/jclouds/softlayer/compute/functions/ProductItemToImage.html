45  @Singleton
46  public class ProductItemToImage implements Function<ProductItem, Image> {
47  
48     /**  Pattern to capture the number of bits e.g. "a (32 bit) os" */
49     private static final Pattern OS_BITS_PATTERN = Pattern.compile(".*\\((\\d+) ?bit\\).*");
50  
51     private static final String CENTOS  = "CentOS";
52     private static final String DEBIAN  = "Debian GNU/Linux";
53     private static final String FEDORA  = "Fedora Release";
54     private static final String RHEL    = "Red Hat Enterprise Linux";
55     private static final String UBUNTU  = "Ubuntu Linux";
56     private static final String WINDOWS = "Windows Server";
57  
58     @Resource
59     @Named(ComputeServiceConstants.COMPUTE_LOGGER)
60     protected Logger logger = Logger.NULL;
61  
62     @Override
63     public Image apply(ProductItem productItem) {
64        checkNotNull(productItem,"productItem");
65  
66        OsFamily family = osFamily().apply(productItem);
67        Integer bits = osBits().apply(productItem);
68        OperatingSystem os = OperatingSystem.builder()
69                                .description(productItem.getDescription())
70                                .family(family)
71                                .version(osVersion().apply(productItem))
72                                .is64Bit(bits.equals(64))
73                                .build();
74  
75        return new ImageBuilder()
76              .ids(imageId().apply(productItem))
77              .description(productItem.getDescription())
78              .operatingSystem(os)
79              .build();
80     }
81  
82     /**
83      * Parses the item description to determine the OSFamily
84      * @return the @see OsFamily or OsFamily.UNRECOGNIZED
85      */
86      public static Function<ProductItem,OsFamily> osFamily() {
87         return new Function<ProductItem,OsFamily>() {
88              @Override
89              public OsFamily apply(ProductItem productItem) {
90                 checkNotNull(productItem,"productItem");
91  
92                 final String description = productItem.getDescription();
93                 if(description.startsWith(CENTOS)) return OsFamily.CENTOS;
94                 else if(description.startsWith(DEBIAN)) return OsFamily.DEBIAN;
95                 else if(description.startsWith(FEDORA)) return OsFamily.FEDORA;
96                 else if(description.startsWith(RHEL)) return OsFamily.RHEL;
97                 else if(description.startsWith(UBUNTU)) return OsFamily.UBUNTU;
98                 else if(description.startsWith(WINDOWS)) return OsFamily.WINDOWS;
99                 return OsFamily.UNRECOGNIZED;
100             }
101         };
102     }
103 
104    /**
105     * Parses the item description to determine the os version
106     * @return the version
107     * @throws java.util.NoSuchElementException if the version cannot be determined
108     */
109     public static Function<ProductItem,String> osVersion() {
110        return new Function<ProductItem,String>() {
111             @Override
112             public String apply(ProductItem productItem) {
113                checkNotNull(productItem,"productItem");
114 
115                final String description = productItem.getDescription();
116                OsFamily family = osFamily().apply(productItem);
117                if (family.equals(OsFamily.CENTOS)) return parseVersion(description, CENTOS);
118                else if(family.equals(OsFamily.DEBIAN)) return parseVersion(description, DEBIAN);
119                else if(family.equals(OsFamily.FEDORA)) return parseVersion(description, FEDORA);
120                else if(family.equals(OsFamily.RHEL)) return parseVersion(description, RHEL);
121                else if(family.equals(OsFamily.UBUNTU)) return parseVersion(description, UBUNTU);
122                else if(family.equals(OsFamily.WINDOWS)) return parseVersion(description, WINDOWS);
123                else throw new NoSuchElementException("No os version for item:"+productItem);
124             }
125         };
126     }
127 
128     private static String parseVersion(String description, String os) {
129        String noOsName = description.replaceFirst(os,"").trim();
130        return noOsName.split(" ")[0];
131     }

133    /**
134     * Parses the item description to determine the number of OS bits
135     * Expects the number to be in parenthesis and to contain the word "bit".
136     * The following return 64: "A (64 bit) OS", "A (64bit) OS"
137     * @return the number of bits
138     * @throws java.util.NoSuchElementException if the number of bits cannot be determined
139     */
140     public static Function<ProductItem,Integer> osBits() {
141        return new Function<ProductItem,Integer>() {
142             @Override
143             public Integer apply(ProductItem productItem) {
144                checkNotNull(productItem,"productItem");
145 
146                Matcher m = OS_BITS_PATTERN.matcher(productItem.getDescription());
147                if (m.matches()) {
148                   return Integer.parseInt(m.group(1));
149                } else {
150                   throw new NoSuchElementException("Cannot determine os-bits for item:"+productItem);
151                }
152             }
153        };
154     }

156   /**
157    * Generates an id for an Image.
158    * @return the generated id
159    */
160    public static Function<ProductItem,String> imageId() {
161       return new Function<ProductItem,String>() {
162          @Override
163          public String apply(ProductItem productItem) {
164             checkNotNull(productItem,"productItem");
165             ProductItemPrice price = ProductItems.price().apply(productItem);
166             return ""+price.getId();
167          }
168       };
169    }
