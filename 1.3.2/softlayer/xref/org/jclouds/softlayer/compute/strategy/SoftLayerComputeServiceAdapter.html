83     private final SoftLayerClient client;
84     private final Supplier<ProductPackage> productPackageSupplier;
85     private final RetryablePredicate<VirtualGuest> loginDetailsTester;
86     private final long guestLoginDelay;
87     private final Pattern cpuPattern;
88     private final Pattern disk0Type;
89     private final float portSpeed;

92     @Inject
93     public SoftLayerComputeServiceAdapter(SoftLayerClient client,
94           VirtualGuestHasLoginDetailsPresent virtualGuestHasLoginDetailsPresent,
95           @Memoized Supplier<ProductPackage> productPackageSupplier, Iterable<ProductItemPrice> prices,
96           @Named(PROPERTY_SOFTLAYER_VIRTUALGUEST_CPU_REGEX) String cpuRegex,
97           @Named(PROPERTY_SOFTLAYER_VIRTUALGUEST_DISK0_TYPE) String disk0Type,
98           @Named(PROPERTY_SOFTLAYER_VIRTUALGUEST_PORT_SPEED) float portSpeed,
99           @Named(PROPERTY_SOFTLAYER_VIRTUALGUEST_LOGIN_DETAILS_DELAY) long guestLoginDelay) {
100       this.client = checkNotNull(client, "client");
101       this.guestLoginDelay = guestLoginDelay;
102       this.productPackageSupplier = checkNotNull(productPackageSupplier, "productPackageSupplier");
103       checkArgument(guestLoginDelay > 500, "guestOrderDelay must be in milliseconds and greater than 500");
104       this.loginDetailsTester = new RetryablePredicate<VirtualGuest>(virtualGuestHasLoginDetailsPresent,
105             guestLoginDelay);
106       this.cpuPattern = Pattern.compile(checkNotNull(cpuRegex, "cpuRegex"));
107       this.prices = checkNotNull(prices, "prices");
108       this.portSpeed = portSpeed;
109       checkArgument(portSpeed > 0, "portSpeed must be greater than zero, often 10, 100, 1000, 10000");
110       this.disk0Type = Pattern.compile(".*" + checkNotNull(disk0Type, "disk0Type") + ".*");
111    }

113    @Override
114    public NodeAndInitialCredentials<VirtualGuest> createNodeWithGroupEncodedIntoName(String group, String name,
115          Template template) {
116       checkNotNull(template, "template was null");
117       checkNotNull(template.getOptions(), "template options was null");
118       checkArgument(template.getOptions().getClass().isAssignableFrom(SoftLayerTemplateOptions.class),
119             "options class %s should have been assignable from SoftLayerTemplateOptions", template.getOptions()
120                   .getClass());
121 
122       String domainName = template.getOptions().as(SoftLayerTemplateOptions.class).getDomainName();
123 
124       VirtualGuest newGuest = VirtualGuest.builder().domain(domainName).hostname(name).build();
125 
126       ProductOrder order = ProductOrder.builder().packageId(productPackageSupplier.get().getId())
127             .location(template.getLocation().getId()).quantity(1).useHourlyPricing(true).prices(getPrices(template))
128             .virtualGuest(newGuest).build();
129 
130       logger.debug(">> ordering new virtualGuest domain(%s) hostname(%s)", domainName, name);
131       ProductOrderReceipt productOrderReceipt = client.getVirtualGuestClient().orderVirtualGuest(order);
132       VirtualGuest result = get(productOrderReceipt.getOrderDetails().getVirtualGuests(), 0);
133       logger.trace("<< virtualGuest(%s)", result.getId());
134 
135       logger.debug(">> awaiting login details for virtualGuest(%s)", result.getId());
136       boolean orderInSystem = loginDetailsTester.apply(result);
137       logger.trace("<< virtualGuest(%s) complete(%s)", result.getId(), orderInSystem);
138 
139       checkState(orderInSystem, "order for guest %s doesn't have login details within %sms", result,
140             Long.toString(guestLoginDelay));
141       result = client.getVirtualGuestClient().getVirtualGuest(result.getId());
142 
143       Password pw = get(result.getOperatingSystem().getPasswords(), 0);
144       return new NodeAndInitialCredentials<VirtualGuest>(result, result.getId() + "", LoginCredentials.builder().user(pw.getUsername()).password(
145             pw.getPassword()).build());
146    }

148    private Iterable<ProductItemPrice> getPrices(Template template) {
149       Builder<ProductItemPrice> result = ImmutableSet.<ProductItemPrice> builder();
150 
151       int imageId = Integer.parseInt(template.getImage().getId());
152       result.add(ProductItemPrice.builder().id(imageId).build());
153 
154       Iterable<String> hardwareIds = Splitter.on(",").split(template.getHardware().getId());
155       for (String hardwareId : hardwareIds) {
156          int id = Integer.parseInt(hardwareId);
157          result.add(ProductItemPrice.builder().id(id).build());
158       }
159       ProductItem uplinkItem = find(productPackageSupplier.get().getItems(),
160             and(capacity(portSpeed), categoryCode("port_speed")));
161       result.add(get(uplinkItem.getPrices(), 0));
162       result.addAll(prices);
163       return result.build();
164    }

166    @Override
167    public Iterable<Iterable<ProductItem>> listHardwareProfiles() {
168       ProductPackage productPackage = productPackageSupplier.get();
169       Set<ProductItem> items = productPackage.getItems();
170       Builder<Iterable<ProductItem>> result = ImmutableSet.<Iterable<ProductItem>> builder();
171       for (ProductItem cpuItem : filter(items, matches(cpuPattern))) {
172          for (ProductItem ramItem : filter(items, categoryCode("ram"))) {
173             for (ProductItem sanItem : filter(items, and(matches(disk0Type), categoryCode("guest_disk0")))) {
174                result.add(ImmutableSet.of(cpuItem, ramItem, sanItem));
175             }
176          }
177       }
178       return result.build();
179    }

213    @Override
214    public void destroyNode(String id) {
215       VirtualGuest guest = getNode(id);
216       if (guest == null)
217          return;
218 
219       if (guest.getBillingItemId() == -1)
220          throw new IllegalStateException(String.format("no billing item for guest(%s) so we cannot cancel the order",
221                id));
222 
223       logger.debug(">> canceling service for guest(%s) billingItem(%s)", id, guest.getBillingItemId());
224       client.getVirtualGuestClient().cancelService(guest.getBillingItemId());
225    }

242    public static class VirtualGuestHasLoginDetailsPresent implements Predicate<VirtualGuest> {
243       private final SoftLayerClient client;
244 
245       @Inject
246       public VirtualGuestHasLoginDetailsPresent(SoftLayerClient client) {
247          this.client = checkNotNull(client, "client was null");
248       }
249 
250       @Override
251       public boolean apply(VirtualGuest guest) {
252          checkNotNull(guest, "virtual guest was null");
253 
254          VirtualGuest newGuest = client.getVirtualGuestClient().getVirtualGuest(guest.getId());
255          boolean hasBackendIp = newGuest.getPrimaryBackendIpAddress() != null;
256          boolean hasPrimaryIp = newGuest.getPrimaryIpAddress() != null;
257          boolean hasPasswords = newGuest.getOperatingSystem() != null
258                && newGuest.getOperatingSystem().getPasswords().size() > 0;
259 
260          return hasBackendIp && hasPrimaryIp && hasPasswords;
261       }
262    }
