27  import com.google.common.collect.ImmutableMultimap;

29  /**
30   * @author Adrian Cole

32  @Test(groups = "unit")
33  public class PipeHttpResponseToTest {
34     PipeHttpResponseTo bash = new PipeHttpResponseTo(Statements.interpret("bash"), "GET", URI
35              .create("https://adriancolehappy.s3.amazonaws.com/java/install"), ImmutableMultimap.of("Host",
36              "adriancolehappy.s3.amazonaws.com", "Date", "Sun, 12 Sep 2010 08:25:19 GMT", "Authorization",
37              "AWS 0ASHDJAS82:JASHFDA="));
38  
39     PipeHttpResponseToBash bash2 = new PipeHttpResponseToBash("GET", URI
40              .create("https://adriancolehappy.s3.amazonaws.com/java/install"), ImmutableMultimap.of("Host",
41              "adriancolehappy.s3.amazonaws.com", "Date", "Sun, 12 Sep 2010 08:25:19 GMT", "Authorization",
42              "AWS 0ASHDJAS82:JASHFDA="));
43  
44     public void testPipeHttpResponseToBashUNIX() {
45        assertEquals(
46                 bash.render(OsFamily.UNIX),
47                 "curl -q -s -S -L --connect-timeout 10 --max-time 600 --retry 20 -X GET -H \"Host: adriancolehappy.s3.amazonaws.com\" -H \"Date: Sun, 12 Sep 2010 08:25:19 GMT\" -H \"Authorization: AWS 0ASHDJAS82:JASHFDA=\" https://adriancolehappy.s3.amazonaws.com/java/install |(bash)\n");
48        assertEquals(bash2.render(OsFamily.UNIX), bash.render(OsFamily.UNIX));
49     }
50  
51     @Test(expectedExceptions = IllegalArgumentException.class)
52     public void testPipeHttpResponseToBashWINDOWS() {
53        bash.render(OsFamily.WINDOWS);
54     }
55  
56     PipeHttpResponseTo untar = new PipeHttpResponseTo(Statements
57              .interpret("{md} {root}stage{fs} &&{cd} {root}stage{fs} &&tar -xpzf -"), "GET", URI
58              .create("https://adriancolehappy.s3.amazonaws.com/java/install"), ImmutableMultimap.of("Host",
59              "adriancolehappy.s3.amazonaws.com", "Date", "Sun, 12 Sep 2010 08:25:19 GMT", "Authorization",
60              "AWS 0ASHDJAS82:JASHFDA="));
61  
62     PipeHttpResponseTo untar2 = new PipeHttpResponseToTarxpzfIntoDirectory("GET", URI
63              .create("https://adriancolehappy.s3.amazonaws.com/java/install"), ImmutableMultimap.of("Host",
64              "adriancolehappy.s3.amazonaws.com", "Date", "Sun, 12 Sep 2010 08:25:19 GMT", "Authorization",
65              "AWS 0ASHDJAS82:JASHFDA="), "{root}stage{fs}");
66  
67     public void testPipeHttpResponseToUntarUNIX() {
68        assertEquals(
69                 untar.render(OsFamily.UNIX),
70                 "curl -q -s -S -L --connect-timeout 10 --max-time 600 --retry 20 -X GET -H \"Host: adriancolehappy.s3.amazonaws.com\" -H \"Date: Sun, 12 Sep 2010 08:25:19 GMT\" -H \"Authorization: AWS 0ASHDJAS82:JASHFDA=\" https://adriancolehappy.s3.amazonaws.com/java/install |(mkdir -p /stage/ &&cd /stage/ &&tar -xpzf -)\n");
71        assertEquals(untar.render(OsFamily.UNIX), untar2.render(OsFamily.UNIX));
72  
73     }
