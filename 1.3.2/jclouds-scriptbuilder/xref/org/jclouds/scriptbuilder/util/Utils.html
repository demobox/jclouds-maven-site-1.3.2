66     /** matches any expression inside curly braces (where the expression does not including an open curly brace) */
67     private static final Pattern pattern = Pattern.compile("\\{([^\\{]+?)\\}");

70     /**
71      * replaces tokens that are expressed as <code>{token}</code>
72      * 
73      * <p/>
74      * ex. if input is "hello {where}"<br/>
75      * and replacements is "where" -> "world" <br/>
76      * then replaceTokens returns "hello world"
77      * 
78      * @param input
79      *           source to replace
80      * @param replacements
81      *           token/value pairs
82      */
83     public static String replaceTokens(String input, Map<String, String> replacements) {
84        Matcher matcher = pattern.matcher(input);
85        StringBuilder builder = new StringBuilder();
86        int i = 0;
87        while (matcher.find()) {
88           String replacement = replacements.get(matcher.group(1));
89           builder.append(input.substring(i, matcher.start()));
90           if (replacement == null)
91              builder.append(matcher.group(0));
92           else
93              builder.append(replacement);
94           i = matcher.end();
95        }
96        builder.append(input.substring(i, input.length()));
97        return builder.toString();
98     }

100    /**
101     * converts a map into variable exports relevant to the specified platform.
102     * <p/>
103     * ex. if variablesInLowerCamelCase is "mavenOpts" -> "-Xms64m -Xmx256m" <br/>
104     * and family is UNIX<br/>
105     * then writeVariableExporters returns literally {@code export MAVEN_OPTS="-Xms64m -Xmx256m"\n}
106     * 
107     * @param variablesInLowerCamelCase
108     *           lower camel keys to values
109     * @param family
110     *           operating system for formatting
111     */
112    public static String writeVariableExporters(Map<String, String> variablesInLowerCamelCase,
113             OsFamily family) {
114       return replaceTokens(writeVariableExporters(variablesInLowerCamelCase), ShellToken
115                .tokenValueMap(family));
116    }

118    /**
119     * converts a map into variable exporters in shell intermediate language.
120     * 
121     * @param variablesInLowerCamelCase
122     *           lower camel keys to values
123     */
124    public static String writeVariableExporters(Map<String, String> variablesInLowerCamelCase) {
125       StringBuilder initializers = new StringBuilder();
126       for (Entry<String, String> entry : variablesInLowerCamelCase.entrySet()) {
127          String key = CaseFormat.LOWER_CAMEL.to(CaseFormat.UPPER_UNDERSCORE, entry.getKey());
128          initializers.append(String.format("{export} %s={vq}%s{vq}{lf}", key, entry.getValue()));
129       }
130       return initializers.toString();
131    }

137    public static String writeFunctionFromResource(String function, OsFamily family) {
138       try {
139          String toReturn = CharStreams.toString(Resources.newReaderSupplier(ClassLoadingUtils.loadResource(Utils.class, String
140                  .format("/functions/%s.%s", function, ShellToken.SH.to(family))), Charsets.UTF_8));
141          String lf = ShellToken.LF.to(family);
142          return toReturn.endsWith(lf) ? toReturn : new StringBuilder(toReturn).append(lf).toString();
143       } catch (IOException e) {
144          throw new FunctionNotFoundException(function, family, e);
145       }
146    }

165    public static final Map<OsFamily, String> OS_TO_POSITIONAL_VAR_PATTERN = ImmutableMap.of(
166             OsFamily.UNIX, "set {key}=$1\nshift\n", OsFamily.WINDOWS, "set {key}=%1\r\nshift\r\n");

168    public static final Map<OsFamily, String> OS_TO_LOCAL_VAR_PATTERN = ImmutableMap.of(
169             OsFamily.UNIX, "set {key}=\"{value}\"\n", OsFamily.WINDOWS, "set {key}={value}\r\n");

171    /**
172     * Writes an initialization statement for use inside a script or a function.
173     * 
174     * @param positionalVariablesInLowerCamelCase
175     *           - transfer the value of args into these statements. Note that there is no check to
176     *           ensure that all source args are indeed present.
177     */
178    public static String writePositionalVars(List<String> positionalVariablesInLowerCamelCase,
179             OsFamily family) {
180       StringBuilder initializers = new StringBuilder();
181       for (String variableInLowerCamelCase : positionalVariablesInLowerCamelCase) {
182          String key = CaseFormat.LOWER_CAMEL.to(CaseFormat.UPPER_UNDERSCORE,
183                   variableInLowerCamelCase);
184          initializers.append(replaceTokens(OS_TO_POSITIONAL_VAR_PATTERN.get(family), ImmutableMap
185                   .of("key", key)));
186       }
187       return initializers.toString();
188    }

190    /**
191     * Ensures that variables come from a known source instead of bleeding in from a profile
192     * 
193     * @param variablesInLowerCamelCase
194     *           - System variables to unset
195     */
196    public static String writeUnsetVariables(List<String> variablesInLowerCamelCase, OsFamily family) {
197       switch (family) {
198          case UNIX:
199             return String.format("unset %s\n", Joiner.on(' ').join(
200                      Iterables.transform(variablesInLowerCamelCase,
201                               FUNCTION_LOWER_CAMEL_TO_UPPER_UNDERSCORE)));
202          case WINDOWS:
203             StringBuilder initializers = new StringBuilder();
204             for (String variableInLowerCamelCase : variablesInLowerCamelCase) {
205                String key = CaseFormat.LOWER_CAMEL.to(CaseFormat.UPPER_UNDERSCORE,
206                         variableInLowerCamelCase);
207                initializers.append(replaceTokens(OS_TO_LOCAL_VAR_PATTERN.get(family), ImmutableMap
208                         .of("key", key, "value", "")));
209             }
210             return initializers.toString();
211          default:
212             throw new UnsupportedOperationException("unsupported os: " + family);
213       }
214 
215    }
