50  /**
51   * Tests behavior of {@code ParseObjectInfoListFromJsonResponse}
52   * 
53   * @author Adrian Cole
54   */
55  @Test(groups = "unit")
56  public class ParseObjectInfoListFromJsonResponseTest {
57  
58     Injector i = Guice.createInjector(new AbstractModule() {
59  
60        @Override
61        protected void configure() {
62           bind(DateAdapter.class).to(Iso8601DateAdapter.class);
63           bind(UriBuilder.class).to(UriBuilderImpl.class);
64        }
65  
66     }, new GsonModule());
67  
68     public void testApplyInputStream() {
69        InputStream is = getClass().getResourceAsStream("/test_list_container.json");
70        Set<ObjectInfo> expects = ImmutableSet.<ObjectInfo> of(ObjectInfoImpl.builder().container("container").name(
71                 "test_obj_1").uri(URI.create("http://localhost/foo/test_obj_1")).hash(
72                 CryptoStreams.hex("4281c348eaf83e70ddce0e07221c3d28")).bytes(14l)
73                 .contentType("application/octet-stream").lastModified(
74                          new SimpleDateFormatDateService().iso8601DateParse("2009-02-03T05:26:32.612Z")).build(),
75                 ObjectInfoImpl.builder().container("container").name("test_obj_2").uri(
76                          URI.create("http://localhost/foo/test_obj_2")).hash(
77                          CryptoStreams.hex("b039efe731ad111bc1b0ef221c3849d0")).bytes(64l).contentType(
78                          "application/octet-stream").lastModified(
79                          new SimpleDateFormatDateService().iso8601DateParse("2009-02-03T05:26:32.612Z")).build());
80        GeneratedHttpRequest<?> request = createMock(GeneratedHttpRequest.class);
81        ListContainerOptions options = new ListContainerOptions();
82        expect(request.getEndpoint()).andReturn(URI.create("http://localhost/foo")).atLeastOnce();
83        expect(request.getArgs()).andReturn(
84                 ImmutableList.<Object> of("container", new ListContainerOptions[] { options })).atLeastOnce();
85        replay(request);
86        ParseObjectInfoListFromJsonResponse parser = i.getInstance(ParseObjectInfoListFromJsonResponse.class);
87        parser.setContext(request);
88        assertEquals(parser.apply(is).toString(), expects.toString());
89     }
