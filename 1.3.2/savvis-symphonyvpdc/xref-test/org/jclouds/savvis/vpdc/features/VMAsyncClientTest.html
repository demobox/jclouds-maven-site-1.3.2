53     public void testAddVMIntoVDCURI() throws SecurityException, NoSuchMethodException, IOException {
54        Method method = VMAsyncClient.class.getMethod("addVMIntoVDC", URI.class, VMSpec.class);
55  
56        CIMOperatingSystem os = Iterables.find(injector.getInstance(Key.get(new TypeLiteral<Set<CIMOperatingSystem>>() {
57        })), new Predicate<CIMOperatingSystem>() {
58  
59           @Override
60           public boolean apply(CIMOperatingSystem arg0) {
61              return arg0.getOsType() == OSType.RHEL_64;
62           }
63  
64        });
65  
66        HttpRequest request = processor.createRequest(method, URI
67                 .create("https://api.symphonyvpdc.savvis.net/rest/api/v0.8/org/11/vdc/22"), VMSpec.builder().name(
68                 "DemoHost-1").networkTierName("VM Tier01").operatingSystem(os).build());
69  
70        assertRequestLineEquals(request,
71                 "GET https://api.symphonyvpdc.savvis.net/rest/api/v0.8/org/11/vdc/22/vApp/ HTTP/1.1");
72        assertNonPayloadHeadersEqual(request, "");
73        assertPayloadEquals(request, Strings2.toStringAndClose(getClass().getResourceAsStream("/vm-default.xml")),
74                 "application/xml", false);
75  
76        assertResponseParserClassEquals(method, request, ParseSax.class);
77        assertSaxResponseParserClassEquals(method, TaskHandler.class);
78        assertExceptionParserClassEquals(method, MapHttp4xxCodesToExceptions.class);
79  
80        checkFilters(request);
81     }
