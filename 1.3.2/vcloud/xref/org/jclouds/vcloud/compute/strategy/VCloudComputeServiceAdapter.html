106       return Iterables.filter(templates.get(), new Predicate<VAppTemplate>() {

127    @Override
128    public Iterable<VApp> listNodes() {
129       // TODO: parallel or cache
130       Builder<VApp> nodes = ImmutableSet.<VApp> builder();
131       for (Org org : nameToOrg.get().values()) {
132          for (ReferenceType vdc : org.getVDCs().values()) {
133             for (ReferenceType resource : client.getVDCClient().getVDC(vdc.getHref()).getResourceEntities().values()) {
134                if (resource.getType().equals(VCloudMediaType.VAPP_XML)) {
135                   addVAppToSetRetryingIfNotYetPresent(nodes, vdc, resource);
136                }
137             }
138          }
139       }
140       return nodes.build();
141    }

143    @VisibleForTesting
144    void addVAppToSetRetryingIfNotYetPresent(Builder<VApp> nodes, ReferenceType vdc, ReferenceType resource) {
145       VApp node = null;
146       int i = 0;
147       while (node == null && i++ < 3) {
148          try {
149             node = client.getVAppClient().getVApp(resource.getHref());
150             nodes.add(node);
151          } catch (NullPointerException e) {
152             logger.warn("vApp %s not yet present in vdc %s", resource.getName(), vdc.getName());
153          }
154       }
155    }

169    @Override
170    public void destroyNode(String id) {
171       URI vappId = URI.create(checkNotNull(id, "node.id"));
172       VApp vApp = cancelAnyRunningTasks(vappId);
173       if (vApp.getStatus() != Status.OFF) {
174          logger.debug(">> powering off VApp vApp(%s), current status: %s", vApp.getName(), vApp.getStatus());
175          try {
176             waitForTask(client.getVAppClient().powerOffVApp(vApp.getHref()));
177             vApp = client.getVAppClient().getVApp(vApp.getHref());
178             logger.debug("<< %s vApp(%s)", vApp.getStatus(), vApp.getName());
179          } catch (IllegalStateException e) {
180             logger.warn(e, "<< %s vApp(%s)", vApp.getStatus(), vApp.getName());
181          }
182          logger.debug(">> undeploying vApp(%s), current status: %s", vApp.getName(), vApp.getStatus());
183          try {
184             waitForTask(client.getVAppClient().undeployVApp(vApp.getHref()));
185             vApp = client.getVAppClient().getVApp(vApp.getHref());
186             logger.debug("<< %s vApp(%s)", vApp.getStatus(), vApp.getName());
187          } catch (IllegalStateException e) {
188             logger.warn(e, "<< %s vApp(%s)", vApp.getStatus(), vApp.getName());
189          }
190       }
191       logger.debug(">> deleting vApp(%s)", vApp.getHref());
192       waitForTask(client.getVAppClient().deleteVApp(vApp.getHref()));
193       logger.debug("<< deleted vApp(%s)", vApp.getHref());
194    }

196    VApp waitForPendingTasksToComplete(URI vappId) {
197       VApp vApp = client.getVAppClient().getVApp(vappId);
198       if (vApp.getTasks().size() == 0)
199          return vApp;
200       for (Task task : vApp.getTasks())
201          waitForTask(task);
202       return client.getVAppClient().getVApp(vappId);
203    }
204 
205    VApp cancelAnyRunningTasks(URI vappId) {
206       VApp vApp = client.getVAppClient().getVApp(vappId);
207       if (vApp.getTasks().size() == 0)
208          return vApp;
209       for (Task task : vApp.getTasks()) {
210          try {
211             client.getTaskClient().cancelTask(task.getHref());
212             waitForTask(task);
213          } catch (TaskInErrorStateException e) {
214          }
215       }
216       return client.getVAppClient().getVApp(vappId);
217 
218    }
219 
220    public void waitForTask(Task task) {
221       if (!successTester.apply(task.getHref())) 
222          throw new TaskStillRunningException(task);
223    }
