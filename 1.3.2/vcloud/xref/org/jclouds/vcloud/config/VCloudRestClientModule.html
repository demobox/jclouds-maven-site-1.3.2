233       return compose(new Function<Map<String, Org>, Map<String, String>>() {
234 
235          @Override
236          public Map<String, String> apply(Map<String, Org> arg0) {
237             Builder<String, String> returnVal = ImmutableMap.<String, String> builder();
238             for (Entry<String, Org> orgr : arg0.entrySet()) {
239                for (String vdc : orgr.getValue().getVDCs().keySet()) {

244          }

381    @Provides
382    @Singleton
383    @org.jclouds.vcloud.endpoints.VCloudLogin
384    protected URI provideAuthenticationURI(VCloudVersionsAsyncClient versionService,
385             @Named(PROPERTY_API_VERSION) String version) throws InterruptedException, ExecutionException,
386             TimeoutException {
387       SortedMap<String, URI> versions = versionService.getSupportedVersions().get(180, TimeUnit.SECONDS);
388       checkState(versions.size() > 0, "No versions present");
389       checkState(versions.containsKey(version), "version " + version + " not present in: " + versions);
390       return versions.get(version);
391    }
