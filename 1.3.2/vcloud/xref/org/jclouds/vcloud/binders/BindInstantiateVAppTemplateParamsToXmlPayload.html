64  @Singleton
65  public class BindInstantiateVAppTemplateParamsToXmlPayload implements MapBinder {

67     protected final String ns;
68     protected final String schema;
69     protected final BindToStringPayload stringBinder;
70     protected final ReferenceType defaultNetwork;
71     protected final FenceMode defaultFenceMode;
72     protected final LoadingCache<URI, VAppTemplate> templateCache;
73     protected final Function<VAppTemplate, String> defaultNetworkNameInTemplate;
74  
75     @Inject
76     public BindInstantiateVAppTemplateParamsToXmlPayload(LoadingCache<URI, VAppTemplate> templateCache,
77              @Network Function<VAppTemplate, String> defaultNetworkNameInTemplate, BindToStringPayload stringBinder,
78              @Named(PROPERTY_VCLOUD_XML_NAMESPACE) String ns, @Named(PROPERTY_VCLOUD_XML_SCHEMA) String schema,
79              @Network ReferenceType network, FenceMode fenceMode) {
80        this.templateCache = templateCache;
81        this.defaultNetworkNameInTemplate = defaultNetworkNameInTemplate;
82        this.ns = ns;
83        this.schema = schema;
84        this.stringBinder = stringBinder;
85        this.defaultNetwork = network;
86        this.defaultFenceMode = fenceMode;
87     }

89     @Override
90     public <R extends HttpRequest> R bindToRequest(R request, Map<String, String> postParams) {
91        checkArgument(checkNotNull(request, "request") instanceof GeneratedHttpRequest<?>,
92              "this binder is only valid for GeneratedHttpRequests!");
93        GeneratedHttpRequest<?> gRequest = (GeneratedHttpRequest<?>) request;
94        checkState(gRequest.getArgs() != null, "args should be initialized at this point");
95        String name = checkNotNull(postParams.remove("name"), "name");
96        URI template = URI.create(checkNotNull(postParams.remove("template"), "template"));
97  
98        Set<NetworkConfig> networkConfig = null;
99  
100       NetworkConfigDecorator networkConfigDecorator = new NetworkConfigDecorator(templateCache.getUnchecked(template),
101             defaultNetwork.getHref(), defaultFenceMode, defaultNetworkNameInTemplate);
102 
103       InstantiateVAppTemplateOptions options = findOptionsInArgsOrNull(gRequest);
104 
105       if (options != null) {
106          if (options.getNetworkConfig().size() > 0)
107             networkConfig = ImmutableSet
108                      .copyOf(transform(options.getNetworkConfig(), networkConfigDecorator));
109       } else {
110          options = new InstantiateVAppTemplateOptions();
111       }
112 
113       if (networkConfig == null)
114          networkConfig = ImmutableSet.of(networkConfigDecorator.apply(null));
115 
116       try {
117          return stringBinder.bindToRequest(request, generateXml(name, options.getDescription(), options.shouldDeploy(),
118                   options.shouldPowerOn(), template, networkConfig));
119       } catch (ParserConfigurationException e) {
120          throw new RuntimeException(e);
121       } catch (FactoryConfigurationError e) {
122          throw new RuntimeException(e);
123       } catch (TransformerException e) {
124          throw new RuntimeException(e);
125       }
126 
127    }

129    @VisibleForTesting
130    Set<Vm> ifCustomizationScriptIsSetGetVmsInTemplate(String customizationScript, final URI template) {
131       Set<Vm> vms = ImmutableSet.of();
132       if (customizationScript != null) {
133          VAppTemplate vAppTemplate = templateCache.getUnchecked(template);
134          checkArgument(vAppTemplate != null, "vAppTemplate %s not found!", template);
135          vms = vAppTemplate.getChildren();
136          checkArgument(vms.size() > 0, "no vms found in vAppTemplate %s", vAppTemplate);
137       }
138       return vms;
139    }

141    protected static final class NetworkConfigDecorator implements Function<NetworkConfig, NetworkConfig> {
142       private final VAppTemplate template;
143       private final URI defaultNetwork;
144       private final FenceMode defaultFenceMode;
145       private final Function<VAppTemplate, String> defaultNetworkNameInTemplate;
146 
147       protected NetworkConfigDecorator(VAppTemplate template, URI defaultNetwork, FenceMode defaultFenceMode,
148                Function<VAppTemplate, String> defaultNetworkNameInTemplate) {
149          this.template = checkNotNull(template, "template");
150          this.defaultNetwork = checkNotNull(defaultNetwork, "defaultNetwork");
151          this.defaultFenceMode = checkNotNull(defaultFenceMode, "defaultFenceMode");
152          this.defaultNetworkNameInTemplate = checkNotNull(defaultNetworkNameInTemplate, "defaultNetworkNameInTemplate");
153       }
154 
155       @Override
156       public NetworkConfig apply(NetworkConfig from) {
157          if (from == null)
158             return new NetworkConfig(defaultNetworkNameInTemplate.apply(template), defaultNetwork, defaultFenceMode);
159          URI network = ifNullDefaultTo(from.getParentNetwork(), defaultNetwork);
160          FenceMode fenceMode = ifNullDefaultTo(from.getFenceMode(), defaultFenceMode);
161          // using conditional statement instead of ifNullDefaultTo so that we lazy invoke the
162          // function, as it is an expensive one.
163          String networkName = from.getNetworkName() != null ? from.getNetworkName() : defaultNetworkNameInTemplate
164                   .apply(template);
165          return new NetworkConfig(networkName, network, fenceMode);
166       }
167    }

169    protected String generateXml(String name, @Nullable String description, boolean deploy, boolean powerOn,
170          URI template, Iterable<NetworkConfig> networkConfig)
171          throws ParserConfigurationException, FactoryConfigurationError, TransformerException {
172       XMLBuilder rootBuilder = buildRoot(name).a("deploy", deploy + "").a("powerOn", powerOn + "");
173       if (description != null)
174          rootBuilder.e("Description").t(description);
175       XMLBuilder instantiationParamsBuilder = rootBuilder.e("InstantiationParams");
176       addNetworkConfig(instantiationParamsBuilder, networkConfig);
177       rootBuilder.e("Source").a("href", template.toASCIIString());
178       rootBuilder.e("AllEULAsAccepted").t("true");
179 
180       Properties outputProperties = new Properties();
181       outputProperties.put(javax.xml.transform.OutputKeys.OMIT_XML_DECLARATION, "yes");
182       return rootBuilder.asString(outputProperties);
183    }
184 
185    protected void addNetworkConfig(XMLBuilder instantiationParamsBuilder,
186          Iterable<NetworkConfig> networkConfig) {
187       XMLBuilder networkConfigBuilder = instantiationParamsBuilder.e("NetworkConfigSection");
188       networkConfigBuilder.e("ovf:Info").t("Configuration parameters for logical networks");
189       for (NetworkConfig n : networkConfig) {
190          XMLBuilder configurationBuilder = networkConfigBuilder.e("NetworkConfig").a("networkName", n.getNetworkName())
191                .e("Configuration");
192          configurationBuilder.e("ParentNetwork").a("href", n.getParentNetwork().toASCIIString());
193          if (n.getFenceMode() != null) {
194             configurationBuilder.e("FenceMode").t(n.getFenceMode().toString());
195          }
196       }
197    }
198 
199    protected XMLBuilder buildRoot(String name) throws ParserConfigurationException, FactoryConfigurationError {
200       return XMLBuilder.create("InstantiateVAppTemplateParams").a("name", name).a("xmlns", ns)
201             .a("xmlns:ovf", "http://schemas.dmtf.org/ovf/envelope/1");
202    }

204    protected InstantiateVAppTemplateOptions findOptionsInArgsOrNull(GeneratedHttpRequest<?> gRequest) {
205       for (Object arg : gRequest.getArgs()) {
206          if (arg instanceof InstantiateVAppTemplateOptions) {
207             return (InstantiateVAppTemplateOptions) arg;
208          } else if (arg instanceof InstantiateVAppTemplateOptions[]) {
209             InstantiateVAppTemplateOptions[] options = (InstantiateVAppTemplateOptions[]) arg;
210             return (options.length > 0) ? options[0] : null;
211          }
212       }
213       return null;
214    }
