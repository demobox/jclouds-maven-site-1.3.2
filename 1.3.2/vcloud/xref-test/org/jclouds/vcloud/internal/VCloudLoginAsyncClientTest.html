53  // NOTE:without testName, this will not call @Before* and fail w/NPE during surefire
54  @Test(groups = "unit", testName = "VCloudLoginAsyncClientTest")
55  public class VCloudLoginAsyncClientTest extends RestClientTest<VCloudLoginAsyncClient> {
56  
57     public void testLogin() throws SecurityException, NoSuchMethodException, IOException {
58        Method method = VCloudLoginAsyncClient.class.getMethod("login");
59        HttpRequest request = processor.createRequest(method);
60  
61        assertEquals(request.getRequestLine(), "POST http://localhost:8080/login HTTP/1.1");
62        assertNonPayloadHeadersEqual(request, HttpHeaders.ACCEPT + ": application/vnd.vmware.vcloud.orgList+xml\n");
63        assertPayloadEquals(request, null, null, false);
64  
65        assertResponseParserClassEquals(method, request, ParseLoginResponseFromHeaders.class);
66        assertSaxResponseParserClassEquals(method, null);
67        assertExceptionParserClassEquals(method, null);
68  
69        checkFilters(request);
70     }
71  
72     @Override
73     protected void checkFilters(HttpRequest request) {
74        assertEquals(request.getFilters().size(), 1);
75        assertEquals(request.getFilters().get(0).getClass(), BasicAuthentication.class);
76     }
77  
78     @Override
79     protected TypeLiteral<RestAnnotationProcessor<VCloudLoginAsyncClient>> createTypeLiteral() {
80        return new TypeLiteral<RestAnnotationProcessor<VCloudLoginAsyncClient>>() {
81        };
82     }

84     @Override
85     protected Module createModule() {
86        return new Module() {

90           }

99        };
