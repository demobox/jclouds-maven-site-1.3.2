65        Method method = VDCAsyncClient.class.getMethod("findVDCInOrgNamed", String.class, String.class);
66        HttpRequest request = processor.createRequest(method, null, "vdc");

68        assertRequestLineEquals(request, "GET https://vcenterprise.bluelock.com/api/v1.0/vdc/1 HTTP/1.1");
