66        assertEquals(result.getTasks(), ImmutableList.of());
67        assertEquals(result.getVAppScopedLocalId(), null);
68        assert result.isOvfDescriptorUploaded();
69        Vm vm = Iterables.getOnlyElement(result.getChildren());
70        assertEquals(vm.getName(), "Ubuntu1004");
71        assertEquals(vm.getHref(), URI.create("https://vcenterprise.bluelock.com/api/v1.0/vAppTemplate/vm-172837194"));

81        GuestCustomizationSection guestC = vm.getGuestCustomizationSection();

118    public static VAppTemplate parseTemplate() {
119       InputStream is = VAppTemplateHandlerTest.class.getResourceAsStream("/vAppTemplate.xml");
120       Injector injector = Guice.createInjector(new SaxParserModule());
121       Factory factory = injector.getInstance(ParseSax.Factory.class);
122       VAppTemplate result = factory.create(injector.getInstance(VAppTemplateHandler.class)).parse(is);
123       return result;
124    }

126    public void testCopyingTemplate() {
127       InputStream is = getClass().getResourceAsStream("/vAppTemplate-copying.xml");
128       Injector injector = Guice.createInjector(new SaxParserModule());
129       Factory factory = injector.getInstance(ParseSax.Factory.class);
130       DateService dateService = injector.getInstance(DateService.class);
131 
132       VAppTemplate result = factory.create(injector.getInstance(VAppTemplateHandler.class)).parse(is);
133       assertEquals(result.getName(), "Ubuntu10.04_v2");
134       assertEquals(result.getHref(), URI
135                .create("https://vcenterprise.bluelock.com/api/v1.0/vAppTemplate/vappTemplate-699683881"));
136       assertEquals(result.getType(), "application/vnd.vmware.vcloud.vAppTemplate+xml");
137       assertEquals(result.getStatus(), Status.UNRESOLVED);
138       assertEquals(result.getVDC(), new ReferenceTypeImpl(null, VCloudMediaType.VDC_XML, URI
139                .create("https://vcenterprise.bluelock.com/api/v1.0/vdc/105186609")));
140       assertEquals(result.getDescription(), null);
141       assertEquals(result.getTasks(), ImmutableList.of(new TaskImpl(URI
142                .create("https://vcenterprise.bluelock.com/api/v1.0/task/q62gxhi32xgd9yrqvr"),
143                "Copying Virtual Application Template Ubuntu10.04_v2(699683881)", TaskStatus.RUNNING, dateService
144                         .iso8601DateParse("2010-09-17T23:20:46.039-04:00"), dateService
145                         .iso8601DateParse("9999-12-31T23:59:59.999-05:00"), dateService
146                         .iso8601DateParse("2010-12-16T23:20:46.039-05:00"), new ReferenceTypeImpl("Ubuntu10.04_v2",
147                         "application/vnd.vmware.vcloud.vAppTemplate+xml",
148                         URI.create("https://vcenterprise.bluelock.com/api/v1.0/vAppTemplate/vappTemplate-699683881")),
149                null)));
150       assertEquals(result.getVAppScopedLocalId(), null);
151       assert result.isOvfDescriptorUploaded();
152       assertEquals(result.getChildren(), ImmutableList.of());
153       assertEquals(result.getNetworkSection(), null);
154 
155    }
