52  //NOTE:without testName, this will not call @Before* and fail w/NPE during surefire
53  @Test(groups = "unit", testName = "ServerToNodeMetadataTest")
54  public class ServerToNodeMetadataTest {
55  
56     @SuppressWarnings("unchecked")
57     @Test
58     public void testApplySetsTagFromNameAndCredentialsFromName() {
59  
60        Map<ServerState, NodeState> serverStateToNodeState = createMock(Map.class);
61        org.jclouds.compute.domain.Image jcImage = createMock(org.jclouds.compute.domain.Image.class);
62        Option dc = new Option(1l, "US-West-1", "US West 1 Datacenter");
63        Option ram = new Option(1l, "512MB", "Server with 512MB RAM");
64  
65        Set<? extends org.jclouds.compute.domain.Image> images = ImmutableSet.of(jcImage);
66        Server server = createMock(Server.class);
67  
68        expect(server.getId()).andReturn(1000l).atLeastOnce();
69        expect(server.getName()).andReturn("group-ff").atLeastOnce();
70        expect(server.getState()).andReturn(ServerState.ON).atLeastOnce();
71  
72        expect(serverStateToNodeState.get(ServerState.ON)).andReturn(NodeState.RUNNING);
73  
74        Location location = new LocationBuilder().scope(LocationScope.ZONE).id("1").description("US-West-1").build();
75        Set< ? extends Location> locations = ImmutableSet.< Location> of( location);
76        
77        expect(server.getIp()).andReturn(new Ip("127.0.0.1"));
78  
79        ServerImage image = createMock(ServerImage.class);
80        expect(server.getImage()).andReturn(image).atLeastOnce();
81        expect(server.getRam()).andReturn(ram).atLeastOnce();
82        expect(server.getDatacenter()).andReturn(dc).atLeastOnce();
83        expect(image.getId()).andReturn(2000l).atLeastOnce();
84        expect(jcImage.getProviderId()).andReturn("2000").atLeastOnce();
85        expect(jcImage.getLocation()).andReturn(location).atLeastOnce();
86        expect(jcImage.getOperatingSystem()).andReturn(createMock(OperatingSystem.class)).atLeastOnce();
87  
88        replay(serverStateToNodeState);
89        replay(server);
90        replay(image);
91        replay(jcImage);
92  
93        ServerToNodeMetadata parser = new ServerToNodeMetadata(serverStateToNodeState, Suppliers
94                 .<Set<? extends Image>> ofInstance(images), Suppliers
95                 .<Set<? extends Hardware>> ofInstance(GoGridHardwareSupplier.H_ALL), Suppliers
96                 .<Set<? extends Location>> ofInstance(locations));
97  
98        NodeMetadata metadata = parser.apply(server);
99        assertEquals(metadata.getLocation(), location);
100       assertEquals(metadata.getImageId(), "2000");
101       assertEquals(metadata.getGroup(), "group");
102 
103       verify(serverStateToNodeState);
104       verify(image);
105       verify(server);
106       verify(jcImage);
107 
108    }
