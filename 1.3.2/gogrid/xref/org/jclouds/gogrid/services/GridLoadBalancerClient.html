19  package org.jclouds.gogrid.services;

21  import java.util.List;
22  import java.util.Set;
23  import java.util.concurrent.TimeUnit;
24  
25  import org.jclouds.concurrent.Timeout;
26  import org.jclouds.gogrid.domain.IpPortPair;
27  import org.jclouds.gogrid.domain.LoadBalancer;
28  import org.jclouds.gogrid.domain.Option;
29  import org.jclouds.gogrid.options.AddLoadBalancerOptions;
30  
31  /**
32   * @author Oleksiy Yarmula
33   */
34  @Timeout(duration = 30, timeUnit = TimeUnit.SECONDS)
35  public interface GridLoadBalancerClient {
36  
37     /**
38      * Returns all load balancers found for the current user.
39      * 
40      * @return load balancers found
41      */
42     Set<LoadBalancer> getLoadBalancerList();
43  
44     /**
45      * Returns the load balancer(s) by unique name(s).
46      * 
47      * Given a name or a set of names, finds one or multiple load balancers.
48      * 
49      * @param names
50      *           to get the load balancers
51      * @return load balancer(s) matching the name(s)
52      */
53     Set<LoadBalancer> getLoadBalancersByName(String... names);
54  
55     /**
56      * Returns the load balancer(s) by unique id(s).
57      * 
58      * Given an id or a set of ids, finds one or multiple load balancers.
59      * 
60      * @param ids
61      *           to get the load balancers
62      * @return load balancer(s) matching the ids
63      */
64     Set<LoadBalancer> getLoadBalancersById(Long... ids);
65  
66     /**
67      * Creates a load balancer with given properties.
68      * 
69      * @param name
70      *           name of the load balancer
71      * @param virtualIp
72      *           virtual IP with IP address set in {@link org.jclouds.gogrid.domain.Ip#ip} and port
73      *           set in {@link IpPortPair#port}
74      * @param realIps
75      *           real IPs to bind the virtual IP to, with IP address set in
76      *           {@link org.jclouds.gogrid.domain.Ip#ip} and port set in {@link IpPortPair#port}
77      * @param options
78      *           options that specify load balancer's type (round robin, least load), persistence
79      *           strategy, or description.
80      * @return created load balancer object
81      */
82     LoadBalancer addLoadBalancer(String name, IpPortPair virtualIp, List<IpPortPair> realIps,
83              AddLoadBalancerOptions... options);
84  
85     /**
86      * Edits the existing load balancer to change the real IP mapping.
87      * 
88      * @param id
89      *           id of the existing load balancer
90      * @param realIps
91      *           real IPs to bind the virtual IP to, with IP address set in
92      *           {@link org.jclouds.gogrid.domain.Ip#ip} and port set in {@link IpPortPair#port}
93      * @return edited object
94      */
95     LoadBalancer editLoadBalancer(long id, List<IpPortPair> realIps);
96  
97     /**
98      * Edits the existing load balancer to change the real IP mapping.
99      * 
100     * @param name
101     *           name of the existing load balancer
102     * @param realIps
103     *           real IPs to bind the virtual IP to, with IP address set in
104     *           {@link org.jclouds.gogrid.domain.Ip#ip} and port set in {@link IpPortPair#port}
105     * @return edited object
106     */
107    LoadBalancer editLoadBalancerNamed(String name, List<IpPortPair> realIps);
108 
109    /**
110     * Deletes the load balancer by Id
111     * 
112     * @param id
113     *           id of the load balancer to delete
114     * @return load balancer before the command is executed
115     */
116    LoadBalancer deleteById(Long id);
117 
118    /**
119     * Deletes the load balancer by name;
120     * 
121     * NOTE: Using this parameter may generate an error if one or more load balancers share a
122     * non-unique name.
123     * 
124     * @param name
125     *           name of the load balancer to be deleted
126     * 
127     * @return load balancer before the command is executed
128     */
129    LoadBalancer deleteByName(String name);
130 
131    /**
132     * Retrieves the list of supported Datacenters to launch servers into. The objects will have
133     * datacenter ID, name and description. In most cases, id or name will be used for
134     * {@link #addLoadBalancer}.
135     * 
136     * @return supported datacenters
137     */
138    Set<Option> getDatacenters();
139 
140 }
