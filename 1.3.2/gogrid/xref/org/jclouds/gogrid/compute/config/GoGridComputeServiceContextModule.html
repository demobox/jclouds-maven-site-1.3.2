121    /**
122     * Finds matches to required configurations. GoGrid's documentation only specifies how much RAM
123     * one can get with different instance types. The # of cores and disk sizes are purely empyrical
124     * and aren't guaranteed. However, these are the matches found: Ram: 512MB, CPU: 1 core, HDD: 28
125     * GB Ram: 1GB, CPU: 1 core, HDD: 57 GB Ram: 2GB, CPU: 1 core, HDD: 113 GB Ram: 4GB, CPU: 3
126     * cores, HDD: 233 GB Ram: 8GB, CPU: 6 cores, HDD: 462 GB (as of March 2010)
127     * 
128     * @return matched size
129     */
130    @Singleton
131    @Provides
132    Function<Hardware, String> provideSizeToRam() {
133       return new Function<Hardware, String>() {
134          @Override
135          public String apply(Hardware hardware) {
136             if (hardware.getRam() >= 8 * 1024 || getCores(hardware) >= 6 || getSpace(hardware) >= 450)
137                return "8GB";
138             if (hardware.getRam() >= 4 * 1024 || getCores(hardware) >= 3 || getSpace(hardware) >= 230)
139                return "4GB";
140             if (hardware.getRam() >= 2 * 1024 || getSpace(hardware) >= 110)
141                return "2GB";
142             if (hardware.getRam() >= 1024 || getSpace(hardware) >= 55)
143                return "1GB";
144             return "512MB"; /* smallest */
145          }
146       };
147    }
