161    /**
162     * All load balancers also have a status attribute to signify the current configuration status of
163     * the device. This status is immutable by the caller and is updated automatically based on state
164     * changes within the service. When a load balancer is first created, it will be placed into a
165     * BUILD status while the configuration is being generated and applied based on the request. Once
166     * the configuration is applied and finalized, it will be in an ACTIVE status. In the event of a
167     * configuration change or update, the status of the load balancer will change to PENDING_UPDATE
168     * to signify configuration changes are in progress but have not yet been finalized. Load
169     * balancers in a SUSPENDED status are configured to reject traffic and will not forward requests
170     * to back-end nodess.
171     */
172    public static enum Status {
173       /**
174        * Load balancer is being provisioned for the first time and configuration is being applied to
175        * bring the service online. The service will not yet be ready to serve incoming requests.
176        */
177       BUILD,
178       /**
179        * Load balancer is configured properly and ready to serve traffic to incoming requests via
180        * the configured virtual IPs.
181        */
182       ACTIVE,
183       /**
184        * Load balancer is online but configuration changes are being applied to update the service
185        * based on a previous request.
186        */
187       PENDING_UPDATE,
188       /**
189        * Load balancer has been taken offline and disabled; contact Support.
190        */
191       SUSPENDED,
192       /**
193        * Load balancer is online but configuration changes are being applied to begin deletion of
194        * the service based on a previous request.
195        */
196       PENDING_DELETE,
197       /**
198        * Load balancers in DELETED status can be displayed for at least 90 days after deletion.
199        */
200       DELETED,
201       /**
202        * The system encountered an error when attempting to configure the load balancer; contact
203        * Support.
204        */
205       ERROR, UNRECOGNIZED;
206

208          try {
209             return valueOf(checkNotNull(status, "status"));

217    /**
218     * All load balancers utilize an algorithm that defines how traffic should be directed between
219     * back-end nodes. The default algorithm for newly created load balancers is RANDOM, which can be
220     * overridden at creation time or changed after the load balancer has been initially provisioned.
221     * The algorithm name is to be constant within a major revision of the load balancing API, though
222     * new algorithms may be created with a unique algorithm name within a given major revision of
223     * the service API.
224     */
225    public static enum Algorithm {
226       /**
227        * The node with the lowest number of connections will receive requests.
228        */
229       LEAST_CONNECTIONS,
230       /**
231        * Back-end servers are selected at random.
232        */
233       RANDOM,
234       /**
235        * Connections are routed to each of the back-end servers in turn.
236        */
237       ROUND_ROBIN,
238       /**
239        * Each request will be assigned to a node based on the number of concurrent connections to
240        * the node and its weight.
241        */
242       WEIGHTED_LEAST_CONNECTIONS,
243       /**
244        * A round robin algorithm, but with different proportions of traffic being directed to the
245        * back-end nodes. Weights must be defined as part of the load balancer's node configuration.
246        */
247       WEIGHTED_ROUND_ROBIN, UNRECOGNIZED;
248 
249       public static Algorithm fromValue(String algorithm) {
250          try {
251             return valueOf(checkNotNull(algorithm, "algorithm"));
252          } catch (IllegalArgumentException e) {
253             return UNRECOGNIZED;
254          }
255       }
256    }
