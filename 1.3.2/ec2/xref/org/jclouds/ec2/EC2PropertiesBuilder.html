24  import static org.jclouds.ec2.reference.EC2Constants.PROPERTY_EC2_AMI_OWNERS;

28  import java.util.Properties;

30  import org.jclouds.PropertiesBuilder;
31  
32  /**
33   * Builds properties used in EC2 Clients
34   * 
35   * @author Adrian Cole
36   */
37  public class EC2PropertiesBuilder extends PropertiesBuilder {
38     @Override
39     protected Properties defaultProperties() {
40        Properties properties = super.defaultProperties();
41        properties.setProperty(PROPERTY_AUTH_TAG, "AWS");
42        properties.setProperty(PROPERTY_HEADER_TAG, "amz");
43        properties.setProperty(PROPERTY_API_VERSION, EC2AsyncClient.VERSION);
44        properties.setProperty(PROPERTY_EC2_AMI_OWNERS, "*");
45        properties.setProperty(PROPERTY_EC2_TIMEOUT_SECURITYGROUP_PRESENT, "500");
46        properties.setProperty(PROPERTY_EC2_AUTO_ALLOCATE_ELASTIC_IPS, "false");
47        return properties;
48     }
49  
50     public EC2PropertiesBuilder(Properties properties) {
51        super(properties);
52     }
53  
54     public EC2PropertiesBuilder() {
55        super();
56     }
57  }
