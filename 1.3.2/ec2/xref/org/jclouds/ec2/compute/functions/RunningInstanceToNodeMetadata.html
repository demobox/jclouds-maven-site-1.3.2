72  public class RunningInstanceToNodeMetadata implements Function<RunningInstance, NodeMetadata> {

83     @Inject
84     protected RunningInstanceToNodeMetadata(Map<InstanceState, NodeState> instanceToNodeState,
85              Map<String, Credentials> credentialStore, Supplier<LoadingCache<RegionAndName, ? extends Image>> imageMap,
86              @Memoized Supplier<Set<? extends Location>> locations, @Memoized Supplier<Set<? extends Hardware>> hardware) {
87        this.locations = checkNotNull(locations, "locations");
88        this.hardware = checkNotNull(hardware, "hardware");
89        this.imageMap = checkNotNull(imageMap, "imageMap");
90        this.instanceToNodeState = checkNotNull(instanceToNodeState, "instanceToNodeState");
91        this.credentialStore = checkNotNull(credentialStore, "credentialStore");

94     @Override
95     public NodeMetadata apply(RunningInstance instance) {
96        if (instance == null || instance.getId() == null)
97           return null;
98        NodeMetadataBuilder builder = new NodeMetadataBuilder();
99        builder = buildInstance(instance, builder);
100       return builder.build();
101    }

103    protected NodeMetadataBuilder buildInstance(final RunningInstance instance, NodeMetadataBuilder builder) {
104       builder.providerId(instance.getId());
105       builder.id(instance.getRegion() + "/" + instance.getId());
106       String group = getGroupForInstance(instance);
107       builder.group(group);
108       // standard convention from aws-ec2, which might not be re-used outside.
109       if (instance.getPrivateDnsName() != null)
110          builder.hostname(instance.getPrivateDnsName().replaceAll("\\..*", ""));
111       addCredentialsForInstance(builder, instance);
112       builder.state(instanceToNodeState.get(instance.getInstanceState()));
113 
114       // collect all ip addresses into one bundle in case the api mistakenly put a private address
115       // into the public address field
116       Builder<String> addressesBuilder = ImmutableSet.<String> builder();
117       if (Strings.emptyToNull(instance.getIpAddress()) != null)
118          addressesBuilder.add(instance.getIpAddress());
119       if (Strings.emptyToNull(instance.getPrivateIpAddress()) != null)
120          addressesBuilder.add(instance.getPrivateIpAddress());
121 
122       Set<String> addresses = addressesBuilder.build();
123 
124       builder.publicAddresses(filter(addresses, not(IsPrivateIPAddress.INSTANCE)));
125       builder.privateAddresses(filter(addresses, IsPrivateIPAddress.INSTANCE));
126       builder.hardware(parseHardware(instance));
127       Location location = getLocationForAvailabilityZoneOrRegion(instance);
128       builder.location(location);
129       builder.imageId(instance.getRegion() + "/" + instance.getImageId());
130 
131       // extract the operating system from the image
132       RegionAndName regionAndName = new RegionAndName(instance.getRegion(), instance.getImageId());
133       try {
134          Image image = imageMap.get().getUnchecked(regionAndName);
135          if (image != null)
136             builder.operatingSystem(image.getOperatingSystem());
137       } catch (CacheLoader.InvalidCacheLoadException e) {
138          logger.debug("image not found for %s: %s", regionAndName, e);
139       } catch (UncheckedExecutionException e) {
140          logger.debug("error getting image for %s: %s", regionAndName, e);
141       }
142       return builder;
143    }
144 
145    protected void addCredentialsForInstance(NodeMetadataBuilder builder, RunningInstance instance) {
146       builder.credentials(LoginCredentials.fromCredentials(credentialStore.get("node#" + instance.getRegion() + "/"
147             + instance.getId())));
148    }
149 
150    protected Hardware parseHardware(final RunningInstance instance) {
151       Hardware hardware = getHardwareForInstance(instance);
152 
153       if (hardware != null) {
154          hardware = HardwareBuilder.fromHardware(hardware).volumes(addEBS(instance, hardware.getVolumes())).build();
155       }
156       return hardware;
157    }

159    @VisibleForTesting
160    static List<Volume> addEBS(final RunningInstance instance, Iterable<? extends Volume> volumes) {
161       Iterable<Volume> ebsVolumes = Iterables.transform(instance.getEbsBlockDevices().entrySet(),
162                new Function<Entry<String, BlockDevice>, Volume>() {
163 
164                   @Override
165                   public Volume apply(Entry<String, BlockDevice> from) {
166                      return new VolumeImpl(from.getValue().getVolumeId(), Volume.Type.SAN, null, from.getKey(),
167                               instance.getRootDeviceName() != null
168                                        && instance.getRootDeviceName().equals(from.getKey()), true);
169                   }
170                });
171 
172       if (instance.getRootDeviceType() == RootDeviceType.EBS) {
173          volumes = Iterables.filter(volumes, new Predicate<Volume>() {
174 
175             @Override
176             public boolean apply(Volume input) {
177                return !input.isBootDevice();
178             }
179 
180          });
181 
182       }
183       return Lists.newArrayList(Iterables.concat(volumes, ebsVolumes));
184 
185    }

187    @VisibleForTesting
188    String getGroupForInstance(final RunningInstance instance) {
189       String group = null;
190       try {
191          group = Iterables.getOnlyElement(Iterables.filter(instance.getGroupIds(), new Predicate<String>() {
192 
193             @Override
194             public boolean apply(String input) {
195                return input.startsWith("jclouds#") && input.endsWith("#" + instance.getRegion());
196             }
197 
198          })).substring(8).replaceAll("#" + instance.getRegion() + "$", "");
199       } catch (NoSuchElementException e) {
200          logger.debug("no group parsed from %s's security groups: %s", instance.getId(), instance.getGroupIds());
201       } catch (IllegalArgumentException e) {
202          logger.debug("too many groups match %s; %s's security groups: %s", "jclouds#", instance.getId(), instance
203                   .getGroupIds());
204       }
205       return group;
206    }

208    @VisibleForTesting
209    Hardware getHardwareForInstance(final RunningInstance instance) {
210       try {
211          return Iterables.find(hardware.get(), new Predicate<Hardware>() {
212 
213             @Override
214             public boolean apply(Hardware input) {
215                return input.getId().equals(instance.getInstanceType());
216             }
217 
218          });
219       } catch (NoSuchElementException e) {
220          logger.debug("couldn't match instance type %s in: %s", instance.getInstanceType(), hardware.get());
221          return null;
222       }
223    }

225    private Location getLocationForAvailabilityZoneOrRegion(final RunningInstance instance) {
226       Location location = findLocationWithId(instance.getAvailabilityZone());
227       if (location == null)
228          location = findLocationWithId(instance.getRegion());
229       return location;
230    }
231 
232    private Location findLocationWithId(final String locationId) {
233       if (locationId == null)
234          return null;
235       try {
236          Location location = Iterables.find(locations.get(), new Predicate<Location>() {
237 
238             @Override
239             public boolean apply(Location input) {
240                return input.getId().equals(locationId);
241             }
242 
243          });
244          return location;
245 
246       } catch (NoSuchElementException e) {
247          logger.debug("couldn't match instance location %s in: %s", locationId, locations.get());
248          return null;
249       }
250    }
