282    @Test(enabled = false, dependsOnMethods = "testCreateAndAttachVolume")
283    void testBundleInstance() {
284       SshClient ssh = sshFactory.create(new IPSocket(instance.getIpAddress(), 22),
285             LoginCredentials.builder().user("ubuntu").privateKey(keyPair.getKeyMaterial()).build());
286       try {
287          ssh.connect();
288       } catch (SshException e) {// try twice in case there is a network timeout
289          try {
290             Thread.sleep(10 * 1000);
291          } catch (InterruptedException e1) {
292          }
293          ssh.connect();
294       }
295       try {
296          System.out.printf("%d: %s writing ebs script%n", System.currentTimeMillis(), instance.getId());
297          String script = "/tmp/mkebsboot-init.sh";
298          ssh.put(script, Payloads.newStringPayload(mkEbsBoot));
299 
300          System.out.printf("%d: %s launching ebs script%n", System.currentTimeMillis(), instance.getId());
301          ssh.exec("chmod 755 " + script);
302          ssh.exec(script + " init");
303          ExecResponse output = ssh.exec("sudo " + script + " start");
304          System.out.println(output);
305          output = ssh.exec(script + " status");
306 
307          assert !output.getOutput().trim().equals("") : output;
308 
309          RetryablePredicate<String> scriptTester = new RetryablePredicate<String>(new ScriptTester(ssh, SCRIPT_END),
310                600, 10, TimeUnit.SECONDS);
311          scriptTester.apply(script);
312       } finally {
313          if (ssh != null)
314             ssh.disconnect();
315       }
316    }

318    public static class ScriptTester implements Predicate<String> {
319       private final SshClient ssh;
320       private final String endMatches;
321 
322       public ScriptTester(SshClient ssh, String endMatches) {
323          this.ssh = ssh;
324          this.endMatches = endMatches;
325       }
326 
327       @Override
328       public boolean apply(String script) {
329          System.out.printf("%d: %s testing status%n", System.currentTimeMillis(), script);
330          ExecResponse output = ssh.exec(script + " status");
331          if (output.getOutput().trim().equals("")) {
332             output = ssh.exec(script + " tail");
333             String stdout = output.getOutput().trim();
334             if (stdout.contains(endMatches)) {
335                return true;
336             } else {
337                output = ssh.exec(script + " tailerr");
338                String stderr = output.getOutput().trim();
339                throw new RuntimeException(String.format(
340                      "script %s ended without token: stdout.log: [%s]; stderr.log: [%s]; ", script, stdout, stderr));
341             }
342          }
343          return false;
344       }
345 
346    }

349    void testAMIFromBundle() {
350       volume = Iterables.getOnlyElement(client.getElasticBlockStoreServices().describeVolumesInRegion(
351             volume.getRegion(), volume.getId()));
352       if (volume.getAttachments().size() > 0) {
353          // should be cleanly unmounted, so force is not necessary.

355          System.out.printf("%d: %s awaiting detachment to complete%n", System.currentTimeMillis(), volume.getId());
356          assert volumeTester.apply(volume);
357       } else {
358          attachment = null; // protect test closer so that it doesn't try to
359          // detach
360       }
361       snapshot = client.getElasticBlockStoreServices().createSnapshotInRegion(volume.getRegion(), volume.getId(),
362             withDescription("EBS Ubuntu Hardy"));
363 
364       System.out.printf("%d: %s awaiting snapshot to complete%n", System.currentTimeMillis(), snapshot.getId());
365 
366       assert snapshotTester.apply(snapshot);
367       Image image = Iterables.getOnlyElement(client.getAMIServices().describeImagesInRegion(snapshot.getRegion(),
368             imageIds(IMAGE_ID)));
369       String description = image.getDescription() == null ? "jclouds" : image.getDescription();
370 
371       System.out.printf("%d: %s creating ami from snapshot%n", System.currentTimeMillis(), snapshot.getId());
372 
373       String amiId = client.getAMIServices().registerUnixImageBackedByEbsInRegion(
374             snapshot.getRegion(),
375             "ebsboot-" + image.getId(),
376             snapshot.getId(),
377             withKernelId(image.getKernelId()).withRamdisk(image.getRamdiskId()).withDescription(description)
378                   .asArchitecture(Architecture.I386));
379       try {
380          ebsImage = Iterables.getOnlyElement(client.getAMIServices().describeImagesInRegion(snapshot.getRegion(),
381                imageIds(amiId)));
382       } catch (AWSResponseException e) {
383          // TODO add a retry handler for this HTTP code 400 and the below error
384          if (e.getError().getClass().equals("InvalidAMIID.NotFound"))
385             ebsImage = Iterables.getOnlyElement(client.getAMIServices().describeImagesInRegion(snapshot.getRegion(),
386                   imageIds(amiId)));
387          else
388             throw e;
389       }
390       verifyImage();
391    }
