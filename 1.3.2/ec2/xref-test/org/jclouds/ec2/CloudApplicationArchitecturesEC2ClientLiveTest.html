356    private RunningInstance blockUntilWeCanSshIntoInstance(RunningInstance instance) throws UnknownHostException {
357       System.out.printf("%d: %s awaiting instance to run %n", System.currentTimeMillis(), instance.getId());
358       assert runningTester.apply(instance);
359 
360       instance = getInstance(instance.getId());
361 
362       System.out
363             .printf("%d: %s awaiting instance to have ip assigned %n", System.currentTimeMillis(), instance.getId());
364       assert hasIpTester.apply(instance);
365 
366       System.out.printf("%d: %s awaiting ssh service to start%n", System.currentTimeMillis(), instance.getIpAddress());
367       assert socketTester.apply(new IPSocket(instance.getIpAddress(), 22));
368 
369       System.out.printf("%d: %s ssh service started%n", System.currentTimeMillis(), instance.getDnsName());
370       sshPing(instance);
371       System.out.printf("%d: %s ssh connection made%n", System.currentTimeMillis(), instance.getId());
372 
373       System.out.printf("%d: %s awaiting http service to start%n", System.currentTimeMillis(), instance.getIpAddress());
374       assert socketTester.apply(new IPSocket(instance.getIpAddress(), 80));
375       System.out.printf("%d: %s http service started%n", System.currentTimeMillis(), instance.getDnsName());
376       return instance;
377    }

389    /**
390     * this tests "personality" as the file looked up was sent during instance creation
391     * 
392     * @throws UnknownHostException
393     */
394    private void sshPing(RunningInstance newDetails) throws UnknownHostException {
395       try {
396          doCheckKey(newDetails);
397       } catch (SshException e) {// try twice in case there is a network timeout
398          try {
399             Thread.sleep(10 * 1000);
400          } catch (InterruptedException e1) {
401          }
402          doCheckKey(newDetails);
403       }
404    }

406    private void doCheckKey(RunningInstance newDetails) throws UnknownHostException {
407       doCheckKey(newDetails.getIpAddress());
408    }

410    private void doCheckKey(String address) {
411       SshClient ssh = sshFactory.create(new IPSocket(address, 22),
412             LoginCredentials.builder().user("root").privateKey(keyPair.getKeyMaterial()).build());
413       try {
414          ssh.connect();
415          ExecResponse hello = ssh.exec("echo hello");
416          assertEquals(hello.getOutput().trim(), "hello");
417       } finally {
418          if (ssh != null)
419             ssh.disconnect();
420       }
421    }
