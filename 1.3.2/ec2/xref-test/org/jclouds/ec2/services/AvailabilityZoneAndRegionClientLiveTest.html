50  /**
51   * Tests behavior of {@code AvailabilityZoneAndRegionClient}

53   * @author Adrian Cole
54   */
55  @Test(groups = "live", singleThreaded = true, testName = "AvailabilityZoneAndRegionClientLiveTest")
56  public class AvailabilityZoneAndRegionClientLiveTest extends BaseVersionedServiceLiveTest {
57     public AvailabilityZoneAndRegionClientLiveTest() {
58        provider = "ec2";
59     }
60     
61     private AvailabilityZoneAndRegionClient client;
62     private RestContext<EC2Client, EC2AsyncClient> context;
63  
64     @BeforeGroups(groups = { "live" })
65     public void setupClient() {
66        setupCredentials();
67        Properties overrides = setupProperties();
68        context = new ComputeServiceContextFactory().createContext(provider,
69              ImmutableSet.<Module> of(new Log4JLoggingModule()), overrides).getProviderSpecificContext();
70        client = context.getApi().getAvailabilityZoneAndRegionServices();
71     }
72  
73     public void testDescribeAvailabilityZones() {
74        for (String region : Lists.newArrayList(null, Region.EU_WEST_1, Region.US_EAST_1, Region.US_WEST_1,
75              Region.AP_SOUTHEAST_1)) {
76           Set<AvailabilityZoneInfo> allResults = client.describeAvailabilityZonesInRegion(region);
77           assertNotNull(allResults);
78           assert allResults.size() >= 2 : allResults.size();
79           Iterator<AvailabilityZoneInfo> iterator = allResults.iterator();
80           String id1 = iterator.next().getZone();
81           String id2 = iterator.next().getZone();
82           Set<AvailabilityZoneInfo> twoResults = client.describeAvailabilityZonesInRegion(region,
83                 availabilityZones(id1, id2));
84           assertNotNull(twoResults);
85           assertEquals(twoResults.size(), 2);
86           iterator = allResults.iterator();
87           assertEquals(iterator.next().getZone(), id1);
88           assertEquals(iterator.next().getZone(), id2);
89        }
90     }
91  
92     public void testDescribeRegions() {
93        SortedMap<String, URI> allResults = Maps.newTreeMap();
94        allResults.putAll(client.describeRegions());
95        assertNotNull(allResults);
96        assert allResults.size() >= 2 : allResults.size();
97        Iterator<Entry<String, URI>> iterator = allResults.entrySet().iterator();
98        String r1 = iterator.next().getKey();
99        String r2 = iterator.next().getKey();
100       SortedMap<String, URI> twoResults = Maps.newTreeMap();
101       twoResults.putAll(client.describeRegions(regions(r1, r2)));
102       assertNotNull(twoResults);
103       assertEquals(twoResults.size(), 2);
104       iterator = twoResults.entrySet().iterator();
105       assertEquals(iterator.next().getKey(), r1);
106       assertEquals(iterator.next().getKey(), r2);
107    }
