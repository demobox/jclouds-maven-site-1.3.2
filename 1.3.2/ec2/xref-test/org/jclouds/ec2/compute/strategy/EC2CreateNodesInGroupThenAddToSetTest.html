69     public void testIpAllocationThenAfterNodeRunningAssignThenUpdateCache() {
70        Location location = ZONE_AP_SOUTHEAST_1A;
71        String region = "ap-southeast-1";
72        String zone = "ap-southeast-1a";
73  
74        String imageId = "ami1";
75        String instanceCreatedId = "instance1";
76        // setup mocks
77        TemplateBuilder templateBuilder = createMock(TemplateBuilder.class);
78        EC2CreateNodesInGroupThenAddToSet strategy = setupStrategy(templateBuilder);
79        InputParams input = new InputParams(location);
80        InstanceClient instanceClient = createMock(InstanceClient.class);
81        ElasticIPAddressClient ipClient = createMock(ElasticIPAddressClient.class);
82        RunInstancesOptions ec2Options = createMock(RunInstancesOptions.class);
83        RunningInstance instance = createMock(RunningInstance.class);
84        Reservation<? extends RunningInstance> reservation = new Reservation<RunningInstance>(region, ImmutableSet
85                 .<String> of(), ImmutableSet.<RunningInstance> of(instance), "ownerId", "requesterId", "reservationId");
86        NodeMetadata nodeMetadata = createMock(NodeMetadata.class);
87  
88        // enable auto-allocation
89        strategy.autoAllocateElasticIps = true;
90  
91        // setup expectations
92        expect(templateBuilder.fromTemplate(input.template)).andReturn(templateBuilder);
93        expect(templateBuilder.build()).andReturn(input.template);
94        expect(strategy.client.getInstanceServices()).andReturn(instanceClient).atLeastOnce();
95        expect(
96                 strategy.createKeyPairAndSecurityGroupsAsNeededAndReturncustomize.execute(region, input.tag,
97                          input.template)).andReturn(ec2Options);
98        expect(strategy.client.getElasticIPAddressServices()).andReturn(ipClient).atLeastOnce();
99  
100       expect(input.template.getLocation()).andReturn(input.location).atLeastOnce();
101       expect(input.template.getImage()).andReturn(input.image).atLeastOnce();
102       expect(input.image.getProviderId()).andReturn(imageId).atLeastOnce();
103 
104       // differences when ip allocation
105       expect(ipClient.allocateAddressInRegion(region)).andReturn("1.1.1.1");
106       expect(strategy.runningInstanceToNodeMetadata.apply(instance)).andReturn(nodeMetadata).atLeastOnce();
107       expect(strategy.nodeRunning.apply(nodeMetadata)).andReturn(true);
108       ipClient.associateAddressInRegion(region, "1.1.1.1", instanceCreatedId);
109       strategy.elasticIpCache.put(new RegionAndName(region, instanceCreatedId), "1.1.1.1");
110 
111       expect(instanceClient.runInstancesInRegion(region, zone, imageId, 1, input.count, ec2Options)).andReturn(
112                Reservation.class.cast(reservation));
113       expect(instance.getId()).andReturn(instanceCreatedId).atLeastOnce();
114       // simulate a lazy credentials fetch
115       Credentials creds = new Credentials("foo", "bar");
116       expect(strategy.instanceToCredentials.apply(instance)).andReturn(creds);
117       expect(instance.getRegion()).andReturn(region).atLeastOnce();
118       expect(strategy.credentialStore.put("node#" + region + "/" + instanceCreatedId, creds)).andReturn(null);
119 
120       expect(strategy.instancePresent.apply(new RegionAndName(region, instanceCreatedId))).andReturn(true);
121       expect(input.template.getOptions()).andReturn(input.options).atLeastOnce();
122 
123       expect(
124                strategy.utils.customizeNodesAndAddToGoodMapOrPutExceptionIntoBadMap(eq(input.options),
125                         containsNodeMetadata(nodeMetadata), eq(input.nodes), eq(input.badNodes),
126                         eq(input.customization))).andReturn(null);
127 
128       // replay mocks
129       replay(templateBuilder);
130       replay(instanceClient);
131       replay(ipClient);
132       replay(ec2Options);
133       replay(instance);
134       replay(nodeMetadata);

136       replayStrategy(strategy);
