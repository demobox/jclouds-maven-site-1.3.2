101    public void testExecuteWithDefaultOptionsEC2() throws SecurityException, NoSuchMethodException {
102       // setup constants
103       String region = Region.AP_SOUTHEAST_1;
104       String group = "group";
105       Hardware size = EC2HardwareBuilder.m1_small().build();
106       String systemGeneratedKeyPairName = "systemGeneratedKeyPair";
107       String generatedGroup = "group";
108       Set<String> generatedGroups = ImmutableSet.of(generatedGroup);
109 
110       // create mocks
111       CreateKeyPairAndSecurityGroupsAsNeededAndReturnRunOptions strategy = createMock(
112             CreateKeyPairAndSecurityGroupsAsNeededAndReturnRunOptions.class,
113             new Method[] {
114                   CreateKeyPairAndSecurityGroupsAsNeededAndReturnRunOptions.class
115                         .getDeclaredMethod("getOptionsProvider"),
116                   CreateKeyPairAndSecurityGroupsAsNeededAndReturnRunOptions.class.getDeclaredMethod(
117                         "createNewKeyPairUnlessUserSpecifiedOtherwise", String.class, String.class,
118                         TemplateOptions.class),
119                   CreateKeyPairAndSecurityGroupsAsNeededAndReturnRunOptions.class.getDeclaredMethod(
120                         "getSecurityGroupsForTagAndOptions", String.class, String.class, TemplateOptions.class) });
121 
122       EC2TemplateOptions options = createMock(EC2TemplateOptions.class);
123       Template template = createMock(Template.class);
124 
125       // setup expectations
126       expect(strategy.getOptionsProvider()).andReturn(OPTIONS_PROVIDER);
127       expect(template.getHardware()).andReturn(size).atLeastOnce();
128       expect(template.getOptions()).andReturn(options).atLeastOnce();
129       expect(options.getBlockDeviceMappings()).andReturn(ImmutableSet.<BlockDeviceMapping> of()).atLeastOnce();
130       expect(strategy.createNewKeyPairUnlessUserSpecifiedOtherwise(region, group, options)).andReturn(
131             systemGeneratedKeyPairName);
132       expect(strategy.getSecurityGroupsForTagAndOptions(region, group, options)).andReturn(generatedGroups);
133       expect(options.getUserData()).andReturn(null);
134 
135       // replay mocks
136       replay(options);
137       replay(template);
138       replay(strategy);
139 
140       // run
141       RunInstancesOptions customize = strategy.execute(region, group, template);
142       assertEquals(customize.buildQueryParameters(), ImmutableMultimap.<String, String> of());
143       assertEquals(
144             customize.buildFormParameters().entries(),
145             ImmutableMultimap.<String, String> of("InstanceType", size.getProviderId(), "SecurityGroup.1",
146                   generatedGroup, "KeyName", systemGeneratedKeyPairName).entries());
147       assertEquals(customize.buildMatrixParameters(), ImmutableMultimap.<String, String> of());
148       assertEquals(customize.buildRequestHeaders(), ImmutableMultimap.<String, String> of());
149       assertEquals(customize.buildStringPayload(), null);
150 
151       // verify mocks
152       verify(options);
153       verify(template);
154       verify(strategy);
155    }
