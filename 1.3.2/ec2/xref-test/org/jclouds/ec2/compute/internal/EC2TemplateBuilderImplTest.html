70     @Override
71     protected EC2TemplateBuilderImpl createTemplateBuilder(final Image knownImage,
72           @Memoized Supplier<Set<? extends Location>> locations, final @Memoized Supplier<Set<? extends Image>> images,
73           @Memoized Supplier<Set<? extends Hardware>> sizes, Location defaultLocation,
74           Provider<TemplateOptions> optionsProvider, Provider<TemplateBuilder> templateBuilderProvider) {
75  
76        LoadingCache<RegionAndName, ? extends Image> imageMap;
77        if (knownImage != null) {
78           final RegionAndName knownRegionAndName = new RegionAndName(knownImage.getLocation().getId(), knownImage.getProviderId());
79  
80           imageMap = CacheBuilder.newBuilder().build(new CacheLoader<RegionAndName, Image>() {
81              @Override
82              public Image load(RegionAndName from) {
83                 return from.equals(knownRegionAndName) ? knownImage : null;
84              }
85  
86           });
87           
88        } else {
89           imageMap = CacheBuilder.newBuilder().build(CacheLoader.from(Functions.forMap(
90                    ImagesToRegionAndIdMap.imagesToMap(images.get()))));
91        }
92  
93        return new EC2TemplateBuilderImpl(locations, images, sizes, Suppliers.ofInstance(defaultLocation),
94              optionsProvider, templateBuilderProvider, Suppliers.<LoadingCache<RegionAndName, ? extends Image>>ofInstance(imageMap));
95     }

106    public void testParseOnDemand() {
107       Location location = new LocationBuilder().scope(LocationScope.REGION).id("region").description("region").build();
108 
109       Supplier<Set<? extends Location>> locations = Suppliers.<Set<? extends Location>> ofInstance(ImmutableSet
110             .<Location> of(location));
111       Supplier<Set<? extends Image>> images = Suppliers.<Set<? extends Image>> ofInstance(Sets
112             .<Image> newLinkedHashSet());
113       Supplier<Set<? extends Hardware>> sizes = Suppliers.<Set<? extends Hardware>> ofInstance(ImmutableSet
114             .<Hardware> of(c1_medium().build()));
115 
116       Provider<TemplateOptions> optionsProvider = createMock(Provider.class);
117       Provider<TemplateBuilder> templateBuilderProvider = createMock(Provider.class);
118       TemplateOptions defaultOptions = createMock(TemplateOptions.class);
119       Image knownImage = createMock(Image.class);
120       OperatingSystem os = createMock(OperatingSystem.class);
121 
122       expect(optionsProvider.get()).andReturn(defaultOptions);
123 
124       expect(knownImage.getId()).andReturn("region/ami").atLeastOnce();
125       expect(knownImage.getLocation()).andReturn(location).atLeastOnce();
126       expect(knownImage.getName()).andReturn(null).atLeastOnce();
127       expect(knownImage.getDescription()).andReturn(null).atLeastOnce();
128       expect(knownImage.getVersion()).andReturn(null).atLeastOnce();
129       expect(knownImage.getProviderId()).andReturn("ami").atLeastOnce();
130 
131       expect(knownImage.getOperatingSystem()).andReturn(os).atLeastOnce();
132 
133       expect(os.getName()).andReturn(null).atLeastOnce();
134       expect(os.getVersion()).andReturn(null).atLeastOnce();
135       expect(os.getFamily()).andReturn(null).atLeastOnce();
136       expect(os.getDescription()).andReturn(null).atLeastOnce();
137       expect(os.getArch()).andReturn("paravirtual").atLeastOnce();
138       expect(os.is64Bit()).andReturn(false).atLeastOnce();
139 
140       replay(knownImage);
141       replay(os);
142       replay(defaultOptions);
143       replay(optionsProvider);
144       replay(templateBuilderProvider);
145 
146       TemplateBuilderImpl template = createTemplateBuilder(knownImage, locations, images, sizes, location,
147             optionsProvider, templateBuilderProvider);
148 
149       assertEquals(template.imageId("region/ami").build().getImage(), knownImage);
150 
151       verify(knownImage);
152       verify(os);
153       verify(defaultOptions);
154       verify(optionsProvider);
155       verify(templateBuilderProvider);
