186    private TemplateBuilder newTemplateBuilder() {
187       final Supplier<Set<? extends Image>> images = Suppliers.<Set<? extends Image>> ofInstance(ImmutableSet.<Image> of(
188                new ImageBuilder().providerId("cc-image").name("image").id("us-east-1/cc-image").location(location)
189                         .operatingSystem(new OperatingSystem(OsFamily.UBUNTU, null, "1.0", "hvm", "ubuntu", true))
190                         .description("description").version("1.0").defaultCredentials(new LoginCredentials("root", false))
191                         .build(), 
192                new ImageBuilder().providerId("normal-image").name("image").id("us-east-1/normal-image").location(location)
193                         .operatingSystem(new OperatingSystem(OsFamily.UBUNTU, null, "1.0", "paravirtual", "ubuntu", true))
194                         .description("description").version("1.0").defaultCredentials(new LoginCredentials("root", false))
195                         .build()));
196       
197       // weird compilation error means have to cast this - see https://bugs.eclipse.org/bugs/show_bug.cgi?id=365818
198       @SuppressWarnings("unchecked")
199       ImmutableMap<RegionAndName, Image> imageMap = (ImmutableMap<RegionAndName, Image>) ImagesToRegionAndIdMap.imagesToMap(images.get());
200       Supplier<LoadingCache<RegionAndName, ? extends Image>> imageCache = Suppliers.<LoadingCache<RegionAndName, ? extends Image>> ofInstance(
201                CacheBuilder.newBuilder().<RegionAndName,Image>build(CacheLoader.from(Functions.forMap(imageMap))));
202 
203       return newTemplateBuilder(images, imageCache);
204    }

206    @SuppressWarnings("unchecked")
207    private TemplateBuilder newTemplateBuilder(Supplier<Set<? extends Image>> images, Supplier<LoadingCache<RegionAndName, ? extends Image>> imageCache) {
208       Provider<TemplateOptions> optionsProvider = createMock(Provider.class);
209       Provider<TemplateBuilder> templateBuilderProvider = createMock(Provider.class);
210       TemplateOptions defaultOptions = createMock(TemplateOptions.class);
211 
212       expect(optionsProvider.get()).andReturn(defaultOptions);
213 
214       replay(optionsProvider);
215       replay(templateBuilderProvider);
216       Supplier<Set<? extends Location>> locations = Suppliers.<Set<? extends Location>> ofInstance(ImmutableSet
217                .<Location> of(location));
218       Supplier<Set<? extends Hardware>> sizes = Suppliers.<Set<? extends Hardware>> ofInstance(ImmutableSet
219                .<Hardware> of(t1_micro().build(), c1_medium().build(), c1_xlarge().build(), m1_large().build(),
220                         m1_small().build(), m1_xlarge().build(), m2_xlarge().build(), m2_2xlarge().build(),
221                         m2_4xlarge().build(), CC1_4XLARGE));
222 
223       return new EC2TemplateBuilderImpl(locations, images, sizes, Suppliers.ofInstance(location), optionsProvider,
224                templateBuilderProvider, imageCache) {
225       };
226    }
