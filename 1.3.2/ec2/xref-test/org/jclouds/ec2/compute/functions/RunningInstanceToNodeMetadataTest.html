191    @Test
192    public void testHandleMissingAMIs() {
193 
194       // Handle the case when the installed AMI no longer can be found in AWS.
195 
196       // Create a null-returning function to simulate that the AMI can't be found.
197       CacheLoader<RegionAndName, Image> nullReturningFunction = new CacheLoader<RegionAndName, Image>() {
198 
199          @Override
200          public Image load(@Nullable RegionAndName from) {
201             return null;
202          }
203       };
204       LoadingCache<RegionAndName, Image> instanceToImage = CacheBuilder.newBuilder().build(nullReturningFunction);
205 
206       RunningInstanceToNodeMetadata parser = createNodeParser(ImmutableSet.of(m1_small().build()), ImmutableSet
207                .of(provider), ImmutableMap.<String, Credentials> of(),
208                EC2ComputeServiceDependenciesModule.instanceToNodeState, instanceToImage);
209 
210       RunningInstance server = firstInstanceFromResource("/describe_instances_running.xml");
211 
212       assertEquals(
213             parser.apply(server),
214             new NodeMetadataBuilder().hostname("ip-10-243-42-70").state(NodeState.RUNNING)
215                   .privateAddresses(ImmutableSet.of("10.243.42.70")).publicAddresses(ImmutableSet.of("174.129.81.68"))
216                   .imageId("us-east-1/ami-82e4b5c7").id("us-east-1/i-0799056f").providerId("i-0799056f")
217                   .hardware(m1_small().build()).location(provider).build());
218    }

226    protected RunningInstanceToNodeMetadata createNodeParser(final ImmutableSet<Hardware> hardware,
227             final ImmutableSet<Location> locations, final Set<org.jclouds.compute.domain.Image> images,
228             Map<String, Credentials> credentialStore) {
229       Map<InstanceState, NodeState> instanceToNodeState = EC2ComputeServiceDependenciesModule.instanceToNodeState;
230       
231       CacheLoader<RegionAndName, Image> getRealImage = new CacheLoader<RegionAndName, Image>() {
232 
233          @Override
234          public Image load(@Nullable RegionAndName from) {
235             return ImagesToRegionAndIdMap.imagesToMap(images).get(from);
236          }
237       };
238       LoadingCache<RegionAndName, Image> instanceToImage = CacheBuilder.newBuilder().build(getRealImage);
239       return createNodeParser(hardware, locations, credentialStore, instanceToNodeState, instanceToImage);
240    }
241 
242    private RunningInstanceToNodeMetadata createNodeParser(final ImmutableSet<Hardware> hardware,
243             final ImmutableSet<Location> locations, Map<String, Credentials> credentialStore,
244             Map<InstanceState, NodeState> instanceToNodeState, LoadingCache<RegionAndName, Image> instanceToImage) {
245       Supplier<Set<? extends Location>> locationSupplier = new Supplier<Set<? extends Location>>() {
246 
247          @Override
248          public Set<? extends Location> get() {
249             return locations;
250          }
251 
252       };
253       Supplier<Set<? extends Hardware>> hardwareSupplier = new Supplier<Set<? extends Hardware>>() {
254 
255          @Override
256          public Set<? extends Hardware> get() {
257             return hardware;
258          }
259 
260       };
261       RunningInstanceToNodeMetadata parser = new RunningInstanceToNodeMetadata(instanceToNodeState, credentialStore,
262             Suppliers.<LoadingCache<RegionAndName, ? extends Image>> ofInstance(instanceToImage), locationSupplier,
263             hardwareSupplier);
264       return parser;
265    }
