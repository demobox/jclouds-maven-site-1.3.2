21  /**
22   * Configuration properties and constants used in BlobStore connections.
23   * 
24   * @author Adrian Cole
25   */
26  public interface BlobStoreConstants {
27     /**
28      * <p/>
29      * To interoperate with other S3 tools, we also accept the following:
30      * <ul>
31      * <li>an object named '#{dirpath}_$folder$' or '#{dirpath}/' denoting a directory marker</li>
32      * <li>an object with content type set to 'application/directory' denoting a directory marker</li>
33      * <li>if there exists any objects with the prefix "#{dirpath}/", then the directory is said to
34      * exist</li>
35      * <li>if both a file with the name of a directory and a marker for that directory exists, then
36      * the *file masks the directory*, and the directory is never returned.</li>
37      * </ul>
38      */
39     public static final String DIRECTORY_SUFFIX_ROOT = "/";
40     public static final String DIRECTORY_SUFFIX_FOLDER = "_$folder$";
41     public static final String[] DIRECTORY_SUFFIXES = { DIRECTORY_SUFFIX_FOLDER,
42              DIRECTORY_SUFFIX_ROOT };
43  
44     /**
45      * Key-value implementations of BlobStore, such as S3, do not have directories. We use an empty
46      * object '#{dirpath}_$folder$' with content type set to 'application/directory'.
47      */
48     public static final String PROPERTY_BLOBSTORE_DIRECTORY_SUFFIX = "jclouds.blobstore.directorysuffix";
49  
50     /**
51      * Any header starting with this prefix is considered user metadata. It will be stored with the
52      * object and returned when you retrieve the object/
53      */
54     public static final String PROPERTY_USER_METADATA_PREFIX = "jclouds.blobstore.metaprefix";
55  
56     public static final String BLOBSTORE_LOGGER = "jclouds.blobstore";
57  
58  }
