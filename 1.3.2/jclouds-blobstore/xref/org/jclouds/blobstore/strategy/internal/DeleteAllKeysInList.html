49  /**
50   * Deletes all keys in the container
51   * 
52   * @author Adrian Cole
53   */
54  @Singleton
55  public class DeleteAllKeysInList implements ClearListStrategy, ClearContainerStrategy {
56     @Resource
57     @Named(BlobStoreConstants.BLOBSTORE_LOGGER)
58     protected Logger logger = Logger.NULL;
59  
60     protected final ListContainerStrategy listContainer;
61     protected final BackoffLimitedRetryHandler retryHandler;
62     private final ExecutorService userExecutor;
63  
64     protected final AsyncBlobStore connection;
65     /**
66      * maximum duration of an blob Request
67      */
68     @Inject(optional = true)
69     @Named(Constants.PROPERTY_REQUEST_TIMEOUT)
70     protected Long maxTime;
71  
72     @Inject
73     DeleteAllKeysInList(@Named(Constants.PROPERTY_USER_THREADS) ExecutorService userExecutor,
74              AsyncBlobStore connection, ListContainerStrategy listContainer,
75              BackoffLimitedRetryHandler retryHandler) {
76  
77        this.userExecutor = userExecutor;
78        this.connection = connection;
79        this.listContainer = listContainer;
80        this.retryHandler = retryHandler;
81     }
82  
83     public void execute(String containerName) {
84        execute(containerName, recursive());
85     }
86  
87     public void execute(final String containerName, final ListContainerOptions options) {
88        String message = options.getDir() != null ? String.format("clearing path %s/%s",
89                 containerName, options.getDir()) : String.format("clearing container %s",
90                 containerName);
91        if (options.isRecursive())
92           message = message + " recursively";
93        Map<StorageMetadata, Exception> exceptions = Maps.newHashMap();
94        Iterable<? extends StorageMetadata> toDelete;
95        for (int i = 0; i < 3; i++) { // TODO parameterize
96           toDelete = getResourcesToDelete(containerName, options);
97           if (Iterables.isEmpty(toDelete)) {
98              break;
99           }
100 
101          Map<StorageMetadata, Future<?>> responses = Maps.newHashMap();
102          try {
103             for (final StorageMetadata md : toDelete) {
104                String fullPath = parentIsFolder(options, md) ? options.getDir() + "/"
105                         + md.getName() : md.getName();
106                switch (md.getType()) {
107                   case BLOB:
108                      responses.put(md, connection.removeBlob(containerName, fullPath));
109                      break;
110                   case FOLDER:
111                      if (options.isRecursive() && !fullPath.equals(options.getDir())) {
112                         execute(containerName, options.clone().inDirectory(fullPath));
113                      }
114                      responses.put(md, connection.deleteDirectory(containerName, fullPath));
115                      break;
116                   case RELATIVE_PATH:
117                      if (options.isRecursive() && !fullPath.equals(options.getDir())) {
118                         execute(containerName, options.clone().inDirectory(fullPath));
119                      }
120                      responses.put(md, connection.deleteDirectory(containerName, md.getName()));
121                      break;
122                   case CONTAINER:
123                      throw new IllegalArgumentException("Container type not supported");
124                }
125             }
126          } finally {
127             exceptions = awaitCompletion(responses, userExecutor, maxTime, logger, message);
128          }
129          if (!exceptions.isEmpty()) {
130             retryHandler.imposeBackoffExponentialDelay(i + 1, message);
131          }
132       }
133       if (!exceptions.isEmpty())
134          throw new BlobRuntimeException(String.format("error %s: %s", message, exceptions));
135       toDelete = getResourcesToDelete(containerName, options);
136       assert Iterables.isEmpty(toDelete) : String.format("items remaining %s: %s", message,
137                toDelete);
138    }
139 
140    private boolean parentIsFolder(final ListContainerOptions options, final StorageMetadata md) {
141       return (options.getDir() != null && md.getName().indexOf('/') == -1);
142    }
143 
144    private Iterable<? extends StorageMetadata> getResourcesToDelete(final String containerName,
145             final ListContainerOptions options) {
146       Iterable<? extends StorageMetadata> toDelete = Iterables.filter(listContainer.execute(
147                containerName, options), new Predicate<StorageMetadata>() {
148 
149          @Override
150          public boolean apply(StorageMetadata input) {
151             switch (input.getType()) {
152                case BLOB:
153                   return true;
154                case FOLDER:
155                case RELATIVE_PATH:
156                   if (options.isRecursive())
157                      return true;
158                   break;
159             }
160             return false;
161          }
162 
163       });
164       return toDelete;
165    }
