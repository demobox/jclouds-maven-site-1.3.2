48     @Test(groups = { "integration", "live" })
49     public void testAllLocations() throws InterruptedException {
50        for (final Location location : context.getBlobStore().listAssignableLocations()) {
51           final String containerName = getScratchContainerName();
52           try {
53              System.err.printf(" >> creating container in location %s%n", location);
54              context.getBlobStore().createContainerInLocation(location, containerName);
55              System.err.printf(" << call complete.. checking%n");
56  
57              assertConsistencyAware(new Runnable() {
58  
59                 @Override
60                 public void run() {
61                    PageSet<? extends StorageMetadata> list = context.getBlobStore().list();
62                    assert Iterables.any(list, new Predicate<StorageMetadata>() {
63                       public boolean apply(StorageMetadata md) {
64                          return containerName.equals(md.getName()) && location.equals(md.getLocation());
65                       }
66                    }) : String.format("container %s/%s not found in list %s", location, containerName, list);
67                 }
68  
69              });
70           } finally {
71              recycleContainer(containerName);
72           }
73        }
74     }

80     @Test(groups = { "integration", "live" })
81     public void testGetAssignableLocations() throws Exception {
82        assertProvider(context.getProviderSpecificContext());
83        for (Location location : context.getBlobStore().listAssignableLocations()) {
84           System.err.printf("location %s%n", location);
85           assert location.getId() != null : location;
86           assert location != location.getParent() : location;
87           assert location.getScope() != null : location;
88           switch (location.getScope()) {
89              case PROVIDER:
90                 assertProvider(location);
91                 break;
92              case REGION:
93                 assertProvider(location.getParent());
94                 assert location.getIso3166Codes().size() == 0
95                          || location.getParent().getIso3166Codes().containsAll(location.getIso3166Codes()) : location
96                          + " ||" + location.getParent();
97                 break;
98              case ZONE:
99                 Location provider = location.getParent().getParent();
100                // zone can be a direct descendant of provider
101                if (provider == null)
102                   provider = location.getParent();
103                assertProvider(provider);
104                assert location.getIso3166Codes().size() == 0
105                         || location.getParent().getIso3166Codes().containsAll(location.getIso3166Codes()) : location
106                         + " ||" + location.getParent();
107                break;
108             case HOST:
109                Location provider2 = location.getParent().getParent().getParent();
110                // zone can be a direct descendant of provider
111                if (provider2 == null)
112                   provider2 = location.getParent().getParent();
113                assertProvider(provider2);
114                break;
115          }
116       }
117    }

119    void assertProvider(Location provider) {
120       assertEquals(provider.getScope(), LocationScope.PROVIDER);
121       assertEquals(provider.getParent(), null);
122       assertEquals(provider.getIso3166Codes(), getIso3166Codes());
123    }
