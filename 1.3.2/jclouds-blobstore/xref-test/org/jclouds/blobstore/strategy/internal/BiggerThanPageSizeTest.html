23  import java.io.IOException;

25  import org.jclouds.blobstore.BlobStore;
26  import org.jclouds.blobstore.BlobStoreContextFactory;
27  import org.jclouds.blobstore.options.ListContainerOptions;
28  import org.testng.annotations.BeforeTest;
29  import org.testng.annotations.Test;

31  /**
32   * 
33   * @author Adrian Cole

35  @Test
36  public class BiggerThanPageSizeTest {
37     private BlobStore blobstore;
38  
39     @BeforeTest
40     void setupBlobStore() {
41        blobstore = new BlobStoreContextFactory().createContext("transient", "foo", "bar").getBlobStore();
42     }
43  
44     public void test() throws IOException {
45        blobstore.createContainerInLocation(null, "goodies");
46        for (int i = 0; i < 1001; i++) {
47           blobstore.putBlob("goodies", blobstore.blobBuilder(i + "").payload(i + "").build());
48        }
49        assertEquals(blobstore.countBlobs("goodies"), 1001);
50        blobstore.clearContainer("goodies");
51        assertEquals(blobstore.countBlobs("goodies"), 0);
52     }
53  
54     public void testStrategies() throws IOException {
55        blobstore.createContainerInLocation(null, "poo");
56        for (int i = 0; i < 1001; i++) {
57           blobstore.putBlob("poo", blobstore.blobBuilder(i + "").payload(i + "").build());
58        }
59  
60        ListContainerAndRecurseThroughFolders lister = new ListContainerAndRecurseThroughFolders(
61              new ConcatenateContainerLists(blobstore));
62        assertEquals(lister.execute("poo", ListContainerOptions.NONE).size(), 1001);
63        blobstore.clearContainer("poo");
64        assertEquals(lister.execute("poo", ListContainerOptions.NONE).size(), 0);
65     }
