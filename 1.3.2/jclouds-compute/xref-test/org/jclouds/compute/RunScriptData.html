57     public static Statement installJavaAndCurl(OperatingSystem os) {
58        if (os == null || OperatingSystemPredicates.supportsApt().apply(os))
59           return APT_RUN_SCRIPT;
60        else if (OperatingSystemPredicates.supportsYum().apply(os))
61           return YUM_RUN_SCRIPT;
62        else if (OperatingSystemPredicates.supportsZypper().apply(os))
63           return ZYPPER_RUN_SCRIPT;
64        else
65           throw new IllegalArgumentException("don't know how to handle" + os.toString());
66     }

68     public static Statement authorizePortsInIpTables(int... ports) {
69        Builder<Statement> builder = ImmutableList.<Statement> builder();
70        for (int port : ports)
71           builder.add(exec("iptables -I INPUT 1 -p tcp --dport " + port + " -j ACCEPT"));
72        builder.add(exec("iptables-save"));
73        return new StatementList(builder.build());
74     }

76     public static StatementList installAdminUserJBossAndOpenPorts(OperatingSystem os) throws IOException {
77        return new StatementList(//

88     // NOTE do not name this the same as your login user, or the init process may kill you!
89     public static InitBuilder startJBoss(String configuration) {
90        return new InitBuilder(
91                 "jboss",
92                 JBOSS_HOME,
93                 JBOSS_HOME,
94                 ImmutableMap.of("jbossHome", JBOSS_HOME),
95                 ImmutableList.<Statement>of(appendFile(JBOSS_HOME + "/standalone/configuration/standalone-custom.xml", Splitter.on('\n').split(configuration))),
96                 ImmutableList
97                          .<Statement> of(interpret(new StringBuilder().append("java ").append(' ')
98                                   .append("-server -Xms128m -Xmx128m -XX:MaxPermSize=128m -Djava.net.preferIPv4Stack=true -XX:+UseFastAccessorMethods -XX:+TieredCompilation -Xverify:none -Dorg.jboss.resolver.warning=true -Dsun.rmi.dgc.client.gcInterval=3600000 -Dsun.rmi.dgc.server.gcInterval=3600000").append(' ')
99                                   .append("-Djboss.modules.system.pkgs=org.jboss.byteman").append(' ')
100                                  .append("-Dorg.jboss.boot.log.file=$JBOSS_HOME/standalone/log/boot.log").append(' ')
101                                  .append("-Dlogging.configuration=file:$JBOSS_HOME/standalone/configuration/logging.properties").append(' ')
102                                  .append("-jar $JBOSS_HOME/jboss-modules.jar").append(' ')
103                                  .append("-mp $JBOSS_HOME/modules").append(' ')
104                                  .append("-logmodule org.jboss.logmanager").append(' ')
105                                  .append("-jaxpmodule javax.xml.jaxp-provider").append(' ')
106                                  .append("org.jboss.as.standalone").append(' ')
107                                  .append("-Djboss.home.dir=$JBOSS_HOME").append(' ')
108                                  .append("--server-config=standalone-custom.xml")
109                                  .toString())));
110    }

118    public static Statement addHostnameToEtcHostsIfMissing() {
119       return exec("grep `hostname` /etc/hosts >/dev/null || awk -v hostname=`hostname` 'END { print $1\" \"hostname }' /proc/net/arp >> /etc/hosts");
120    }
121 
122    public static Statement addDnsToResolverIfMissing() {
123       return exec("nslookup yahoo.com >/dev/null || echo nameserver 208.67.222.222 >> /etc/resolv.conf");
124    }

152    public static String aptInstallLazyUpgrade(String packageName) {
153       return aptInstall + " " + packageName + "|| (" + "apt-get update -qq&&" + "apt-get upgrade -y -qq" + ")&&"
154                + aptInstall + " " + packageName;
