222       try {
223          client.destroyNodesMatching(inGroup(group));
224       } catch (Exception e) {
225 
226       }
227 
228       TemplateOptions options = client.templateOptions().blockOnPort(22, 120);
229       try {
230          Set<? extends NodeMetadata> nodes = client.createNodesInGroup(group, 1, options);
231          NodeMetadata node = get(nodes, 0);
232          LoginCredentials good = node.getCredentials();
233          assert good.identity != null : nodes;
234          assert good.credential != null : nodes;
235 
236          for (Entry<? extends NodeMetadata, ExecResponse> response : client.runScriptOnNodesMatching(
237                runningInGroup(group), Statements.exec("hostname"),
238                wrapInInitScript(false).runAsRoot(false).overrideLoginCredentials(good)).entrySet()) {
239             checkResponseEqualsHostname(response.getValue(), response.getKey());
240          }
241 
242          // test single-node execution
243          ExecResponse response = client.runScriptOnNode(node.getId(), "hostname",
244                wrapInInitScript(false).runAsRoot(false));
245          checkResponseEqualsHostname(response, node);
246          OperatingSystem os = node.getOperatingSystem();
247 
248          // test bad password
249          tryBadPassword(group, good);
250 
251          runScriptWithCreds(group, os, good);
252 
253          checkNodes(nodes, group, "runScriptWithCreds");
254 
255          // test adding AdminAccess later changes the default boot user, in this
256          // case to foo
257          ListenableFuture<ExecResponse> future = client.submitScriptOnNode(node.getId(), AdminAccess.builder()
258                .adminUsername("foo").build(), nameTask("adminUpdate"));
259 
260          response = future.get(3, TimeUnit.MINUTES);
261 
262          assert response.getExitCode() == 0 : node.getId() + ": " + response;
263 
264          node = client.getNodeMetadata(node.getId());
265          // test that the node updated to the correct admin user!
266          assertEquals(node.getCredentials().identity, "foo");
267          assert node.getCredentials().credential != null : nodes;
268 
269          weCanCancelTasks(node);
270 
271          assert response.getExitCode() == 0 : node.getId() + ": " + response;
272 
273          response = client.runScriptOnNode(node.getId(), "echo $USER", wrapInInitScript(false).runAsRoot(false));
274 
275          assert response.getOutput().trim().equals("foo") : node.getId() + ": " + response;
276 
277       } finally {
278          client.destroyNodesMatching(inGroup(group));
279       }
280    }

300    @Test(enabled = false)
301    public void weCanCancelTasks(NodeMetadata node) throws InterruptedException, ExecutionException {
302       ListenableFuture<ExecResponse> future = client.submitScriptOnNode(node.getId(), Statements.exec("sleep 300"),
303             nameTask("sleeper").runAsRoot(false));
304       ExecResponse response = null;
305       try {
306          response = future.get(1, TimeUnit.MILLISECONDS);
307          assert false : node.getId() + ": " + response;
308       } catch (TimeoutException e) {
309          assert !future.isDone();
310          response = client.runScriptOnNode(node.getId(), Statements.exec("./sleeper status"), wrapInInitScript(false)
311                .runAsRoot(false));
312          assert !response.getOutput().trim().equals("") : node.getId() + ": " + response;
313          future.cancel(true);
314          response = client.runScriptOnNode(node.getId(), Statements.exec("./sleeper status"), wrapInInitScript(false)
315                .runAsRoot(false));
316          assert response.getOutput().trim().equals("") : node.getId() + ": " + response;
317          try {
318             future.get();
319             assert false : future;
320          } catch (CancellationException e1) {
321 
322          }
323       }
324    }

400    @Test(enabled = true, dependsOnMethods = "testCreateTwoNodesWithRunScript")
401    public void testCreateAnotherNodeWithANewContextToEnsureSharedMemIsntRequired() throws Exception {
402       initializeContextAndClient();
403 
404       Location existingLocation = Iterables.get(this.nodes, 0).getLocation();
405       boolean existingLocationIsAssignable = Iterables.any(client.listAssignableLocations(),
406             Predicates.equalTo(existingLocation));
407 
408       if (existingLocationIsAssignable) {
409          getAnonymousLogger().info("creating another node based on existing nodes' location: " + existingLocation);
410          template = addRunScriptToTemplate(client.templateBuilder().fromTemplate(template)
411                .locationId(existingLocation.getId()).build());
412       } else {
413          refreshTemplate();
414          getAnonymousLogger().info(
415                format("%s is not assignable; using template's location %s as  ", existingLocation,
416                      template.getLocation()));
417       }
418 
419       Set<? extends NodeMetadata> nodes = client.createNodesInGroup(group, 1, template);
420       assertEquals(nodes.size(), 1);
421       checkNodes(nodes, group, "bootstrap");
422       NodeMetadata node = Iterables.getOnlyElement(nodes);
423       if (existingLocationIsAssignable)
424          assertEquals(node.getLocation(), existingLocation);
425       else
426          this.assertLocationSameOrChild(node.getLocation(), template.getLocation());
427       checkOsMatchesTemplate(node);
428       this.nodes.add(node);
429    }
