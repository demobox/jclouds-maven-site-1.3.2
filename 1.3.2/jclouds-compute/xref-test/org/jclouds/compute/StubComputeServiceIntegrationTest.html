260          private void runScriptAndService(SshClient client, SshClient clientNew) {
261             client.connect();
262 
263             try {
264                String scriptName = "configure-jboss";
265                client.put("/tmp/init-" + scriptName, Strings2.toStringAndClose(StubComputeServiceIntegrationTest.class

268                expect(client.exec("ln -fs /tmp/init-" + scriptName + " " + scriptName)).andReturn(EXEC_GOOD);
269                expect(client.getUsername()).andReturn("root").atLeastOnce();
270                expect(client.getHostAddress()).andReturn("localhost").atLeastOnce();
271                expect(client.exec("./" + scriptName + " init")).andReturn(EXEC_GOOD);
272                expect(client.exec("./" + scriptName + " start")).andReturn(EXEC_GOOD);
273                expect(client.exec("./" + scriptName + " status")).andReturn(EXEC_GOOD);

397    private static class PayloadEquals implements IArgumentMatcher, Serializable {
398 
399       private static final long serialVersionUID = 583055160049982067L;
400 
401       private final Object expected;
402 
403       public PayloadEquals(Object expected) {
404          this.expected = expected;
405       }
406 
407       public boolean matches(Object actual) {
408          if (this.expected == null) {
409             return actual == null;
410          }
411          try {
412             String real = Strings2.toStringAndClose(((Payload) actual).getInput());
413             assertEquals(real, expected);
414             return true;
415          } catch (IOException e) {
416             Throwables.propagate(e);
417             return false;
418          }
419       }
420 
421       public void appendTo(StringBuffer buffer) {
422          appendQuoting(buffer);
423          buffer.append(expected);
424          appendQuoting(buffer);
425       }
426 
427       private void appendQuoting(StringBuffer buffer) {
428          if (expected instanceof String) {
429             buffer.append("\"");
430          } else if (expected instanceof Character) {
431             buffer.append("'");
432          }
433       }
434 
435       @Override
436       public boolean equals(Object o) {
437          if (o == null || !this.getClass().equals(o.getClass()))
438             return false;
439          PayloadEquals other = (PayloadEquals) o;
440          return this.expected == null && other.expected == null || this.expected != null
441                   && this.expected.equals(other.expected);
442       }
443 
444       @Override
445       public int hashCode() {
446          throw new UnsupportedOperationException("hashCode() is not supported");
447       }
