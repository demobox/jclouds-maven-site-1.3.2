184    @Override
185    public Set<? extends NodeMetadata> createNodesInGroup(String group, int count, Template template)
186          throws RunNodesException {
187       checkNotNull(group, "group cannot be null");
188       checkNotNull(template.getLocation(), "location");
189       logger.debug(">> running %d node%s group(%s) location(%s) image(%s) hardwareProfile(%s) options(%s)", count,
190             count > 1 ? "s" : "", group, template.getLocation().getId(), template.getImage().getId(), template
191                   .getHardware().getId(), template.getOptions());
192       Set<NodeMetadata> goodNodes = newLinkedHashSet();
193       Map<NodeMetadata, Exception> badNodes = newLinkedHashMap();
194       Multimap<NodeMetadata, CustomizationResponse> customizationResponses = LinkedHashMultimap.create();
195 
196       if (template.getOptions().getRunScript() != null)
197          initAdminAccess.visit(template.getOptions().getRunScript());
198 
199       Map<?, Future<Void>> responses = runNodesAndAddToSetStrategy.execute(group, count, template, goodNodes, badNodes,
200             customizationResponses);
201       Map<?, Exception> executionExceptions = awaitCompletion(responses, executor, null, logger, "createNodesInGroup("
202             + group + ")");
203       Function<NodeMetadata, NodeMetadata> fn = persistNodeCredentials.always(template.getOptions().getRunScript());
204       badNodes = Maps2.transformKeys(badNodes, fn);
205       goodNodes = ImmutableSet.copyOf(Iterables.transform(goodNodes, fn));
206       if (executionExceptions.size() > 0 || badNodes.size() > 0) {
207          throw new RunNodesException(group, count, template, goodNodes, executionExceptions, badNodes);
208       }
209       return goodNodes;
210    }

226    @Override
227    public void destroyNode(final String id) {
228       checkNotNull(id, "id");
229       logger.debug(">> destroying node(%s)", id);
230       final AtomicReference<NodeMetadata> node = new AtomicReference<NodeMetadata>();
231       RetryablePredicate<String> tester = new RetryablePredicate<String>(new Predicate<String>() {
232 
233          @Override
234          public boolean apply(String input) {
235             try {
236                NodeMetadata md = destroyNodeStrategy.destroyNode(id);
237                if (md != null)
238                   node.set(md);
239                return true;
240             } catch (IllegalStateException e) {
241                logger.warn("<< illegal state destroying node(%s)", id);
242                return false;
243             }
244          }
245 
246       }, timeouts.nodeRunning, 1000, TimeUnit.MILLISECONDS);
247       boolean successful = tester.apply(id) && (node.get() == null || nodeTerminated.apply(node.get()));
248       if (successful)
249          credentialStore.remove("node#" + id);
250       logger.debug("<< destroyed node(%s) success(%s)", id, successful);
251    }

482    /**
483     * {@inheritDoc}
484     */
485    @Override
486    public Map<NodeMetadata, ExecResponse> runScriptOnNodesMatching(Predicate<NodeMetadata> filter, Statement runScript,
487          RunScriptOptions options) throws RunScriptOnNodesException {
488 
489       checkNotNull(filter, "filter");
490       checkNotNull(runScript, "runScript");
491       checkNotNull(options, "options");
492 
493       Map<NodeMetadata, ExecResponse> goodNodes = newLinkedHashMap();
494       Map<NodeMetadata, Exception> badNodes = newLinkedHashMap();
495       Map<NodeMetadata, Future<ExecResponse>> responses = newLinkedHashMap();
496       Map<?, Exception> exceptions = ImmutableMap.<Object, Exception> of();
497 
498       initAdminAccess.visit(runScript);
499 
500       Iterable<? extends RunScriptOnNode> scriptRunners = transformNodesIntoInitializedScriptRunners(
501             nodesMatchingFilterAndNotTerminatedExceptionIfNotFound(filter), runScript, options, badNodes);
502       if (Iterables.size(scriptRunners) > 0) {
503          for (RunScriptOnNode runner : scriptRunners) {
504             responses.put(runner.getNode(), executor.submit(new RunScriptOnNodeAndAddToGoodMapOrPutExceptionIntoBadMap(
505                   runner, goodNodes, badNodes)));
506          }
507          exceptions = awaitCompletion(responses, executor, null, logger, "runScriptOnNodesMatching(" + filter + ")");
508       }
509 
510       Function<NodeMetadata, NodeMetadata> fn = persistNodeCredentials.ifAdminAccess(runScript);
511       badNodes = Maps2.transformKeys(badNodes, fn);
512       goodNodes = Maps2.transformKeys(goodNodes, fn);
513 
514       if (exceptions.size() > 0 || badNodes.size() > 0) {
515          throw new RunScriptOnNodesException(runScript, options, goodNodes, exceptions, badNodes);
516       }
517       return goodNodes;
518    }

544    /**
545     * {@inheritDoc}
546     */
547    @Override
548    public ExecResponse runScriptOnNode(String id, Statement runScript, RunScriptOptions options) {
549       NodeMetadata node = this.getNodeMetadata(id);
550       if (node == null)
551          throw new NoSuchElementException(id);
552       if (node.getState() != NodeState.RUNNING)
553          throw new IllegalStateException("node " + id
554                + " needs to be running before executing a script on it. current state: " + node.getState());
555       initAdminAccess.visit(runScript);
556       node = updateNodeWithCredentialsIfPresent(node, options);
557       ExecResponse response = runScriptOnNodeFactory.create(node, runScript, options).init().call();
558       persistNodeCredentials.ifAdminAccess(runScript).apply(node);
559       return response;
560    }

565    @Override
566    public ListenableFuture<ExecResponse> submitScriptOnNode(String id, final Statement runScript,
567          RunScriptOptions options) {
568       NodeMetadata node = this.getNodeMetadata(id);
569       if (node == null)
570          throw new NoSuchElementException(id);
571       if (node.getState() != NodeState.RUNNING)
572          throw new IllegalStateException("node " + id
573                + " needs to be running before executing a script on it. current state: " + node.getState());
574       initAdminAccess.visit(runScript);
575       final NodeMetadata node1 = updateNodeWithCredentialsIfPresent(node, options);
576       ListenableFuture<ExecResponse> response = runScriptOnNodeFactory.submit(node1, runScript, options);
577       response.addListener(new Runnable() {
578 
579          @Override
580          public void run() {
581             persistNodeCredentials.ifAdminAccess(runScript).apply(node1);
582          }
583 
584       }, executor);
585       return response;
586    }

605    protected NodeMetadata updateNodeWithCredentialsIfPresent(NodeMetadata node, RunScriptOptions options) {
606       checkNotNull(node, "node");
607       Builder builder = LoginCredentials.builder(node.getCredentials());
608       if (options.getLoginUser() != null)
609          builder.user(options.getLoginUser());
610       if (options.hasLoginPasswordOption()) {
611           if (options.hasLoginPassword()) {
612              builder.password(options.getLoginPassword());
613           } else {
614              builder.noPassword();
615           }
616       }
617       if (options.hasLoginPrivateKeyOption()) {
618           if (options.hasLoginPrivateKey()) {
619              builder.privateKey(options.getLoginPrivateKey());
620           } else {
621              builder.noPrivateKey();
622           }
623       }
