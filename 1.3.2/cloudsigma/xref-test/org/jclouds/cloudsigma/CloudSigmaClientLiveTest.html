269    public void testCreateAndDestroyStaticIP() throws Exception {

285          client.stopServer(server.getUuid());

287          assertEquals(server.getNics().get(0).getDhcp(), ip2.getAddress());
288       } finally {
289          if (server != null) {
290             client.destroyServer(server.getUuid());
291             client.destroyDrive(drive.getUuid());
292          }
293          client.destroyStaticIP(ip.getAddress());
294          client.destroyStaticIP(ip2.getAddress());
295       }

396    protected void doConnectViaSsh(Server server, LoginCredentials creds) throws IOException {
397       SshClient ssh = Guice.createInjector(new SshjSshClientModule()).getInstance(SshClient.Factory.class)
398             .create(new IPSocket(server.getVnc().getIp(), 22), creds);
399       try {
400          ssh.connect();
401          ExecResponse hello = ssh.exec("echo hello");
402          assertEquals(hello.getOutput().trim(), "hello");
403          System.err.println(ssh.exec("df -k").getOutput());
404          System.err.println(ssh.exec("mount").getOutput());
405          System.err.println(ssh.exec("uname -a").getOutput());
406       } finally {
407          if (ssh != null)
408             ssh.disconnect();
409       }
410    }

444    protected void prepareDrive() {
445       client.destroyDrive(drive.getUuid());
446       drive = client.cloneDrive(imageId, drive.getName(),
447             new CloneDriveOptions()
448             .size(driveSize)
449             .tags("cat:mouse", "monkey:banana")
450       );

452       assert driveNotClaimed.apply(drive) : client.getDriveInfo(drive.getUuid());
453       
454       DriveInfo clonedDrive = client.getDriveInfo(drive.getUuid());
455       System.err.println("after prepare" + clonedDrive);
456       assertEquals(clonedDrive.getTags(), ImmutableSet.of("cat:mouse", "monkey:banana"));
