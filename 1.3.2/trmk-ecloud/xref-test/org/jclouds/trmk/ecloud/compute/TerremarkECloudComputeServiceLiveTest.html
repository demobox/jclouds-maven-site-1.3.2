50  @Test(groups = "live", enabled = true, singleThreaded = true)
51  public class TerremarkECloudComputeServiceLiveTest extends BaseComputeServiceLiveTest {
52  
53     @Override
54     protected Properties setupProperties() {
55        Properties props = super.setupProperties();
56        props.setProperty(VCloudConstants.PROPERTY_VCLOUD_DEFAULT_VDC,
57              ".* - " + System.getProperty("test.trmk-ecloud.datacenter", "MIA"));
58        return props;
59     }
60  
61     public TerremarkECloudComputeServiceLiveTest() {
62        provider = "trmk-ecloud";
63     }
64  
65     @Override
66     public void setServiceDefaults() {
67        group = "te";
68     }
69  
70     // terremark does not support metadata
71     @Override
72     protected void checkUserMetadataInNodeEquals(NodeMetadata node, ImmutableMap<String, String> userMetadata) {
73        assert node.getUserMetadata().equals(ImmutableMap.<String, String> of()) : String.format(
74              "node userMetadata did not match %s %s", userMetadata, node);
75     }
76  
77     @Override
78     protected Template buildTemplate(TemplateBuilder templateBuilder) {
79        Template template = super.buildTemplate(templateBuilder);
80        Image image = template.getImage();
81        assert image.getDefaultCredentials() != null && image.getDefaultCredentials().identity != null : image;
82        assert image.getDefaultCredentials() != null && image.getDefaultCredentials().credential != null : image;
83        return template;
84     }
85  
86     @Override
87     public void testListImages() throws Exception {
88        for (Image image : client.listImages()) {
89           assert image.getProviderId() != null : image;
90           // image.getLocationId() can be null, if it is a location-free image
91           assertEquals(image.getType(), ComputeType.IMAGE);
92           if (image.getOperatingSystem().getFamily() != OsFamily.WINDOWS
93                 && image.getOperatingSystem().getFamily() != OsFamily.SOLARIS) {
94              assert image.getDefaultCredentials() != null && image.getDefaultCredentials().identity != null : image;
95              assert image.getDefaultCredentials().credential != null : image;
96           }
97        }
98     }
99  
100    @Override
101    public void testListNodes() throws Exception {
102       for (ComputeMetadata node : client.listNodes()) {
103          assert node.getProviderId() != null;
104          assert node.getLocation() != null;
105          assertEquals(node.getType(), ComputeType.NODE);
106          NodeMetadata allData = client.getNodeMetadata(node.getId());
107          RestContext<TerremarkVCloudClient, TerremarkVCloudClient> tmContext = new ComputeServiceContextFactory()
108                .createContext(provider, identity, credential).getProviderSpecificContext();
109          VApp vApp = tmContext.getApi().findVAppInOrgVDCNamed(allData.getLocation().getParent().getDescription(),
110                   allData.getLocation().getDescription(), allData.getName());
111          assertEquals(vApp.getName(), allData.getName());
112       }
113    }

118    }

123    }
