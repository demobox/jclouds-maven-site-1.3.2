49  @Test(groups = "unit")
50  public class ParseServerListFromJsonResponseTest {
51  
52     Injector i = Guice.createInjector(new GsonModule());
53  
54     public void testApplyInputStream() {
55        InputStream is = getClass().getResourceAsStream("/test_list_servers.json");
56  
57        List<Server> expects = ImmutableList.of(new Server(1234, "sample-server"), new Server(5678, "sample-server2"));
58  
59        UnwrapOnlyJsonValue<List<Server>> parser = i.getInstance(Key
60              .get(new TypeLiteral<UnwrapOnlyJsonValue<List<Server>>>() {
61              }));
62        List<Server> response = parser.apply(new HttpResponse(200, "ok", Payloads.newInputStreamPayload(is)));
63  
64        assertEquals(response, expects);
65     }
66  
67     public void testApplyInputStreamDetails() throws UnknownHostException {
68        InputStream is = getClass().getResourceAsStream("/test_list_servers_detail.json");
69  
70        UnwrapOnlyJsonValue<List<Server>> parser = i.getInstance(Key
71              .get(new TypeLiteral<UnwrapOnlyJsonValue<List<Server>>>() {
72              }));
73        List<Server> response = parser.apply(new HttpResponse(200, "ok", Payloads.newInputStreamPayload(is)));
74  
75        assertEquals(response.get(0).getId(), 1234);
76        assertEquals(response.get(0).getName(), "sample-server");
77        assertEquals(response.get(0).getImageId(), new Integer(2));
78        assertEquals(response.get(0).getFlavorId(), new Integer(1));
79        assertEquals(response.get(0).getHostId(), "e4d909c290d0fb1ca068ffaddf22cbd0");
80        assertEquals(response.get(0).getStatus(), ServerStatus.BUILD);
81        assertEquals(response.get(0).getProgress(), new Integer(60));
82        List<String> publicAddresses = Lists.newArrayList("67.23.10.132", "67.23.10.131");
83        List<String> privateAddresses = Lists.newArrayList("10.176.42.16");
84        Addresses addresses1 = new Addresses();
85        addresses1.getPrivateAddresses().addAll(privateAddresses);
86        addresses1.getPublicAddresses().addAll(publicAddresses);
87        assertEquals(response.get(0).getAddresses(), addresses1);
88        assertEquals(response.get(0).getMetadata(), ImmutableMap.of("Server Label", "Web Head 1", "Image Version", "2.1"));
89        assertEquals(response.get(1).getId(), 5678);
90        assertEquals(response.get(1).getName(), "sample-server2");
91        assertEquals(response.get(1).getImageId(), new Integer(2));
92        assertEquals(response.get(1).getFlavorId(), new Integer(1));
93        assertEquals(response.get(1).getHostId(), "9e107d9d372bb6826bd81d3542a419d6");
94        assertEquals(response.get(1).getStatus(), ServerStatus.ACTIVE);
95        assertEquals(response.get(1).getProgress(), null);
96        List<String> publicAddresses2 = Lists.newArrayList("67.23.10.133");
97        List<String> privateAddresses2 = Lists.newArrayList("10.176.42.17");
98        Addresses addresses2 = new Addresses();
99        addresses2.getPrivateAddresses().addAll(privateAddresses2);
100       addresses2.getPublicAddresses().addAll(publicAddresses2);
101       assertEquals(response.get(1).getAddresses(), addresses2);
102       assertEquals(response.get(1).getMetadata(), ImmutableMap.of("Server Label", "DB 1"));
103 
104    }
