54  /**
55   * Configures the {@link CloudServersComputeServiceContext}; requires {@link BaseComputeService}
56   * bound.
57   * 
58   * @author Adrian Cole
59   */
60  public class CloudServersComputeServiceContextModule
61           extends
62           ComputeServiceAdapterContextModule<CloudServersClient, CloudServersAsyncClient, Server, Flavor, org.jclouds.cloudservers.domain.Image, Location> {
63     public CloudServersComputeServiceContextModule() {
64        super(CloudServersClient.class, CloudServersAsyncClient.class);
65     }
66  
67     @SuppressWarnings("unchecked")
68     @Override
69     protected void configure() {
70        super.configure();
71        bind(new TypeLiteral<ComputeServiceAdapter<Server, Flavor, org.jclouds.cloudservers.domain.Image, Location>>() {
72        }).to(CloudServersComputeServiceAdapter.class);
73        
74        bind(new TypeLiteral<Function<Server, NodeMetadata>>() {
75        }).to(ServerToNodeMetadata.class);
76  
77        bind(new TypeLiteral<Function<org.jclouds.cloudservers.domain.Image, Image>>() {
78        }).to(CloudServersImageToImage.class);
79        bind(new TypeLiteral<Function<org.jclouds.cloudservers.domain.Image, OperatingSystem>>() {
80        }).to(CloudServersImageToOperatingSystem.class);
81  
82        bind(new TypeLiteral<Function<Flavor, Hardware>>() {
83        }).to(FlavorToHardware.class);
84        
85        // we aren't converting location from a provider-specific type
86        bind(new TypeLiteral<Function<Location, Location>>() {
87        }).to((Class) IdentityFunction.class);
88        
89        // there are no locations except  the provider
90        bind(new TypeLiteral<Supplier<Location>>() {
91        }).to(OnlyLocationOrFirstZone.class);
92  
93     }
94
