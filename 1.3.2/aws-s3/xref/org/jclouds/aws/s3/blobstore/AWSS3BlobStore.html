57  /**
58   * Provide AWS S3 specific extensions.
59   *
60   * @author Tibor Kiss, Andrei Savu
61   */
62  public class AWSS3BlobStore extends S3BlobStore {
63  
64     private final Provider<MultipartUploadStrategy> multipartUploadStrategy;
65     private final LoadingCache<String, AccessControlList> bucketAcls;
66     private final BlobToObject blob2Object;
67  
68     @Inject
69     AWSS3BlobStore(BlobStoreContext context, BlobUtils blobUtils, Supplier<Location> defaultLocation,
70              @Memoized Supplier<Set<? extends Location>> locations, AWSS3Client sync,
71              BucketToResourceMetadata bucket2ResourceMd, ContainerToBucketListOptions container2BucketListOptions,
72              BucketToResourceList bucket2ResourceList, ObjectToBlob object2Blob,
73              BlobToHttpGetOptions blob2ObjectGetOptions, BlobToObject blob2Object, ObjectToBlobMetadata object2BlobMd,
74              Provider<FetchBlobMetadata> fetchBlobMetadataProvider, LoadingCache<String, AccessControlList> bucketAcls,
75              Provider<MultipartUploadStrategy> multipartUploadStrategy) {
76        super(context, blobUtils, defaultLocation, locations, sync, bucket2ResourceMd, container2BucketListOptions,
77                 bucket2ResourceList, object2Blob, blob2ObjectGetOptions, blob2Object, object2BlobMd,
78                 fetchBlobMetadataProvider, bucketAcls);
79        this.multipartUploadStrategy = multipartUploadStrategy;
80        this.bucketAcls = bucketAcls;
81        this.blob2Object = blob2Object;
82     }
83  
84     @Override
85     public String putBlob(String container, Blob blob, PutOptions options) {
86        if (options.isMultipart()) {
87           // need to use a provider if the strategy object is stateful
88           return multipartUploadStrategy.get().execute(container, blob, options);
89  
90        } else if ((options instanceof AWSS3PutOptions) &&
91           (((AWSS3PutOptions) options).getStorageClass() == REDUCED_REDUNDANCY)) {
92           return putBlobWithReducedRedundancy(container, blob);
93  
94        } else {
95           return super.putBlob(container, blob, options);
96        }
97     }
98  
99     private String putBlobWithReducedRedundancy(String container, Blob blob) {
100       AWSS3PutObjectOptions options = new AWSS3PutObjectOptions();
101       try {
102          AccessControlList acl = bucketAcls.getUnchecked(container);
103          if (acl != null && acl.hasPermission(AccessControlList.GroupGranteeURI.ALL_USERS,
104                                               AccessControlList.Permission.READ)) {
105             options.withAcl(CannedAccessPolicy.PUBLIC_READ);
106          }
107          options.storageClass(ObjectMetadata.StorageClass.REDUCED_REDUNDANCY);
108 
109       } catch (CacheLoader.InvalidCacheLoadException e) {
110          // nulls not permitted from cache loader
111       }
112       return S3Client.class.cast(getContext().getProviderSpecificContext().getApi())
113          .putObject(container, blob2Object.apply(blob), options);
114    }
