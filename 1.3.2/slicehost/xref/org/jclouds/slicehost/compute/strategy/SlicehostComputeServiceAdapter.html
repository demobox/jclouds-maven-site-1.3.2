19  package org.jclouds.slicehost.compute.strategy;

21  import static com.google.common.base.Preconditions.checkNotNull;
22  
23  import javax.inject.Inject;
24  import javax.inject.Singleton;
25  
26  import org.jclouds.compute.ComputeService;
27  import org.jclouds.compute.ComputeServiceAdapter;
28  import org.jclouds.compute.domain.Template;
29  import org.jclouds.domain.Location;
30  import org.jclouds.domain.LoginCredentials;
31  import org.jclouds.location.suppliers.JustProvider;
32  import org.jclouds.slicehost.SlicehostClient;
33  import org.jclouds.slicehost.domain.Flavor;
34  import org.jclouds.slicehost.domain.Image;
35  import org.jclouds.slicehost.domain.Slice;
36  
37  /**
38   * defines the connection between the {@link SlicehostClient} implementation and the jclouds
39   * {@link ComputeService}
40   * 
41   */
42  @Singleton
43  public class SlicehostComputeServiceAdapter implements ComputeServiceAdapter<Slice, Flavor, Image, Location> {
44  
45     protected final SlicehostClient client;
46     protected final JustProvider locationSupplier;
47  
48     @Inject
49     protected SlicehostComputeServiceAdapter(SlicehostClient client, JustProvider locationSupplier) {
50        this.client = checkNotNull(client, "client");
51        this.locationSupplier = checkNotNull(locationSupplier, "locationSupplier");
52  
53     }
54  
55     @Override
56     public NodeAndInitialCredentials<Slice> createNodeWithGroupEncodedIntoName(String group, String name,
57              Template template) {
58        Slice server = client
59                 .createSlice(name, Integer.parseInt(template.getImage().getProviderId()), Integer.parseInt(template
60                          .getHardware().getProviderId()));
61  
62        return new NodeAndInitialCredentials<Slice>(server, server.getId() + "", LoginCredentials.builder().password(
63                 server.getRootPassword()).build());
64     }
65  
66     @Override
67     public Iterable<Flavor> listHardwareProfiles() {
68        return client.listFlavors();
69  
70     }
71  
72     @Override
73     public Iterable<Image> listImages() {
74        return client.listImages();
75     }
76  
77     @Override
78     public Iterable<Slice> listNodes() {
79        return client.listSlices();
80     }
81  
82     @SuppressWarnings("unchecked")
83     @Override
84     public Iterable<Location> listLocations() {
85        return (Iterable<Location>) locationSupplier.get();
86     }
87  
88     @Override
89     public Slice getNode(String id) {
90        int serverId = Integer.parseInt(id);
91        return client.getSlice(serverId);
92     }
93  
94     @Override
95     public void destroyNode(String id) {
96        int serverId = Integer.parseInt(id);
97        // if false server wasn't around in the first place
98        client.destroySlice(serverId);
99     }
100 
101    @Override
102    public void rebootNode(String id) {
103       int sliceId = Integer.parseInt(id);
104       client.hardRebootSlice(sliceId);
105    }
106 
107    @Override
108    public void resumeNode(String id) {
109       throw new UnsupportedOperationException("suspend not supported");
110    }
111 
112    @Override
113    public void suspendNode(String id) {
114       throw new UnsupportedOperationException("suspend not supported");
115    }
116 
117 }
