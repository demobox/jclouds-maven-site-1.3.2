19  package org.jclouds.slicehost.xml;
20  
21  import java.util.Set;
22  
23  import javax.inject.Inject;
24  
25  import org.jclouds.http.functions.ParseSax;
26  import org.jclouds.slicehost.domain.Image;
27  import org.xml.sax.Attributes;
28  import org.xml.sax.SAXException;
29  
30  import com.google.common.collect.Sets;
31  
32  /**
33   * @author Adrian Cole
34   */
35  public class ImagesHandler extends ParseSax.HandlerWithResult<Set<? extends Image>> {
36     private StringBuilder currentText = new StringBuilder();
37  
38     private Set<Image> images = Sets.newLinkedHashSet();
39     private final ImageHandler locationHandler;
40  
41     @Inject
42     public ImagesHandler(ImageHandler locationHandler) {
43        this.locationHandler = locationHandler;
44     }
45  
46     public Set<? extends Image> getResult() {
47        return images;
48     }
49  
50     @Override
51     public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
52        locationHandler.startElement(uri, localName, qName, attributes);
53     }
54  
55     @Override
56     public void endElement(String uri, String localName, String qName) throws SAXException {
57        locationHandler.endElement(uri, localName, qName);
58        if (qName.equals("image") && currentText.toString().trim().equals("")) {
59           this.images.add(locationHandler.getResult());
60        }
61        currentText = new StringBuilder();
62     }
63  
64     public void characters(char ch[], int start, int length) {
65        locationHandler.characters(ch, start, length);
66        currentText.append(ch, start, length);
67     }
68  }
