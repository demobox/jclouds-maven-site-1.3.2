62  @Singleton
63  public class ServerInfoToNodeMetadata implements Function<ServerInfo, NodeMetadata> {
64     public static final Map<ServerStatus, NodeState> serverStatusToNodeState = ImmutableMap
65           .<ServerStatus, NodeState> builder().put(ServerStatus.ACTIVE, NodeState.RUNNING)//
66           .put(ServerStatus.STOPPED, NodeState.SUSPENDED)//
67           .put(ServerStatus.PAUSED, NodeState.SUSPENDED)//
68           .put(ServerStatus.DUMPED, NodeState.PENDING)//
69           .put(ServerStatus.DEAD, NodeState.TERMINATED)//
70           .put(ServerStatus.UNRECOGNIZED, NodeState.UNRECOGNIZED)//
71           .build();
72  
73     private final Function<Server, String> getImageIdFromServer;
74     private final Function<String, Image> findImageForId;
75     private final Supplier<Location> locationSupplier;
76     private final Function<Device, Volume> deviceToVolume;
77  
78     @Inject
79     ServerInfoToNodeMetadata(Function<Server, String> getImageIdFromServer, Function<String, Image> findImageForId,
80           Function<Device, Volume> deviceToVolume, Supplier<Location> locationSupplier) {
81        this.locationSupplier = checkNotNull(locationSupplier, "locationSupplier");
82        this.deviceToVolume = checkNotNull(deviceToVolume, "deviceToVolume");
83        this.findImageForId = checkNotNull(findImageForId, "findImageForId");
84        this.getImageIdFromServer = checkNotNull(getImageIdFromServer, "getImageIdFromServer");
85     }
86  
87     @SuppressWarnings({ "unchecked", "rawtypes" })
88     @Override
89     public NodeMetadata apply(ServerInfo from) {
90        NodeMetadataBuilder builder = new NodeMetadataBuilder();
91        builder.ids(from.getUuid());
92        builder.name(from.getName());
93        builder.location(locationSupplier.get());
94        builder.group(parseGroupFromName(from.getName()));
95  
96        String imageId = getImageIdFromServer.apply(from);
97        if (imageId != null) {
98           Image image = findImageForId.apply(imageId);
99           if (image != null) {
100             builder.operatingSystem(image.getOperatingSystem());
101          }
102       }
103       builder.hardware(new HardwareBuilder().ids(from.getUuid())
104             .processors(ImmutableList.of(new Processor(1, from.getCpu()))).ram(from.getMem())
105             .volumes((List) ImmutableList.of(Iterables.transform(from.getDevices().values(), deviceToVolume))).build());
106       builder.state(serverStatusToNodeState.get(from.getStatus()));
107       builder.publicAddresses(ImmutableSet.<String> of(from.getNics().get(0).getDhcp()));
108       builder.privateAddresses(ImmutableSet.<String> of());
109       return builder.build();
110    }

112    @Singleton
113    public static final class DeviceToVolume implements Function<Device, Volume> {
114       @Resource
115       protected Logger logger = Logger.NULL;
116 
117       private final LoadingCache<String, DriveInfo> cache;
118 
119       @Inject
120       public DeviceToVolume(LoadingCache<String, DriveInfo> cache) {
121          this.cache = checkNotNull(cache, "cache");
122       }
123 
124       @Override
125       public Volume apply(Device input) {
126          VolumeBuilder builder = new VolumeBuilder();
127          builder.id(input.getId());
128          try {
129             DriveInfo drive = cache.getUnchecked(input.getDriveUuid());
130             builder.size(drive.getSize() / 1024 / 1024f);
131          } catch (NullPointerException e) {
132             logger.debug("drive %s not found", input.getDriveUuid());
133          } catch (UncheckedExecutionException e) {
134             logger.warn(e, "error finding drive %s: %s", input.getDriveUuid(), e.getMessage());
135          }
136          return new VolumeBuilder().durable(true).type(Volume.Type.NAS).build();
137       }
138    }

140    /**
141     * When we create the boot drive of the server, by convention we set the name
142     * to the image it came from.
143     * 
144     * @author Adrian Cole
145     * 
146     */
147    @Singleton
148    public static class GetImageIdFromServer implements Function<Server, String> {
149       @Resource
150       protected Logger logger = Logger.NULL;
151 
152       private final LoadingCache<String, DriveInfo> cache;
153 
154       @Inject
155       public GetImageIdFromServer(LoadingCache<String, DriveInfo> cache) {
156          this.cache = cache;
157       }
158 
159       @Override
160       public String apply(Server from) {
161          String imageId = null;
162          String bootDeviceId = Iterables.get(from.getBootDeviceIds(), 0);
163          Device bootDevice = from.getDevices().get(bootDeviceId);
164          if (bootDevice != null) {
165             try {
166                DriveInfo drive = cache.getUnchecked(bootDevice.getDriveUuid());
167                imageId = drive.getName();
168             } catch (NullPointerException e) {
169                logger.debug("drive %s not found", bootDevice.getDriveUuid());
170             } catch (UncheckedExecutionException e) {
171                logger.warn(e, "error finding drive %s: %s", bootDevice.getDriveUuid(), e.getMessage());
172             }
173          }
174          return imageId;
175       }
176    }
