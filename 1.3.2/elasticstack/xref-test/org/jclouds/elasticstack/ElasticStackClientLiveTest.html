254

256       assertEquals(client.getServerInfo(server.getUuid()).getStatus(), ServerStatus.ACTIVE);
257 
258       client.shutdownServer(server.getUuid());
259       // behavior on shutdown depends on how your server OS is set up to respond to an ACPI power
260       // button signal
261       assert (client.getServerInfo(server.getUuid()).getStatus() == ServerStatus.ACTIVE || client.getServerInfo(
262                server.getUuid()).getStatus() == ServerStatus.STOPPED);

297    protected void doConnectViaSsh(Server server, LoginCredentials creds) throws IOException {
298       SshClient ssh = Guice.createInjector(new SshjSshClientModule()).getInstance(SshClient.Factory.class).create(
299                new IPSocket(server.getVnc().getIp(), 22), creds);
300       try {
301          ssh.connect();
302          ExecResponse hello = ssh.exec("echo hello");
303          assertEquals(hello.getOutput().trim(), "hello");
304          System.err.println(ssh.exec("df -k").getOutput());
305          System.err.println(ssh.exec("mount").getOutput());
306          System.err.println(ssh.exec("uname -a").getOutput());
307       } finally {
308          if (ssh != null)
309             ssh.disconnect();
310       }
311    }

338    @Test(dependsOnMethods = "testWeCanReadAndWriteToDrive")
339    public void testWeCopyADriveContentsViaGzip() throws IOException {
340       try {
341          drive3 = client
342                   .createDrive(new CreateDriveRequest.Builder().name(prefix + "3").size(1 * 1024 * 1024l).build());
343          System.err.println("before image; drive 2" + client.getDriveInfo(drive2.getUuid()));
344          System.err.println("before image; drive 3" + client.getDriveInfo(drive3.getUuid()));
345          client.imageDrive(drive2.getUuid(), drive3.getUuid());
346          assert driveNotClaimed.apply(drive3) : client.getDriveInfo(drive3.getUuid());
347          assert driveNotClaimed.apply(drive2) : client.getDriveInfo(drive2.getUuid());
348          System.err.println("after image; drive 2" + client.getDriveInfo(drive2.getUuid()));
349          System.err.println("after image; drive 3" + client.getDriveInfo(drive3.getUuid()));
350          assertEquals(Strings2.toStringAndClose(client.readDrive(drive3.getUuid(), 0, 3).getInput()), "foo");
351       } finally {
352          client.destroyDrive(drive2.getUuid());
353          client.destroyDrive(drive3.getUuid());
354       }
355    }
