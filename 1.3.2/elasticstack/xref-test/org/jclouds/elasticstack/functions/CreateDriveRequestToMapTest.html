37  @Test(groups = { "unit" })
38  public class CreateDriveRequestToMapTest {
39  
40     private static final CreateDriveRequestToMap BASEDRIVE_TO_MAP = Guice.createInjector().getInstance(
41           CreateDriveRequestToMap.class);
42  
43     public void testBasics() {
44        assertEquals(BASEDRIVE_TO_MAP.apply(new CreateDriveRequest.Builder().name("foo").size(100l).build()),
45              ImmutableMap.of("name", "foo", "size", "100"));
46     }
47  
48     public void testComplete() throws IOException {
49        CreateDriveRequest one = new CreateDriveRequest.Builder()
50              .name("Ubuntu 10.10 Server Edition Linux 64bit Preinstalled System")
51              //
52              .size(8589934592l)//
53              .claimType(ClaimType.SHARED)//
54              .readers(ImmutableSet.of("ffffffff-ffff-ffff-ffff-ffffffffffff"))//
55              .tags(ImmutableSet.of("tag1", "tag2")).userMetadata(ImmutableMap.of("foo", "bar", "baz", "raz"))//
56              .encryptionCipher("aes-xts-plain").avoid(ImmutableSet.of("avoid1")).build();
57        assertEquals(
58              BASEDRIVE_TO_MAP.apply(one),
59              ImmutableMap.builder().put("name", "Ubuntu 10.10 Server Edition Linux 64bit Preinstalled System")
60                    .put("size", "8589934592").put("claim:type", "shared")
61                    .put("readers", "ffffffff-ffff-ffff-ffff-ffffffffffff").put("tags", "tag1 tag2")
62                    .put("user:foo", "bar").put("user:baz", "raz").put("encryption:cipher", "aes-xts-plain")
63                    .put("avoid", "avoid1").build()
64  
65        );
66  
67     }
68  }
