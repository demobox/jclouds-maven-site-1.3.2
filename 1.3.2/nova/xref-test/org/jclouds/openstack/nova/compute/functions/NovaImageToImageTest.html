41  @Test(groups = "unit")

44     @Test
45     public void testApplyWhereImageNotFound() {
46        Image image = new ImageBuilder()
47              .name("CentOS 5.2")
48              .operatingSystem(
49                    new OperatingSystem.Builder().family(OsFamily.CENTOS).version("5.2").description("CentOS 5.2")
50                          .is64Bit(true).build()).description("CentOS 5.2").ids("2").version("1286712000000")
51              .uri(URI.create("https://servers.api.rackspacecloud.com/v1.1/1234/images/1")).build();
52        Image parsedImage = convertImage();
53  
54        assertEquals(parsedImage, image);
55     }
56  
57     public static Image convertImage() {
58        org.jclouds.openstack.nova.domain.Image image = ParseImageFromJsonResponseTest.parseImage();
59  
60        NovaImageToImage parser = new NovaImageToImage(new NovaImageToOperatingSystem(
61              new BaseComputeServiceContextModule() {
62              }.provideOsVersionMap(new ComputeServiceConstants.ReferenceData(), Guice.createInjector(new GsonModule())
63                    .getInstance(Json.class))));
64  
65        return parser.apply(image);
66     }
