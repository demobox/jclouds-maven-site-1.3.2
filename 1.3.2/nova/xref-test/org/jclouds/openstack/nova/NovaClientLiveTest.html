220    public void testCreateServer() throws Exception {
221       String flavorId = "1";
222       Server server = null;
223       while (server == null) {
224          String serverName = serverPrefix + "createserver" + new SecureRandom().nextInt();
225          try {
226             server = client.createServer(serverName, imageId, flavorId, withFile("/etc/jclouds.txt",
227                      "nova".getBytes()).withMetadata(metadata));
228          } catch (UndeclaredThrowableException e) {
229             HttpResponseException htpe = (HttpResponseException) e.getCause().getCause();
230             if (htpe.getResponse().getStatusCode() == 400)
231                continue;
232             throw e;
233          }
234       }
235       assertNotNull(server.getAdminPass());
236       serverId = server.getId();
237       adminPass = server.getAdminPass();
238       assertEquals(server.getStatus(), ServerStatus.BUILD);
239       blockUntilServerActive(serverId);
240    }

242    private void blockUntilServerActive(int serverId) throws InterruptedException {
243       Server currentDetails = null;
244       for (currentDetails = client.getServer(serverId); currentDetails.getStatus() != ServerStatus.ACTIVE; currentDetails = client
245                .getServer(serverId)) {
246          System.out.printf("blocking on status active%n%s%n", currentDetails);
247          Thread.sleep(5 * 1000);
248       }
249    }

292    /**
293     * this tests "personality" as the file looked up was sent during server creation
294     */
295    private void checkPassOk(Server newDetails, String pass) throws IOException {
296       try {
297          doCheckPass(newDetails, pass);
298       } catch (SshException e) {// try twice in case there is a network timeout
299          try {
300             Thread.sleep(10 * 1000);
301          } catch (InterruptedException e1) {
302          }
303          doCheckPass(newDetails, pass);
304       }
305    }
306 
307    private void doCheckPass(Server newDetails, String pass) throws IOException {
308       IPSocket socket = new IPSocket(Iterables.get(newDetails.getAddresses().getPublicAddresses(), 0).getAddress(), 22);
309       socketTester.apply(socket);
310 
311       SshClient client = sshFactory.create(socket, LoginCredentials.builder().user("root").password(pass).build());
312       try {
313          client.connect();
314          Payload etcPasswd = client.get("/etc/jclouds.txt");
315          String etcPasswdContents = Strings2.toStringAndClose(etcPasswd.getInput());
316          assertEquals("nova", etcPasswdContents.trim());
317       } finally {
318          if (client != null)
319             client.disconnect();
320       }
321    }
