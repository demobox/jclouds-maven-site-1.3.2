218    /**
219     * This implementation invokes {@link AtmosAsyncClient#listDirectory}
220     */
221    @Override
222    public ListenableFuture<PageSet<? extends StorageMetadata>> list(String container,
223             org.jclouds.blobstore.options.ListContainerOptions options) {
224       container = AtmosUtils.adjustContainerIfDirOptionPresent(container, options);
225       ListOptions nativeOptions = container2ContainerListOptions.apply(options);
226       ListenableFuture<BoundedSet<? extends DirectoryEntry>> returnVal = async.listDirectory(container, nativeOptions);
227       ListenableFuture<PageSet<? extends StorageMetadata>> list = Futures.compose(returnVal, container2ResourceList,
228                service);
229       return (ListenableFuture<PageSet<? extends StorageMetadata>>) (options.isDetailed() ? Futures.compose(list,
230                fetchBlobMetadataProvider.get().setContainerName(container), service) : list);
231    }

233    /**
234     * This implementation invokes {@link AtmosAsyncClient#createFile}
235     * <p/>
236     * Since there is no etag support in atmos, we just return the path.
237     */
238    @Override
239    public ListenableFuture<String> putBlob(final String container, final Blob blob) {
240       final org.jclouds.atmos.options.PutOptions options = new org.jclouds.atmos.options.PutOptions();
241       try {
242          if (isPublic.getUnchecked(container + "/"))
243             options.publicRead();
244       } catch (CacheLoader.InvalidCacheLoadException e) {
245          // nulls not permitted
246       }
247       return Futures.makeListenable(service.submit(new Callable<String>() {
248 
249          @Override
250          public String call() throws Exception {
251             return AtmosUtils.putBlob(sync, crypto, blob2Object, container, blob, options);
252          }
253 
254          @Override
255          public String toString() {
256             return "putBlob(" + container + "," + blob.getMetadata().getName() + ")";
257          }
258       }), service);
259 
260    }
