189    /**
190     * Write a {@link Blob} {@link Payload} into a file
191     * 
192     * @param container
193     * @param blobKey
194     * @param payload
195     * @throws IOException
196     */
197    @Override
198    public void writePayloadOnFile(String container, String blobKey, Payload payload) throws IOException {
199       filesystemContainerNameValidator.validate(container);
200       filesystemBlobKeyValidator.validate(blobKey);
201       File outputFile = getFileForBlobKey(container, blobKey);
202       if (payload.getRawContent().equals(outputFile)){
203          // we shouldn't re-copy the same contents
204          return;
205       }
206       FileOutputStream output = null;
207       try {
208          Files.createParentDirs(outputFile);
209          if (payload.getRawContent() instanceof File)
210             Files.copy(File.class.cast(payload.getRawContent()), outputFile);
211          else {
212             output = new FileOutputStream(outputFile);
213             payload.writeTo(output);
214          }
215       } catch (IOException ex) {
216          if (outputFile != null) {
217             outputFile.delete();
218          }
219          throw ex;
220       } finally {
221          Closeables.closeQuietly(output);
222          payload.release();
223       }
224    }

308    /**
309     * Facility method used to concatenate path tokens normalizing separators
310     * 
311     * @param pathTokens
312     *           all the string in the proper order that must be concatenated in order to obtain the
313     *           filename
314     * @return the resulting string
315     */
316    protected String buildPathStartingFromBaseDir(String... pathTokens) {
317       String normalizedToken = removeFileSeparatorFromBorders(normalize(baseDirectory), true);
318       StringBuilder completePath = new StringBuilder(normalizedToken);
319       if (pathTokens != null && pathTokens.length > 0) {
320          for (int i = 0; i < pathTokens.length; i++) {
321             if (pathTokens[i] != null) {
322                normalizedToken = removeFileSeparatorFromBorders(normalize(pathTokens[i]), false);
323                completePath.append(File.separator).append(normalizedToken);
324             }
325          }
326       }
327       return completePath.toString();
328    }

330    /**
331     * Substitutes all the file separator occurrences in the path with a file separator for the
332     * current operative system
333     * 
334     * @param pathToBeNormalized
335     * @return
336     */
337    private String normalize(String pathToBeNormalized) {
338       if (null != pathToBeNormalized && pathToBeNormalized.contains(BACK_SLASH)) {
339          if (!BACK_SLASH.equals(File.separator)) {
340             return pathToBeNormalized.replaceAll(BACK_SLASH, File.separator);
341          }
342       }
343       return pathToBeNormalized;
344    }

346    /**
347     * Remove leading and trailing {@link File.separator} character from the string.
348     * 
349     * @param pathToBeCleaned
350     * @param remove
351     *           only trailing separator char from path
352     * @return
353     */
354    private String removeFileSeparatorFromBorders(String pathToBeCleaned, boolean onlyTrailing) {
355       if (null == pathToBeCleaned || pathToBeCleaned.equals(""))
356          return pathToBeCleaned;
357 
358       int beginIndex = 0;
359       int endIndex = pathToBeCleaned.length();
360 
361       // search for separator chars
362       if (!onlyTrailing) {
363          if (pathToBeCleaned.substring(0, 1).equals(File.separator))
364             beginIndex = 1;
365       }
366       if (pathToBeCleaned.substring(pathToBeCleaned.length() - 1).equals(File.separator))
367          endIndex--;
368 
369       return pathToBeCleaned.substring(beginIndex, endIndex);
370    }

372    /**
373     * Removes recursively the directory structure of a complex blob key, only if the directory is
374     * empty
375     * 
376     * @param container
377     * @param normalizedKey
378     */
379    private void removeDirectoriesTreeOfBlobKey(String container, String blobKey) {
380       String normalizedBlobKey = normalize(blobKey);
381       // exists is no path is present in the blobkey
382       if (!normalizedBlobKey.contains(File.separator))
383          return;
384 
385       File file = new File(normalizedBlobKey);
386       // TODO
387       // "/media/data/works/java/amazon/jclouds/master/filesystem/aa/bb/cc/dd/eef6f0c8-0206-460b-8870-352e6019893c.txt"
388       String parentPath = file.getParent();
389       // no need to manage "/" parentPath, because "/" cannot be used as start
390       // char of blobkey
391       if (null != parentPath || "".equals(parentPath)) {
392          // remove parent directory only it's empty
393          File directory = new File(buildPathStartingFromBaseDir(container, parentPath));
394          String[] children = directory.list();
395          if (null == children || children.length == 0) {
396             directory.delete();
397             // recursively call for removing other path
398             removeDirectoriesTreeOfBlobKey(container, parentPath);
399          }
400       }
401    }

403    private File openFolder(String folderName) throws IOException {
404       String baseFolderName = buildPathStartingFromBaseDir(folderName);
405       File folder = new File(baseFolderName);
406       if (folder.exists()) {
407          if (!folder.isDirectory()) {
408             throw new IOException("Resource " + baseFolderName + " isn't a folder.");
409          }
410       }
411       return folder;
412    }

414    private class FileIterator implements Iterator<String> {
415       int currentFileIndex = 0;
416       File[] children = new File[0];
417       File currentFile = null;
418 
419       public FileIterator(String fileName, FileFilter filter) {
420          File file = new File(fileName);
421          if (file.exists() && file.isDirectory()) {
422             children = file.listFiles(filter);
423          }
424       }
425 
426       @Override
427       public boolean hasNext() {
428          return currentFileIndex < children.length;
429       }
430 
431       @Override
432       public String next() {
433          currentFile = children[currentFileIndex++];
434          return currentFile.getName();
435       }
436 
437       @Override
438       public void remove() {
439          if (currentFile != null && currentFile.exists()) {
440             if (!currentFile.delete()) {
441                throw new RuntimeException("An error occurred deleting " + currentFile.getName());
442             }
443          }
444       }
445    }

447    private void populateBlobKeysInContainer(File directory, Set<String> blobNames) {
448       File[] children = directory.listFiles();
449       for (File child : children) {
450          if (child.isFile()) {
451             blobNames.add(child.getAbsolutePath());
452          } else if (child.isDirectory()) {
453             populateBlobKeysInContainer(child, blobNames);
454          }
455       }
456    }

458    /**
459     * Creates a directory and returns the result
460     * 
461     * @param container
462     * @param directory
463     * @return true if the directory was created, otherwise false
464     */
465    protected boolean createDirectoryWithResult(String container, String directory) {
466       String directoryFullName = buildPathStartingFromBaseDir(container, directory);
467       logger.debug("Creating directory %s", directoryFullName);
468 
469       // cannot use directoryFullName, because the following method rebuild
470       // another time the path starting from base directory
471       if (buildPathAndChecksIfDirectoryExists(container, directory)) {
472          logger.debug("Directory %s already exists", directoryFullName);
473          return false;
474       }
475 
476       File directoryToCreate = new File(directoryFullName);
477       boolean result = directoryToCreate.mkdirs();
478       return result;
479    }
