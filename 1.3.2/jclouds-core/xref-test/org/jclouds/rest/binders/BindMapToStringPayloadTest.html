40  /**
41   * Tests behavior of {@code BindMapToStringPayload}
42   * 
43   * @author Adrian Cole
44   */
45  @Test(groups = "unit")
46  public class BindMapToStringPayloadTest {
47     static interface TestPayload {
48        @org.jclouds.rest.annotations.Payload("name {fooble}")
49        void testPayload(@PathParam("foo") String path);
50  
51        void noPayload(String path);
52     }
53  
54     @Test
55     public void testCorrect() throws SecurityException, NoSuchMethodException {
56        Method testPayload = TestPayload.class.getMethod("testPayload", String.class);
57        GeneratedHttpRequest<TestPayload> request = GeneratedHttpRequest.<TestPayload>requestBuilder()
58              .declaring(TestPayload.class).javaMethod(testPayload).args(ImmutableList.<Object> of("robot"))
59              .method(HttpMethod.POST).endpoint(URI.create("http://localhost")).build();
60  
61        GeneratedHttpRequest<TestPayload> newRequest = binder()
62              .bindToRequest(request, ImmutableMap.of("fooble", "robot"));
63  
64        assertEquals(newRequest.getRequestLine(), request.getRequestLine());
65        assertEquals(newRequest.getPayload().getRawContent(), "name robot");
66     }
67  
68     @Test(expectedExceptions = IllegalArgumentException.class)
69     public void testMustHavePayloadAnnotation() throws SecurityException, NoSuchMethodException {
70        Method noPayload = TestPayload.class.getMethod("noPayload", String.class);
71        GeneratedHttpRequest<TestPayload> request = GeneratedHttpRequest.<TestPayload>requestBuilder()
72              .declaring(TestPayload.class).javaMethod(noPayload).args(ImmutableList.<Object> of("robot"))
73              .method(HttpMethod.POST).endpoint(URI.create("http://localhost")).build();
74        binder().bindToRequest(request, ImmutableMap.of("fooble", "robot"));
75     }
76  
77     @Test(expectedExceptions = IllegalArgumentException.class)
78     public void testMustBeMap() {
79        BindMapToStringPayload binder = binder();
80        HttpRequest request = new HttpRequest(HttpMethod.POST, URI.create("http://localhost"));
81        binder.bindToRequest(request, new File("foo"));
82     }
83  
84     @Test(expectedExceptions = NullPointerException.class)
85     public void testNullIsBad() {
86        BindMapToStringPayload binder = binder();
87        HttpRequest request = HttpRequest.builder().method("GET").endpoint(URI.create("http://momma")).build();
88        binder.bindToRequest(request, null);
89     }
90  
91     public BindMapToStringPayload binder() {
92        return new BindMapToStringPayload(Providers.<UriBuilder> of(new UriBuilderImpl()));
93     }
