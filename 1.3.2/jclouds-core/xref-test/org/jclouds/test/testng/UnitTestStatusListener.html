19  package org.jclouds.test.testng;
20  
21  import java.util.concurrent.atomic.AtomicInteger;
22  
23  import org.testng.IClass;
24  import org.testng.ITestContext;
25  import org.testng.ITestListener;
26  import org.testng.ITestResult;
27  
28  /**
29   * adapted from the following class:
30   * 
31   * @see org.UnitTestStatusListener.test.testng.UnitTestTestNGListener
32   * @author Adrian Cole
33   * @author dpospisi@redhat.com
34   * @author Mircea.Markus@jboss.com
35   */
36  public class UnitTestStatusListener implements ITestListener {
37  
38     /**
39      * Holds test classes actually running in all threads.
40      */
41     private ThreadLocal<IClass> threadTestClass = new ThreadLocal<IClass>();
42     private ThreadLocal<Long> threadTestStart = new ThreadLocal<Long>();
43  
44     private AtomicInteger failed = new AtomicInteger(0);
45     private AtomicInteger succeded = new AtomicInteger(0);
46     private AtomicInteger skipped = new AtomicInteger(0);
47  
48     public void onTestStart(ITestResult res) {
49        System.out.println("Starting test " + getTestDesc(res));
50        threadTestClass.set(res.getTestClass());
51        threadTestStart.set(System.currentTimeMillis());
52     }
53  
54     synchronized public void onTestSuccess(ITestResult arg0) {
55        System.out.println(getThreadId() + " Test " + getTestDesc(arg0) + " succeeded: "
56                 + (System.currentTimeMillis() - threadTestStart.get()) + "ms");
57        succeded.incrementAndGet();
58        printStatus();
59     }
60  
61     synchronized public void onTestFailure(ITestResult arg0) {
62        System.out.println(getThreadId() + " Test " + getTestDesc(arg0) + " failed.");
63        failed.incrementAndGet();
64        printStatus();
65     }
66  
67     synchronized public void onTestSkipped(ITestResult arg0) {
68        System.out.println(getThreadId() + " Test " + getTestDesc(arg0) + " skipped.");
69        skipped.incrementAndGet();
70        printStatus();
71     }
72  
73     public void onTestFailedButWithinSuccessPercentage(ITestResult arg0) {
74     }
75  
76     public void onStart(ITestContext arg0) {
77     }
78  
79     public void onFinish(ITestContext arg0) {
80     }
81  
82     private String getThreadId() {
83        return "[" + Thread.currentThread().getName() + "]";
84     }
85  
86     private String getTestDesc(ITestResult res) {
87        return res.getMethod().getMethodName() + "(" + res.getTestClass().getName() + ")";
88     }
89  
90     private void printStatus() {
91        System.out.println("Test suite progress: tests succeeded: " + succeded.get() + ", failed: "
92                 + failed.get() + ", skipped: " + skipped.get() + ".");
93     }
94  }
