39  @Test(groups="performance")
40  public abstract class PerformanceTest {
41     protected static int LOOP_COUNT = 1000;
42     protected static int THREAD_COUNT = 100;
43     protected ExecutorService exec;
44  
45     @BeforeTest
46     public void setupExecutorService() {
47        exec = DynamicExecutors.newScalingThreadPool(1, THREAD_COUNT, 1000);
48     }
49  
50     @AfterTest
51     public void teardownExecutorService() {
52        exec.shutdownNow();
53        exec = null;
54     }
55  
56     /**
57      * Executes a list of Runnable tasks in {@link #THREAD_COUNT} simultaneous threads, and outputs
58      * the timing results.
59      * <p>
60      * This method is careful to time only the actual task execution time, not the overhead of
61      * creating and queuing the tasks. We also use CountDownLatches to ensure that all tasks start at
62      * the same time, so concurrency is fully tested without ramp-up or ramp-down times.
63      * <p>
64      * This code is heavily based on Listing 5.11 in "Java Concurrency in Practice" by Brian Goetz et
65      * al, Addison-Wesley Professional.
66      * 
67      * @see {@link DateService} for example usage.
68      * 
69      * @param performanceTestName
70      * @param tasks
71      * @throws InterruptedException
72      * @throws ExecutionException
73      * @throws Throwable
74      */
75     protected void executeMultiThreadedPerformanceTest(String performanceTestName,
76              List<Runnable> tasks) throws InterruptedException, ExecutionException, Throwable {
77        CompletionService<Throwable> completer = new ExecutorCompletionService<Throwable>(exec);
78        final CountDownLatch startGate = new CountDownLatch(1);
79        final CountDownLatch endGate = new CountDownLatch(THREAD_COUNT);
80  
81        for (int i = 0; i < THREAD_COUNT; i++) {
82           final Runnable task = tasks.get(i % tasks.size());
83           // Wrap task so we can count down endGate.
84           completer.submit(new Callable<Throwable>() {
85              public Throwable call() {
86                 try {
87                    startGate.await(); // Wait to start simultaneously
88                    task.run();
89                    return null;
90                 } catch (Throwable t) {
91                    return t;
92                 } finally {
93                    endGate.countDown(); // Notify that I've finished
94                 }
95              }
96           });
97        }
98  
99        // Only time the execution time for all tasks, not start/stop times.
100       long startTime = System.nanoTime();
101       startGate.countDown(); // Trigger start of all tasks
102       endGate.await();
103       long endTime = System.nanoTime() - startTime;
104 
105       // Check for assertion failures
106       Throwable t;
107       for (int i = 0; i < THREAD_COUNT; i++) {
108          t = completer.take().get();
109          if (t != null) {
110             throw t;
111          }
112       }
113       if (performanceTestName != null) {
114          System.out.printf("TIMING: Multi-threaded %s took %.3fms for %d threads\n",
115                   performanceTestName, ((double) endTime / 1000000), THREAD_COUNT);
116       }
117    }
118 
119    protected void executeMultiThreadedCorrectnessTest(List<Runnable> tasks)
120             throws InterruptedException, ExecutionException, Throwable {
121       executeMultiThreadedPerformanceTest(null, tasks);
122    }
