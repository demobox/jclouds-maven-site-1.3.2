109    protected <T> T parseThingFromReaderOrNull(String toFind, JsonReader reader, Type type) throws IOException {
110       AtomicReference<String> name = new AtomicReference<String>();
111       JsonToken token = reader.peek();
112       for (; token != JsonToken.END_DOCUMENT && nnn(toFind, reader, token, name); token = skipAndPeek(token, reader))
113          ;
114       T val = gson.<T>fromJson(reader, type);
115       reader.close();
116       return val;
117    }
118 
119    protected boolean nnn(String toFind, JsonReader reader, JsonToken token, AtomicReference<String> name)
120          throws IOException {
121       if (token == JsonToken.NAME) {
122          String name2 = reader.nextName();
123          if (toFind.equals(name2)) {
124             name.set(name2);
125             return false;
126          }
127       }
128       return true;
129 
130    }
131 
132    public JsonToken skipAndPeek(JsonToken token, JsonReader reader) throws IOException {
133       switch (token) {
134       case BEGIN_ARRAY:
135          reader.beginArray();
136          break;
137       case END_ARRAY:
138          reader.endArray();
139          break;
140       case BEGIN_OBJECT:
141          reader.beginObject();
142          break;
143       case END_OBJECT:
144          reader.endObject();
145          break;
146       case NAME:
147          // NOTE that we have already advanced on NAME in the eval block;
148          break;
149       case STRING:
150          reader.nextString();
151          break;
152       case NUMBER:
153          reader.nextString();
154          break;
155       case BOOLEAN:
156          reader.nextBoolean();
157          break;
158       case NULL:
159          reader.nextNull();
160          break;
161       case END_DOCUMENT:
162          break;
163       }
164       return reader.peek();
165    }
