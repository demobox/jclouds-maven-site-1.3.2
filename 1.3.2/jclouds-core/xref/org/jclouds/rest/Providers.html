64     public static Iterable<String> getSupportedProvidersOfType(
65           @SuppressWarnings("rawtypes") Class<? extends RestContextBuilder> type) {
66        Properties properties = new Properties();
67        try {
68           properties.load(SaxUtils.class.getResourceAsStream("/rest.properties"));
69        } catch (IOException e) {

75     public static Iterable<String> getSupportedProvidersOfTypeInProperties(
76           @SuppressWarnings("rawtypes") final Class<? extends RestContextBuilder> type, final Properties properties) {
77        return filter(transform(filter(properties.entrySet(), new Predicate<Map.Entry<Object, Object>>() {
78  
79           @Override
80           public boolean apply(Entry<Object, Object> input) {
81              String keyString = input.getKey().toString();
82              return keyString.endsWith(".contextbuilder") || keyString.endsWith(".sync");
83           }
84  
85        }), new Function<Map.Entry<Object, Object>, String>() {
86  
87           @Override
88           public String apply(Entry<Object, Object> from) {
89              String keyString = from.getKey().toString();
90              try {
91                 String provider = get(Splitter.on('.').split(keyString), 0);
92                 Class<RestContextBuilder<Object, Object>> clazz = Providers.resolveContextBuilderClass(provider,
93                       properties);
94                 if (clazz != null && type.isAssignableFrom(clazz))
95                    return provider;
96              } catch (ClassNotFoundException e) {
97              } catch (Exception e) {
98                 propagate(e);
99              }
100             return null;
101          }
102 
103       }), notNull());
104    }

106    @SuppressWarnings("unchecked")
107    public static <S, A> Class<RestContextBuilder<S, A>> resolveContextBuilderClass(String provider,
108          Properties properties) throws ClassNotFoundException, IllegalArgumentException, SecurityException,
109          InstantiationException, IllegalAccessException, InvocationTargetException, NoSuchMethodException {
110       String contextBuilderClassName = properties.getProperty(provider + ".contextbuilder");
111       String syncClassName = properties.getProperty(provider + ".sync");
112       String asyncClassName = properties.getProperty(provider + ".async");
113       if (syncClassName != null) {
114          checkArgument(asyncClassName != null, "please configure async class for " + syncClassName);
115          if(ClassLoadingUtils.loadClass(Providers.class, syncClassName) == null) {
116              throw new ClassNotFoundException();
117          }
118          if(ClassLoadingUtils.loadClass(Providers.class, asyncClassName) == null) {
119              throw new ClassNotFoundException();
120          }
121          return (Class<RestContextBuilder<S, A>>) (contextBuilderClassName != null ? ClassLoadingUtils.
122                  loadClass(Providers.class, contextBuilderClassName) : RestContextBuilder.class);
123       } else {
124          checkArgument(contextBuilderClassName != null, "please configure contextbuilder class for " + provider);
125          return (Class<RestContextBuilder<S, A>>)  ClassLoadingUtils.
126                  loadClass(Providers.class, contextBuilderClassName);
127       }
128    }

130    public static <S, A> RestContextBuilder<S, A> initContextBuilder(
131          Class<RestContextBuilder<S, A>> contextBuilderClass, @Nullable Class<S> sync, @Nullable Class<A> async,
132          Properties properties) throws ClassNotFoundException, IllegalArgumentException, SecurityException,
133          InstantiationException, IllegalAccessException, InvocationTargetException, NoSuchMethodException {
134       checkArgument(properties != null, "please configure properties for " + contextBuilderClass);
135       try {
136          return (RestContextBuilder<S, A>) contextBuilderClass.getConstructor(Properties.class).newInstance(properties);
137       } catch (NoSuchMethodException e) {
138          checkArgument(sync != null, "please configure sync class for " + contextBuilderClass);
139          checkArgument(async != null, "please configure async class for " + contextBuilderClass);
140          return (RestContextBuilder<S, A>) contextBuilderClass.getConstructor(sync.getClass(), async.getClass(),
141                Properties.class).newInstance(sync, async, properties);
142       }
143    }

145    @SuppressWarnings("unchecked")
146    public static Class<PropertiesBuilder> resolvePropertiesBuilderClass(String providerName, Properties props)
147          throws ClassNotFoundException, InstantiationException, IllegalAccessException, InvocationTargetException,
148          NoSuchMethodException {
149       String propertiesBuilderClassName = props.getProperty(providerName + ".propertiesbuilder", null);
150       if (propertiesBuilderClassName != null) {
151          return (Class<PropertiesBuilder>) ClassLoadingUtils.loadClass(Providers.class, propertiesBuilderClassName);
152       } else {
153          return PropertiesBuilder.class;
154       }
