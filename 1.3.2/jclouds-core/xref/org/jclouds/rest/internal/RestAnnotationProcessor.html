198    static class GetAnnotationsForMethodParameterIndex implements Function<Integer, Set<Annotation>> {
199       private final Method method;
200       private final Class<?> clazz;
201 
202       protected GetAnnotationsForMethodParameterIndex(Method method, Class<?> clazz) {
203          this.method = method;
204          this.clazz = clazz;
205       }
206 
207       public Set<Annotation> apply(final Integer index) {
208          return ImmutableSet.<Annotation> copyOf(filter(ImmutableList.copyOf(method.getParameterAnnotations()[index]),
209                new Predicate<Annotation>() {
210                   public boolean apply(Annotation input) {
211                      return input.annotationType().equals(clazz);
212                   }
213                }));
214       }

262          Method method, HttpRequest request) {

277    public static Function<HttpResponse, ?> getTransformerForMethod(Method method, Injector injector) {
278       Function<HttpResponse, ?> transformer;
279       if (method.isAnnotationPresent(SelectJson.class)) {
280          Type returnVal = getReturnTypeForMethod(method);
281          if (method.isAnnotationPresent(OnlyElement.class))
282             returnVal = Types.newParameterizedType(Set.class, returnVal);
283          transformer = new ParseFirstJsonValueNamed(injector.getInstance(GsonWrapper.class),
284                TypeLiteral.get(returnVal), method.getAnnotation(SelectJson.class).value());
285          if (method.isAnnotationPresent(OnlyElement.class))
286             transformer = Functions.compose(new OnlyElementOrNull(), transformer);
287       } else {
288          transformer = injector.getInstance(getParserOrThrowException(method));
289       }
290       return transformer;
291    }

333    public static class MethodKey {
334 
335       @Override
336       public int hashCode() {
337          final int prime = 31;
338          int result = 1;
339          result = prime * result + ((declaringPackage == null) ? 0 : declaringPackage.hashCode());
340          result = prime * result + ((name == null) ? 0 : name.hashCode());
341          result = prime * result + parametersTypeHashCode;
342          return result;
343       }
344 
345       @Override
346       public boolean equals(Object obj) {
347          if (this == obj)
348             return true;
349          if (obj == null)
350             return false;
351          if (getClass() != obj.getClass())
352             return false;
353          MethodKey other = (MethodKey) obj;
354          if (declaringPackage == null) {
355             if (other.declaringPackage != null)
356                return false;
357          } else if (!declaringPackage.equals(other.declaringPackage))
358             return false;
359          if (name == null) {
360             if (other.name != null)
361                return false;
362          } else if (!name.equals(other.name))
363             return false;
364          if (parametersTypeHashCode != other.parametersTypeHashCode)
365             return false;
366          return true;
367       }
368 
369       private final String name;
370       private final int parametersTypeHashCode;
371       private final Package declaringPackage;
372 
373       public MethodKey(Method method) {
374          this.name = method.getName();
375          this.declaringPackage = method.getDeclaringClass().getPackage();
376          int parametersTypeHashCode = 0;
377          for (Class<?> param : method.getParameterTypes())
378             parametersTypeHashCode += param.hashCode();
379          this.parametersTypeHashCode = parametersTypeHashCode;
380       }

404    public GeneratedHttpRequest<T> createRequest(Method method, Object... args) {
405       try {
406          inputParamValidator.validateMethodParametersOrThrow(method, args);
407          ClassMethodArgs cma = logger.isTraceEnabled() ? new ClassMethodArgs(method.getDeclaringClass(), method, args)
408                : null;
409 
410          URI endpoint = callerEndpoint;
411          try {
412             if (endpoint == null) {
413                endpoint = getEndpointFor(method, args, injector);
414                logger.trace("using endpoint %s for %s", endpoint, cma);
415             } else {
416                logger.trace("using endpoint %s from caller %s for %s", caller, endpoint, cma);
417             }
418          } catch (IllegalStateException e) {
419             logger.trace("looking up default endpoint for %s", cma);
420             endpoint = injector.getInstance(Key.get(URI.class, org.jclouds.location.Provider.class));
421             logger.trace("using default endpoint %s for %s", endpoint, cma);
422          }
423          GeneratedHttpRequest.Builder<T> requestBuilder;
424          HttpRequest r = RestAnnotationProcessor.findHttpRequestInArgs(args);
425          if (r != null) {
426             requestBuilder = GeneratedHttpRequest.Builder.fromRequest(r);
427             endpoint = r.getEndpoint();
428          } else {
429             requestBuilder = GeneratedHttpRequest.requestBuilder();
430             requestBuilder.method(getHttpMethodOrConstantOrThrowException(method));
431          }
432 
433          requestBuilder.declaring(declaring).javaMethod(method).args(args).skips(skips);
434          requestBuilder.filters(getFiltersIfAnnotated(method));
435 
436          UriBuilder builder = uriBuilderProvider.get().uri(endpoint);
437 
438          Multimap<String, String> tokenValues = LinkedHashMultimap.create();
439 
440          tokenValues.put(Constants.PROPERTY_API_VERSION, apiVersion);
441          tokenValues.put(Constants.PROPERTY_BUILD_VERSION, buildVersion);
442 
443          tokenValues.putAll(addPathAndGetTokens(declaring, method, args, builder));
444 
445          Multimap<String, String> formParams = addFormParams(tokenValues.entries(), method, args);
446          Multimap<String, String> queryParams = addQueryParams(tokenValues.entries(), method, args);
447          Multimap<String, String> matrixParams = addMatrixParams(tokenValues.entries(), method, args);
448          Multimap<String, String> headers = buildHeaders(tokenValues.entries(), method, args);
449          if (r != null)
450             headers.putAll(r.getHeaders());
451 
452          if (shouldAddHostHeader(method)) {
453             StringBuilder hostHeader = new StringBuilder(endpoint.getHost());
454             if (endpoint.getPort() != -1)
455                hostHeader.append(":").append(endpoint.getPort());
456             headers.put(HOST, hostHeader.toString());
457          }
458 
459          Payload payload = null;
460          HttpRequestOptions options = findOptionsIn(method, args);
461          if (options != null) {
462             injector.injectMembers(options);// TODO test case
463             for (Entry<String, String> header : options.buildRequestHeaders().entries()) {
464                headers.put(header.getKey(), Strings2.replaceTokens(header.getValue(), tokenValues.entries()));
465             }
466             for (Entry<String, String> matrix : options.buildMatrixParameters().entries()) {
467                matrixParams.put(matrix.getKey(), Strings2.replaceTokens(matrix.getValue(), tokenValues.entries()));
468             }
469             for (Entry<String, String> query : options.buildQueryParameters().entries()) {
470                queryParams.put(query.getKey(), Strings2.replaceTokens(query.getValue(), tokenValues.entries()));
471             }
472             for (Entry<String, String> form : options.buildFormParameters().entries()) {
473                formParams.put(form.getKey(), Strings2.replaceTokens(form.getValue(), tokenValues.entries()));
474             }
475 
476             String pathSuffix = options.buildPathSuffix();
477             if (pathSuffix != null) {
478                builder.path(pathSuffix);
479             }
480             String stringPayload = options.buildStringPayload();
481             if (stringPayload != null)
482                payload = Payloads.newStringPayload(stringPayload);
483          }
484 
485          if (matrixParams.size() > 0) {
486             for (String key : matrixParams.keySet())
487                builder.matrixParam(key, Lists.newArrayList(matrixParams.get(key)).toArray());
488          }
489 
490          if (queryParams.size() > 0) {
491             builder.replaceQuery(ModifyRequest.makeQueryLine(queryParams, null, skips));
492          }
493 
494          requestBuilder.headers(filterOutContentHeaders(headers));
495 
496          try {
497             requestBuilder.endpoint(builder.buildFromEncodedMap(Maps2.convertUnsafe(tokenValues)));
498          } catch (IllegalArgumentException e) {
499             throw new IllegalStateException(e);
500          } catch (UriBuilderException e) {
501             throw new IllegalStateException(e);
502          }
503 
504          if (payload == null)
505             payload = findPayloadInArgs(args);
506          List<? extends Part> parts = getParts(method, args, concat(tokenValues.entries(), formParams.entries()));
507          if (parts.size() > 0) {
508             if (formParams.size() > 0) {
509                parts = newLinkedList(concat(transform(formParams.entries(), ENTRY_TO_PART), parts));
510             }
511             payload = new MultipartForm(BOUNDARY, parts);
512          } else if (formParams.size() > 0) {
513             payload = Payloads.newUrlEncodedFormPayload(formParams, skips);
514          } else if (headers.containsKey(CONTENT_TYPE)) {
515             if (payload == null)
516                payload = Payloads.newByteArrayPayload(new byte[] {});
517             payload.getContentMetadata().setContentType(Iterables.get(headers.get(CONTENT_TYPE), 0));
518          }
519          if (payload != null) {
520             requestBuilder.payload(payload);
521          }
522          GeneratedHttpRequest<T> request = requestBuilder.build();
523 
524          org.jclouds.rest.MapBinder mapBinder = getMapPayloadBinderOrNull(method, args);
525          if (mapBinder != null) {
526             Map<String, String> mapParams = buildPostParams(method, args);
527             if (method.isAnnotationPresent(PayloadParams.class)) {
528                PayloadParams params = method.getAnnotation(PayloadParams.class);
529                addMapPayload(mapParams, params, headers.entries());
530             }
531             request = mapBinder.bindToRequest(request, mapParams);
532          } else {
533             request = decorateRequest(request);
534          }
535 
536          if (request.getPayload() != null)
537             request.getPayload().getContentMetadata().setPropertiesFromHttpHeaders(headers);
538          utils.checkRequestHasRequiredProperties(request);
539          return request;
540       } catch (ExecutionException e) {
541          Throwables.propagate(e);
542          return null;
543       }
544    }

546    public static Multimap<String, String> filterOutContentHeaders(Multimap<String, String> headers) {
547       // TODO make a filter like {@link Maps.filterKeys} instead of this
548       ImmutableMultimap.Builder<String, String> headersBuilder = ImmutableMultimap.builder();
549       // http message usually comes in as a null key header, let's filter it
550       // out.
551       for (String header : Iterables.filter(headers.keySet(), Predicates.notNull())) {
552          if (!ContentMetadata.HTTP_HEADERS.contains(header)) {
553             headersBuilder.putAll(header, headers.get(header));
554          }
555       }
556       return headersBuilder.build();
557    }

789       ResponseParser annotation = method.getAnnotation(ResponseParser.class);

791          if (method.getReturnType().equals(void.class)
792                || TypeLiteral.get(method.getGenericReturnType()).equals(futureVoidLiteral)) {

795                || TypeLiteral.get(method.getGenericReturnType()).equals(futureBooleanLiteral)) {

802             return Key.get((Class) IdentityFunction.class);

804             return getJsonParserKeyForMethod(method);
805          } else if (getAcceptHeadersOrNull(method).contains(MediaType.APPLICATION_XML)
806                 || method.isAnnotationPresent(JAXBResponseParser.class)) {
807             return getJAXBParserKeyForMethod(method);
808          } else if (method.getReturnType().equals(String.class)
809                || TypeLiteral.get(method.getGenericReturnType()).equals(futureStringLiteral)) {
810             return Key.get(ReturnStringIf2xx.class);
811          } else if (method.getReturnType().equals(URI.class)
812                || TypeLiteral.get(method.getGenericReturnType()).equals(futureURILiteral)) {
813             return Key.get(ParseURIFromListOrLocationHeaderIf20x.class);
814          } else {
815             throw new IllegalStateException("You must specify a ResponseParser annotation on: " + method.toString());
816          }
817       }
818       return Key.get(annotation.value());

934    public GeneratedHttpRequest<T> decorateRequest(GeneratedHttpRequest<T> request) throws NegativeArraySizeException,
935             ExecutionException {
936       OUTER: for (Entry<Integer, Set<Annotation>> entry : concat(//
937                filterValues(methodToIndexOfParamToBinderParamAnnotation.get(request.getJavaMethod()).asMap(), notEmpty)
938                         .entrySet(), //
939                filterValues(methodToIndexOfParamToWrapWithAnnotation.get(request.getJavaMethod()).asMap(), notEmpty)
940                         .entrySet())) {
941          boolean shouldBreak = false;
942          Annotation annotation = Iterables.get(entry.getValue(), 0);
943          Binder binder;
944          if (annotation instanceof BinderParam)
945             binder = injector.getInstance(BinderParam.class.cast(annotation).value());
946          else
947             binder = injector.getInstance(BindToJsonPayloadWrappedWith.Factory.class).create(
948                      WrapWith.class.cast(annotation).value());
949          if (request.getArgs().size() >= entry.getKey() + 1 && request.getArgs().get(entry.getKey()) != null) {
950             Object input;
951             Class<?> parameterType = request.getJavaMethod().getParameterTypes()[entry.getKey()];
952             Class<? extends Object> argType = request.getArgs().get(entry.getKey()).getClass();
953             if (!argType.isArray() && request.getJavaMethod().isVarArgs() && parameterType.isArray()) {
954                int arrayLength = request.getArgs().size() - request.getJavaMethod().getParameterTypes().length + 1;
955                if (arrayLength == 0)
956                   break OUTER;
957                input = (Object[]) Array.newInstance(request.getArgs().get(entry.getKey()).getClass(), arrayLength);
958                System.arraycopy(request.getArgs().toArray(), entry.getKey(), input, 0, arrayLength);
959                shouldBreak = true;
960             } else if (argType.isArray() && request.getJavaMethod().isVarArgs() && parameterType.isArray()) {
961                input = request.getArgs().get(entry.getKey());
962             } else {
963                input = request.getArgs().get(entry.getKey());
964                if (input.getClass().isArray()) {
965                   Object[] payloadArray = (Object[]) input;
966                   input = payloadArray.length > 0 ? payloadArray[0] : null;
967                }
968             }
969             if (input != null) {
970                request = binder.bindToRequest(request, input);
971             }
972             if (shouldBreak)
973                break OUTER;
974          } else {
975             // either arg is null, or request.getArgs().size() < entry.getKey() + 1
976             // in either case, we require that null be allowed
977             // (first, however, let's make sure we have enough args on the actual method)
978             if (entry.getKey() >= request.getJavaMethod().getParameterAnnotations().length) {
979                // not known whether this happens
980                throw new IllegalArgumentException("Argument index " + (entry.getKey() + 1)
981                         + " is out of bounds for method " + request.getJavaMethod());
982             }
983 
984             if (request.getJavaMethod().isVarArgs()
985                      && entry.getKey() + 1 == request.getJavaMethod().getParameterTypes().length)
986                // allow null/missing for var args
987                continue OUTER;
988 
989             Annotation[] annotations = request.getJavaMethod().getParameterAnnotations()[entry.getKey()];
990             for (Annotation a : annotations) {
991                if (Nullable.class.isAssignableFrom(a.annotationType()))
992                   continue OUTER;
993             }
994             Preconditions.checkNotNull(null, request.getJavaMethod().getName() + " parameter " + (entry.getKey() + 1));
995          }
996       }
997 
998       return request;
999    }
